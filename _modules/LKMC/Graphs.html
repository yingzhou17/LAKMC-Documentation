[docs]def getIsomorphism(params, defectVolume1, lattice1, defectVolume2, lattice2):
    """
    Get isomorphism between two graphs.

    Parameters
    ----------
    params : `Input.InputParams`
        The input parameters
    defectVolume1 : `Defects.DefectVolume`
        The first defect volume (graph)
    lattice1 : `Lattice.Lattice`
        The Lattice containing the first defect volume
    defectVolume2 : `Defects.DefectVolume`
        The second defect volume (graph)
    lattice2 : `Lattice.Lattice`
        The Lattice containing the second defect volume

    Returns
    -------
    ismLst1 : `numpy.ndarray`
        The isomorphism list for the first volume
    ismLst2 : `numpy.ndarray`
        The isomorphism list for the second volume

    Raises
    ------
    RuntimeError
        If the isomorphism could not be computed or if the volumes have
        different lengths

    """
    defVolLen1 = len(defectVolume1)
    defVolLen2 = len(defectVolume2)
    if defVolLen1 != defVolLen2:
        raise RuntimeError("Graphs cannot be isomorphic if they have different lengths (%d != %d)" % (defVolLen1, defVolLen2))

    nvolatoms = defVolLen1

    # make volume position and specie arrays for volume 1
    defVolPos1 = np.empty(3 * nvolatoms, np.float64)
    defVolSpecie1 = np.empty(nvolatoms, np.int32)
    c_graphs.makeDefectVolumePosAndSpecie(defectVolume1, defVolPos1, defVolSpecie1, lattice1.pos, lattice1.specie)

    # make volume position and specie arrays for volume 1
    defVolPos2 = np.empty(3 * nvolatoms, np.float64)
    defVolSpecie2 = np.empty(nvolatoms, np.int32)
    c_graphs.makeDefectVolumePosAndSpecie(defectVolume2, defVolPos2, defVolSpecie2, lattice2.pos, lattice2.specie)

    NSpecies = len(lattice1.specieList)
    #TODO: what happens if species are different?

    cellDims3 = np.empty(3, np.float64)
    cellDims3[0] = lattice1.cellDims[0]
    cellDims3[1] = lattice1.cellDims[4]
    cellDims3[2] = lattice1.cellDims[8]

    ismList1 = np.empty(defVolLen1, np.int32)
    ismList2 = np.empty(defVolLen2, np.int32)
    status = nauty_iface.findIsomorphism(nvolatoms, defVolPos1, defVolSpecie1, defVolPos2, defVolSpecie2, NSpecies,
                                         params.graphRadius, cellDims3, Globals.PBC, ismList1, ismList2)

    if status:
        raise RuntimeError("Could not compute isomorphism (%d)" % status)

    return ismList1, ismList2



[docs]def getHashKeyForAVolume(params, volumeAtoms, lattice):
    """
    Calculate the hash key of a defect volume.

    Parameters
    ----------
    params : `Input.InputParams`
        The input parameters
    volumeAtoms : `numpy.ndarray`
        The indexes of the atoms in the defect volume (graph)
    lattice : `Lattice.Lattice`
        The Lattice object containing the defect volume

    Returns
    -------
    hashKey : str
        The hash key of the defect volume

    Raises
    ------
    RuntimeError
        If the hash key could not be calculated

    """
    numAtoms = len(volumeAtoms)
    numSpecies = len(lattice.specieList)

    cellDims3 = np.empty(3, np.float64)
    cellDims3[0] = lattice.cellDims[0]
    cellDims3[1] = lattice.cellDims[4]
    cellDims3[2] = lattice.cellDims[8]

    # make volume position and specie arrays
    volumePos = np.empty(3 * numAtoms, np.float64)
    volumeSpecie = np.empty(numAtoms, np.int32)
    c_graphs.makeDefectVolumePosAndSpecie(volumeAtoms, volumePos, volumeSpecie, lattice.pos, lattice.specie)

    # compute the hash key
    hashKeyChar = np.zeros(64, dtype=np.character)
    status = nauty_iface.getGraphKey(numAtoms, volumePos, volumeSpecie, numSpecies, params.graphRadius, cellDims3,
                                     Globals.PBC, hashKeyChar)

    if status:
        raise RuntimeError("Error computing hash key (%d)" % status)

    # convert hash key to str
    hashKey = "".join(hashKeyChar)

    return hashKey

    consHashKey = getHashKeyForAVolume(params, np.arange(volTransObj.NAtomsVolume, dtype=np.int32), consLattice)
    if key != consHashKey:
        consLattice = prepareLatticeToBeTransf(params, consLattice, np.arange(volTransObj.NAtomsVolume, dtype=np.int32), True, noOfCoords=3)
        consHashKey = getHashKeyForAVolume(params, np.arange(volTransObj.NAtomsVolume, dtype=np.int32), consLattice)

            log(__name__,"Inconsistent hashkeys: assigned key %s, calculated key %s " % (key, consHashKey),2,0)

    return True

[docs]def prepareTheMatrix(params, defectVolume1, lattice1In, pbcInflVol1, defectVolume2, lattice2In, pbcInflVol2, noOfCoords=3):
    """
    Construct the transformation matrix and other data required for reuse.

    Parameters
    ----------
    params : `Input.InputParams`
        The input parameters
    defectVolume1 : `numpy.ndarray`
        The indexes of the atoms in first volume
    lattice1In : `Lattice.Lattice`
        The Lattice object containing the first volume
    pbcInflVol1 : bool
        Expand periodic boundaries for the first volume (in case it crosses them)
    defectVolume2 : `numpy.ndarray`
        The indexes of the atoms in second volume
    lattice2In : `Lattice.Lattice`
        The Lattice object containing the second volume
    pbcInflVol2 : bool
        Expand periodic boundaries for the second volume (in case it crosses them)
    noOfCoords : int, optional
        Dimension of the system. Defaults to 3.

    Returns
    -------
    trnsfMatrix : `numpy.ndarray[noOfCoords,noOfCoords]`
        The transformation matrix
    lattice1 : `Lattice.Lattice`
        The first Lattice with periodic boundaries removed if required
    atLst1 : `numpy.ndarray`
        The list of atoms from the isomorphism calculation (first volume)
    lattice2 : `Lattice.Lattice`
        The second Lattice with periodic boundaries removed if required
    atLst2 : `numpy.ndarray`
        The list of atoms from the isomorphism calculation (second volume)
    cntr1 : `numpy.ndarray`
        The centre of mass of the first volume
    cntr2 : `numpy.ndarray`
        The centre of mass of the second volume

    """
    # prepare the lattices
    lattice1 = prepareLatticeToBeTransf(params, lattice1In, defectVolume1, pbcInflVol1, noOfCoords=noOfCoords)
    lattice2 = prepareLatticeToBeTransf(params, lattice2In, defectVolume2, pbcInflVol2, noOfCoords=noOfCoords)

    # centre of mass of defect volumes
    cntr1 = np.zeros(noOfCoords, np.float64)
    cntr2 = np.zeros(noOfCoords, np.float64)
    c_graphs.getCenterOfMassOfAlist(defectVolume1, lattice1.pos, noOfCoords, cntr1)
    c_graphs.getCenterOfMassOfAlist(defectVolume2, lattice2.pos, noOfCoords, cntr2)

    # isomorphism calculation
    atLst1, atLst2, trnsfMatrix = getIsomorphismBetweenTwoGraphs(defectVolume1, lattice1, defectVolume2, lattice2, params, cntr1=cntr1, cntr2=cntr2)

    return trnsfMatrix, lattice1, atLst1, lattice2, atLst2, cntr1, cntr2



[docs]def removePBCsEffectOnVolume(defectVolume, lattice, latticeOrig, inclRad, noOfCoords=3, getDiffVector=0, cntrOrig=None):
    """
    Modifies atoms positions of a defect volume to make sure that they are in
    the same region.

    """

    defectVolumeLen = len(defectVolume)

    # if defect volume contains only one atom or none
    if defectVolumeLen <= 1:
        return lattice

    # creates a list of atoms that are separated more than a defect inclusion
    #    radius form the first atom in the defect volume atoms list
    sameDefectVolumeArr = np.zeros(defectVolumeLen, np.int32)

    c_graphs.checkSepOnDefectVolumeAtoms(defectVolume, inclRad, lattice.pos,
                                         noOfCoords, sameDefectVolumeArr)

    # calculates center positions
    cntr = np.zeros(noOfCoords, np.float64)

    c_graphs.getCenterOfMassOfAlistExt(defectVolume, lattice.pos, noOfCoords,
                                       sameDefectVolumeArr, cntr)

    # modifies positions of the atoms that are separated from the volume
    c_graphs.modifyVolumeAtoms(defectVolume, lattice.pos, noOfCoords,
                               sameDefectVolumeArr, cntr, lattice.cellDims)

    if cntrOrig is not None:
        for j in range(noOfCoords):
            diff = cntr[j] - cntrOrig[j]
            if math.fabs(diff) > lattice.cellDims[4*j]/2:
                modifyNum = math.copysign(1.0, diff) * lattice.cellDims[4*j]
                for i in range(defectVolumeLen):
                    lattice.pos[i*noOfCoords+j] -= modifyNum

[docs]def getCenterOfMassOfAlist(defectVolume, lattice, noOfCoords=3):
    result =  np.zeros(noOfCoords, np.float64)

    NAtoms = len(defectVolume)
    NAtomsf = float(NAtoms)

    for i in range(NAtoms):
        atom = defectVolume[i]

        for j in range(noOfCoords):
            result[j] += lattice.pos[noOfCoords*atom+j]

    for j in range(noOfCoords):
        result[j] = result[j] / NAtomsf

    return result

[docs]def reuseTransition(transformMatrix, atomsList1, defectVolume1, lattice1, cntr1,
        saddle1, final1, atomsList2, defectVolume2, lattice2, cntr2, noOfCoords=3):

    if saddle1 is not None:
        matrixToSaddle = getDiffMatrix2StatesVolume(defectVolume1, lattice1,
                                                    saddle1, atomsList1, atomsList2,
                                                    noOfCoords=noOfCoords)

        matrixToSaddleTransf = np.dot(transformMatrix, matrixToSaddle)

        saddle2 = applyTransformMatrix(atomsList2, defectVolume2, lattice2,
                                       matrixToSaddleTransf, noOfCoords=noOfCoords)
    else:
        saddle2 = None

    if final1 is not None:
        matrixToFinal = getDiffMatrix2StatesVolume(defectVolume1, lattice1,
                                                   final1, atomsList1, atomsList2,
                                                   noOfCoords=noOfCoords)

        matrixToFinalTransf = np.dot(transformMatrix, matrixToFinal)

        final2 = applyTransformMatrix(atomsList2, defectVolume2, lattice2,
                                      matrixToFinalTransf, noOfCoords=noOfCoords)

    else:
        final2 = None

    return saddle2, final2






[docs]def getSepSqBetweenTwoAtoms(lattice, atom1, atom2, noOfCoords=3):
    dist = 0.0

    for i in range(noOfCoords):
        dist += math.pow(lattice.pos[noOfCoords*atom1+i] - lattice.pos[noOfCoords*atom2+i], 2)

    return dist

[docs]def removePBCsEffectOnAtomPos(defectVolume, lattice, latticeOrig, inclRad, noOfCoords=3):

    inclRadSq = inclRad * inclRad
    defectVolumeLen = len(defectVolume)

    defectVolumeIn = []
    for i in range(defectVolumeLen):
        atom = defectVolume[i]
        defectVolumeIn.append(atom)

    tempDefectVolume = []
    xcontinue = True

    while xcontinue:
        xcontinue = False

        if len(tempDefectVolume) > 0:
            # lets find atoms that are within inclRadius of newly built defectVolume
            defVolTmpLen = len(tempDefectVolume)
            defVolInLen = len(defectVolumeIn)

            xcontinue = False

            for i in range(defVolTmpLen):
                breakLoop = False

                for j in range(defVolInLen):
                    atom1 = tempDefectVolume[i]
                    atom2 = defectVolumeIn[j]

                    dist = getSepSqBetweenTwoAtoms(lattice, atom1, atom2)

                    if dist <= inclRadSq:
                        tempDefectVolume.append(atom2)
                        defectVolumeIn.pop(j)
                        defVolInLen -= 1

                        xcontinue = True
                        breakLoop = True
                        break

                if breakLoop:
                    break

        else:
            atom = defectVolumeIn[0]
            defectVolumeIn.pop(0)
            tempDefectVolume.append(atom)
            xcontinue = True

    cntr = getCenterOfMassOfAlist(tempDefectVolume, lattice, noOfCoords=noOfCoords)

    defVolInLen = len(defectVolumeIn)

    for i in range(defVolInLen):
        for j in range(noOfCoords):
            if math.fabs(lattice.pos[defectVolumeIn[i]*noOfCoords+j] - cntr[j]) > (lattice.cellDims[4*j]/2.0):
                if lattice.pos[defectVolumeIn[i]*noOfCoords+j] > cntr[j]:
                    lattice.pos[defectVolumeIn[i]*noOfCoords+j] -= lattice.cellDims[4*j]
                else:
                    lattice.pos[defectVolumeIn[i]*noOfCoords+j] += lattice.cellDims[4*j]

    diffVector = np.empty(noOfCoords*defectVolumeLen, np.float64)

    for i in range(defectVolumeLen):
        atom = defectVolume[i]
        for j in range(noOfCoords):
            diffVector[i*noOfCoords+j] = latticeOrig.pos[atom*noOfCoords+j] - lattice.pos[atom*noOfCoords+j]

    return lattice, diffVector

[docs]def convertOldVolumeFiles(params):
    """
    Convert old defect volume files to use hash key from new nauty interface.

    Old files are backed up...

    """
    from . import Lattice

    log(__name__, "Checking if there are any old defect volume files that need to be converted", 2, 0)

    if not os.path.isdir(params.volumeDirectory):
        log(__name__, "The volume database directory does not exist; skipping", 2, 1)
        return

    import glob
    volFiles = glob.glob(os.path.join(params.volumeDirectory, "*.vol"))

    if not len(volFiles):
        log(__name__, "No volume files exists; skipping", 2, 1)
        return

    log(__name__, "Found {0} volume files to check...".format(len(volFiles)), 2, 1)

    import datetime
    backupDate = datetime.datetime.now().strftime("%y%m%d-%H%M%S")

    updateCount = 0
    for count, oldfn in enumerate(volFiles):
        bn = os.path.basename(oldfn)
        log(__name__, "Checking old volume file {0}: '{1}'".format(count, bn), 3, 2)

        oldHash = os.path.splitext(bn)[0]
        log(__name__, "Old hash key is: '{0}' (LEN={1})".format(oldHash, len(oldHash)), 3, 3)

        # read in positions of graph atoms
        f = open(oldfn)
        line = f.readline()
        array = line.split()
        natoms = int(array[0])
        log(__name__, "{0} atoms in the volume".format(natoms), 4, 4)

        lattice = Lattice.Lattice(0)

        for _ in xrange(natoms):
            line = f.readline()
            array = line.split()
            pos = map(float, array[1:])
            assert len(pos) == 3
            sym = array[0]
            lattice.addAtom(sym, pos, 0.0)

        # get new graph key
        volatoms = np.arange(natoms, dtype=np.int32)
        newHash = getHashKeyForAVolume(params, volatoms, lattice)
        log(__name__, "New hash key is: '{0}' (LEN={1})".format(newHash, len(newHash)), 3, 3)

        if oldHash == newHash:
            log(__name__, "Old and new keys are the same; skipping update", 3, 4)

        else:
            # backup the old file
            backupName = oldfn + ".backup-%s" % backupDate
            log(__name__, "Backing up old volume file as: '{0}'".format(os.path.basename(backupName)), 3, 4)
            shutil.copyfile(oldfn, backupName)

            # check new file does not already exist
            newfn = os.path.join(params.volumeDirectory, "%s.vol" % newHash)
            if os.path.exists(newfn):
                pass

            else:
                # rename old volume file to new
                log(__name__, "Renaming volume file to: '{0}'".format(os.path.basename(newfn)), 3, 4)
                os.rename(oldfn, newfn)

                updateCount += 1

    if updateCount:
        log(__name__, "Successfully updated {0} old volume files".format(updateCount), 2, 1)
