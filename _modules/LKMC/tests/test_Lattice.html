[docs]    def test_getMinSepList(self):
        """
        Lattice getMinSepList
        
        """
        # input position
        inputPos = np.asarray([5,8,4], dtype=np.float64)
        
        # atom list
        atomList = np.asarray([0, 1, 5, 3], dtype=np.int32)
        
        # call with full list
        minSep, minSepIndex = self.lattice.getMinSepList(inputPos, atomList)
        self.assertEqual(minSepIndex, 3)
        self.assertAlmostEqual(minSep, 5.477225575051661)
        
        # call with partial list
        minSep, minSepIndex = self.lattice.getMinSepList(inputPos, atomList, atomListSize=3)
        self.assertEqual(minSepIndex, 1)
        self.assertAlmostEqual(minSep, 8.94427190999916)
        
        # max spacing smaller than closest separation
        minSep, minSepIndex = self.lattice.getMinSepList(inputPos, atomList, maxSpacing=3.0)
        self.assertIsNone(minSepIndex)
        self.assertIsNone(minSep)
        minSep, minSepIndex = self.lattice.getMinSepList(inputPos, atomList, maxSpacing=5.45)
        self.assertIsNone(minSepIndex)
        self.assertIsNone(minSep)
        
        # check raises exception
        with self.assertRaises(ValueError):
            self.lattice.getMinSepList(inputPos, atomList, atomListSize=-1)
        with self.assertRaises(ValueError):
            self.lattice.getMinSepList(inputPos, atomList, atomListSize=8)

    

[docs]    def test_getIndicesFromPositions(self):
        """
        Lattice getIndicesFromPositions
        
        """
        Globals.PBC[:] = 1
        self.lattice.minPos[:] = 0.0
        self.lattice.maxPos[:] = 100.0
        
        # input positions (0-0.05, -, 6-0.05, 4-0.15, 2-0.01)
        inputPos = np.asarray([0,0,0.05, 1,1,1, 14.95,15,15, 8,8,8.15, 3,3,0.01], dtype=np.float64)
        
        # call
        indices = self.lattice.getIndicesFromPositions(inputPos)
        self.assertEqual(len(indices), len(inputPos) / 3)
        self.assertEqual(indices[0], 0)
        self.assertEqual(indices[1], -1)
        self.assertEqual(indices[2], 6)
        self.assertEqual(indices[3], -1)
        self.assertEqual(indices[4], 2)
        
        # call again with a large maxSep
        indices = self.lattice.getIndicesFromPositions(inputPos, maxSep=0.2)
        self.assertEqual(len(indices), len(inputPos) / 3)
        self.assertEqual(indices[0], 0)
        self.assertEqual(indices[1], -1)
        self.assertEqual(indices[2], 6)
        self.assertEqual(indices[3], 4)
        self.assertEqual(indices[4], 2)
        
        # check raises error
        with self.assertRaises(ValueError):
            self.lattice.getIndicesFromPositions(inputPos, maxSep=-2)

[docs]    def test_findNeighbours(self):
        """
        Lattice find atoms neighbours
        
        """
        nebs, nebSeps = self.lattice.findNeighbours(0, 5.0, sort=False)
        self.assertTrue(np.allclose(nebs, np.asarray([5, 2], dtype=np.int32)))
        self.assertTrue(np.allclose(nebSeps, np.asarray([1.73205081,  4.24264069])))
        
        nebs, nebSeps = self.lattice.findNeighbours(0, 6.0, sort=False)
        self.assertTrue(np.allclose(nebs, np.asarray([5, 1, 2], dtype=np.int32)))
        self.assertTrue(np.allclose(nebSeps, np.asarray([1.73205081,  5.0, 4.24264069])))
        
        nebs, nebSeps = self.lattice.findNeighbours(0, 6.0, sort=True)
        self.assertTrue(np.allclose(nebs, np.asarray([5, 2, 1], dtype=np.int32)))
        self.assertTrue(np.allclose(nebSeps, np.asarray([1.73205081, 4.24264069,  5.0])))
        
        nebs, nebSeps = self.lattice.findNeighbours(2, 6.0, sort=False)
        self.assertTrue(np.allclose(nebs, np.asarray([5, 0, 1], dtype=np.int32)))
        self.assertTrue(np.allclose(nebSeps, np.asarray([5.74456265,  4.24264069,  3.60555128])))
        
        nebs, nebSeps = self.lattice.findNeighbours(2, 6.0, sort=True)
        self.assertTrue(np.allclose(nebs, np.asarray([1, 0, 5], dtype=np.int32)))
        self.assertTrue(np.allclose(nebSeps, np.asarray([3.60555128,  4.24264069,  5.74456265])))
        
        self.assertRaises(IndexError, self.lattice.findNeighbours, -1, 6.0)
        self.assertRaises(IndexError, self.lattice.findNeighbours, 20, 6.0)
