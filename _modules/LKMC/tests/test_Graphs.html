[docs]    def setUp(self):
        fn = self.pathToInputFile()
        Input.readGlobals(fn)
        self.params = Input.getLKMCParams(inputFile=fn)
        self.params.graphRadius = 2.5
        
        baselat = Lattice.Lattice(0)
        baselat.setCellDims([100,0,0,0,100,0,0,0,100])
        sep = 2.0
        for i in xrange(3):
            rx = i * sep
            for j in xrange(3):
                ry = j * sep
                for k in xrange(3):
                    rz = k * sep 
                    baselat.addAtom("Pu", (rx, ry, rz), 0.0)
        
        self.volAtoms = np.arange(baselat.NAtoms, dtype=np.int32)
        
        self.lat = []
        
        # these lattices have a single Ga (first two are isomorphic, third isn't)
        self.lat.append(copy.deepcopy(baselat))
        self.lat[0].changeAtomSpecie(0, "Ga")
        
        self.lat.append(copy.deepcopy(baselat))
        self.lat[1].changeAtomSpecie(26, "Ga")
        
        self.lat.append(copy.deepcopy(baselat))
        self.lat[2].changeAtomSpecie(1, "Ga")
        
        # these lattices have two Ga on opposite corners (except for last one)
        self.lat.append(copy.deepcopy(baselat))
        self.lat[3].changeAtomSpecie(0, "Ga")
        self.lat[3].changeAtomSpecie(26, "Ga")
        
        self.lat.append(copy.deepcopy(baselat))
        self.lat[4].changeAtomSpecie(2, "Ga")
        self.lat[4].changeAtomSpecie(24, "Ga")
        
        self.lat.append(copy.deepcopy(baselat))
        self.lat[5].changeAtomSpecie(6, "Ga")
        self.lat[5].changeAtomSpecie(20, "Ga")
        
        self.lat.append(copy.deepcopy(baselat))
        self.lat[6].changeAtomSpecie(18, "Ga")
        self.lat[6].changeAtomSpecie(8, "Ga")
        
        self.lat.append(copy.deepcopy(baselat))
        self.lat[7].changeAtomSpecie(18, "Ga")
        self.lat[7].changeAtomSpecie(11, "Ga")
        
        # these lattices have one vacancy (first two isomorphic, third isn't)
        self.lat.append(copy.deepcopy(baselat))
        self.lat[8].removeAtom(0)
        
        self.lat.append(copy.deepcopy(baselat))
        self.lat[9].removeAtom(26)
        
        self.lat.append(copy.deepcopy(baselat))
        self.lat[10].removeAtom(13)

[docs]    def test_makeDefectVolumePosAndSpecie(self):
        """
        Graphs make volume pos and specie
        
        """
        volAtoms = self.volAtoms
        volPos = np.empty(3 * len(volAtoms), np.float64)
        volSpec = np.empty(len(volAtoms), np.int32)
        status =  _graphs.makeDefectVolumePosAndSpecie(self.volAtoms, volPos, volSpec, self.lat[0].pos, self.lat[0].specie)
        self.assertEqual(status, 0)
        self.assertTrue(np.array_equal(volPos, self.lat[0].pos))
        self.assertTrue(np.array_equal(volSpec, self.lat[0].specie))
        
        volAtoms = np.array([0, 2, 7, 8], dtype=np.int32)
        volPos = np.empty(3 * len(volAtoms), np.float64)
        volSpec = np.empty(len(volAtoms), np.int32)
        status =  _graphs.makeDefectVolumePosAndSpecie(volAtoms, volPos, volSpec, self.lat[0].pos, self.lat[0].specie)
        self.assertEqual(status, 0)
        for i in xrange(4):
            j = volAtoms[i]
            self.assertTrue(np.array_equal(volPos[3*i:3*i+3], self.lat[0].pos[3*j:3*j+3]))
            self.assertEqual(volSpec[i], self.lat[0].specie[j])

    

[docs]    def test_getHashKey(self):
        """
        Graphs get hash key
        
        """
        k1 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms, self.lat[0])
        k2 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms, self.lat[1])
        k3 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms, self.lat[2])
        self.assertEqual(k1, k2)
        self.assertNotEqual(k1, k3)
        
        k1 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms, self.lat[3])
        k2 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms, self.lat[4])
        k3 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms, self.lat[5])
        k4 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms, self.lat[6])
        k5 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms, self.lat[7])
        self.assertTrue(k1 == k2 == k3 == k4)
        self.assertFalse(k1 == k5)
        
        k1 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms[:-1], self.lat[8])
        k2 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms[:-1], self.lat[9])
        k3 = Graphs.getHashKeyForAVolume(self.params, self.volAtoms[:-1], self.lat[10])
        self.assertEqual(k1, k2)
        self.assertNotEqual(k1, k3)

[docs]    def test_prepareTheMatrix(self):
        """
        Graphs prepare the matrix
         
        """
        # case 1
        trnsfMatrix, lattice1, atLst1, lattice2, atLst2, cntr1, cntr2 = Graphs.prepareTheMatrix(self.params, self.volAtoms, self.lat[0], 
                                                                                                True, self.volAtoms, self.lat[1], True)
         
        r1 = np.arange(len(self.volAtoms), dtype=np.int32)
        r2 = np.array([26, 17, 8, 23, 14, 5, 20, 11, 2, 25, 16, 7, 22, 13, 4, 19, 10, 1, 24, 15, 6, 21, 12, 3, 18, 9, 0], dtype=np.int32)
        self.assertTrue(np.array_equal(atLst1, r1))
        self.assertTrue(np.array_equal(atLst2, r2))
         
        self.assertTrue(np.array_equal(cntr1, [2,2,2]))
        self.assertTrue(np.array_equal(cntr2, [2,2,2]))
         
        rA = np.array([[0, 0, -1], [0, -1, 0], [-1, 0, 0]], dtype=np.float64)
        self.assertTrue(np.allclose(trnsfMatrix, rA))
         
        # case 2
        trnsfMatrix, lattice1, atLst1, lattice2, atLst2, cntr1, cntr2 = Graphs.prepareTheMatrix(self.params, self.volAtoms, self.lat[3], 
                                                                                                True, self.volAtoms, self.lat[4], True)
         
        r1 = np.arange(len(self.volAtoms), dtype=np.int32)
        r2 = np.array([2, 1, 0, 5, 4, 3, 8, 7, 6, 11, 10, 9, 14, 13, 12, 17, 16, 15, 20, 19, 18, 23, 22, 21, 26, 25, 24], dtype=np.int32)
        self.assertTrue(np.array_equal(atLst1, r1))
        self.assertTrue(np.array_equal(atLst2, r2))
         
        self.assertTrue(np.array_equal(cntr1, [2,2,2]))
        self.assertTrue(np.array_equal(cntr2, [2,2,2]))
         
        rA = np.array([[1, 0, 0], [0, 1, 0], [0, 0, -1]], dtype=np.float64)
        self.assertTrue(np.allclose(trnsfMatrix, rA))
         
        # case 3
        trnsfMatrix, lattice1, atLst1, lattice2, atLst2, cntr1, cntr2 = Graphs.prepareTheMatrix(self.params, self.volAtoms[:-1], self.lat[8], 
                                                                                                True, self.volAtoms[:-1], self.lat[9], True)
         
        r1 = np.arange(len(self.volAtoms) - 1, dtype=np.int32)
        r2 = np.array([17, 8, 23, 14, 5, 20, 11, 2, 25, 16, 7, 22, 13, 4, 19, 10, 1, 24, 15, 6, 21, 12, 3, 18, 9, 0], dtype=np.int32)
        self.assertTrue(np.array_equal(atLst1, r1))
        self.assertTrue(np.array_equal(atLst2, r2))
         
        self.assertTrue(np.allclose(cntr1, np.array([2.07692308, 2.07692308, 2.07692308])))
        self.assertTrue(np.allclose(cntr2, np.array([1.92307692, 1.92307692, 1.92307692])))
         
        rA = np.array([[0, 0, -1], [0, -1, 0], [-1, 0, 0]], dtype=np.float64)
        self.assertTrue(np.allclose(trnsfMatrix, rA))
