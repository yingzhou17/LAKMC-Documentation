[docs]    def test_forcedSpecies(self):
        """
        Find defects forced species list
        
        """
        specieList = ["Pu", "Ga", "H_"]
        self.params.forceSpecieDefect = "H_"
        forcedSpecs = Defects.forcedSpecies(self.params, specieList)
        self.assertEqual(len(forcedSpecs), 1)
        self.assertEqual(forcedSpecs[0], 2)
        
        specieList = ["Pu", "Ga"]
        self.params.forceSpecieDefect = "H_"
        forcedSpecs = Defects.forcedSpecies(self.params, specieList)
        self.assertEqual(len(forcedSpecs), 0)
        
        specieList = ["Pu", "Ga", "H_", "Fe"]
        self.params.forceSpecieDefect = "Ga+Fe"
        forcedSpecs = Defects.forcedSpecies(self.params, specieList)
        self.assertEqual(len(forcedSpecs), 2)
        self.assertEqual(forcedSpecs[0], 1)
        self.assertEqual(forcedSpecs[1], 3)

    

[docs]    def test_excludedSpecies(self):
        """
        Find defects excluded species list
        
        """
        inputSpecieList = ["Pu", "Ga", "H_"]
        refSpecieList = ["Pu", "Ga", "Fe", "H_"]
        self.params.excludeSpecieDefect = "H_"
        inputExcluded, refExcluded = Defects.excludedSpecies(self.params, inputSpecieList, refSpecieList)
        self.assertEqual(len(inputExcluded), 1)
        self.assertEqual(inputExcluded[0], 2)
        self.assertEqual(len(refExcluded), 1)
        self.assertEqual(refExcluded[0], 3)
        
        inputSpecieList = ["Pu", "Ga", "H_"]
        refSpecieList = ["Pu", "Ga", "Fe", "H_"]
        self.params.excludeSpecieDefect = "Ga+Fe"
        inputExcluded, refExcluded = Defects.excludedSpecies(self.params, inputSpecieList, refSpecieList)
        self.assertEqual(len(inputExcluded), 1)
        self.assertEqual(inputExcluded[0], 1)
        self.assertEqual(len(refExcluded), 2)
        self.assertEqual(refExcluded[0], 1)
        self.assertEqual(refExcluded[1], 2)
        
        inputSpecieList = ["Pu", "Ga", "H_"]
        refSpecieList = ["Pu", "Ga", "Fe", "H_"]
        self.params.excludeSpecieDefect = "Pd+O_"
        inputExcluded, refExcluded = Defects.excludedSpecies(self.params, inputSpecieList, refSpecieList)
        self.assertEqual(len(inputExcluded), 0)
        self.assertEqual(len(refExcluded), 0)
