        """
        Unique
        
        """
        a = np.array([1,2,3,2,1], np.int32)
        b = [1,2,3,2,1]
        result1 = np.array([1,2,3], np.int32)
        result2 = [1,2,3]
        
        self.assertTrue(np.allclose(Utilities.unique(a), result1))
        self.assertTrue(np.allclose(Utilities.unique(b), result2))
