[docs]    def pathToData(self, relpath):
        """
        Find path to `relpath` within data repo.
        
        """
        dn = os.path.dirname(__file__)
        dn = os.path.abspath(os.path.join(dn, "..", "..", "..", "LKMCData"))
        if not os.path.isdir(dn):
            self.skipTest("Could not locate data directory: '{0}'".format(dn))
        
        path = os.path.join(dn, relpath)
        if not os.path.exists(path):
            self.skipTest("Path does not exist: '{0}'".format(path))
        
        return path
