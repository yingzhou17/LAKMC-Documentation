[docs]def calcHessian(system, mainIndexes, includeAtoms, displacement, noOfWorkers=1, includeMasses=0, jobID=None):
    
    """
    Calculates Hessian matrix of a system. Multiprocessing is possible.
    
    """
   
    atomsArrLen = len(mainIndexes)

    hessianMatrix = np.zeros([3*atomsArrLen, 3*atomsArrLen], np.float64)
    
    if noOfWorkers > 1:
        status = calcHessianMultiprocess(hessianMatrix, atomsArrLen, 
            system, mainIndexes, includeAtoms, displacement, noOfWorkers, includeMasses)
                
    else:
        status = calcHessianStandalone(hessianMatrix, atomsArrLen, 
            system, mainIndexes, includeAtoms, displacement, includeMasses, jobID=jobID)
    
    if not status:
        
        for i in range(3*atomsArrLen):
            for j in range(i):
                hessianMatrix[i][j] += hessianMatrix[j][i]
                hessianMatrix[i][j] /= 2.0
                hessianMatrix[j][i] = hessianMatrix[i][j]        
        
    return status, hessianMatrix



[docs]def calcHessianMultiprocess(hessianMatrix, atomsArrLen, system, mainIndexes, includeAtoms, displacement, noOfWorkers, includeMasses):
    """
    Calculates implicit Hessian matrix using multiprocessing module
    
    """
    
    jobQ = multiprocessing.Queue()
    resultQ = multiprocessing.Queue()
            
    for i in xrange(atomsArrLen):                
        jobQ.put([i, mainIndexes[i]])
    
    for i in xrange(noOfWorkers):
        jobQ.put(["EXIT"])
        
    workers = []
        
    for i in xrange(noOfWorkers):
        p = multiprocessing.Process(target=calcHessianWorker, args=(jobQ, resultQ, system, 
                mainIndexes, includeAtoms, displacement, includeMasses))
        
        p.start()
        workers.append(p)
    
    for i in xrange(atomsArrLen):
        result = resultQ.get()
        
        status = result[0]
        
        if status:
            print "FORCE CALC FAILED ON ", result[1]
           
            for p in workers:
                p.terminate()
                                
            return 1, None
        
        index = result[1]
        atomIndex = result[2]
        array = result[3]
        
        for j in range(0, 3):
            for k in range(3*atomsArrLen):
                hessianMatrix[3*index+j][k] = array[j][k]
                                
    for p in workers:
        p.join()

            

[docs]def calcHessianWorker(jobQ, resultQ, system, mainIndexes, includeAtoms, displacement, includeMasses):
    """
    Worker to calculate Hessian matrix.
    
    """
    
    mass1 = 0.0
    mass2 = 0.0    
    
    while True:
        try:
            job = jobQ.get()
            
        except:
            time.sleep(1)
            continue
        
        if job[0] == "EXIT":
            return

        index = job[0]
        atomIndex = job[1]       
                    
        status, result = calcHessianColumn(atomIndex, system, mainIndexes, includeAtoms, displacement, includeMasses)
        
        if status:
            resultQ.put([status, index, atomIndex])
        else:
            resultQ.put([status, index, atomIndex, result])



[docs]def calcHessianStandalone(hessianMatrix, atomsArrLen, system, mainIndexes, includeAtoms, displacement, includeMasses, jobID=None):
    
    """
    Calculates implicit Hessian matrix as a standalone module
        
    """
    
    index = 0

    for atomIndex in mainIndexes:
        status, result = calcHessianColumn(atomIndex, system, mainIndexes, includeAtoms, displacement, includeMasses)

        if status:
            print "FORCE CALC FAILED ON ", atomIndex
            
            return 1
        
        for j in range(0, 3):
            for k in range(3*atomsArrLen):
                hessianMatrix[3*index+j][k] = result[j][k]
        
        if jobID is not None and index % 20 == 0:
            if ClientServerInterface.checkJobStatus(jobID):
                print "TERMINATING Algebra.calcHessianStandalone DUE TO JOB STATUS"
                return 1
        
        index += 1
    
    return 0

  

[docs]def calcHessianColumn(atomIndex, system, mainIndexes, includeAtoms, displacement, includeMasses):
    
    mass1 = 0.0
    mass2 = 0.0 
    
    tempSystem = copy.deepcopy(system)
        
    atomsLen = len(mainIndexes)  
    result = np.zeros([3, 3*atomsLen], np.float64)

    for i in range(0, 3):
            
        tempSystem.pos[3*atomIndex+i] += displacement                     
        status = tempSystem.calcForce(mainIndexes=mainIndexes, includeAtoms=includeAtoms)
        tempSystem.pos[3*atomIndex+i] -= displacement

        if status:
            return status, None
        
        for j in range(atomsLen):
            for k in range(0, 3):
                result[i][3*j+k] = tempSystem.force[3*mainIndexes[j]+k]
                
        tempSystem.pos[3*atomIndex+i] -= displacement
        status = tempSystem.calcForce(mainIndexes=mainIndexes, includeAtoms=includeAtoms)
        tempSystem.pos[3*atomIndex+i] += displacement
                
        if status:
            return status, None
        
        for j in range(atomsLen):
            for k in range(0, 3):
                result[i][3*j+k] -= tempSystem.force[3*mainIndexes[j]+k]
        
    multiplyer = 1.0 / (-2.0 * displacement)
    
    if includeMasses:
        mass1 = Atoms.atomicMass(tempSystem.specieList[tempSystem.specie[atomIndex]])

    for i in range(0, 3):
        for j in range(atomsLen):
            if includeMasses:
                mass2 = Atoms.atomicMass(tempSystem.specieList[tempSystem.specie[mainIndexes[j]]])
                
                mass = math.sqrt(mass1*mass2)
            else:
                mass = 1.0
                
            for k in range(0, 3): 
                result[i][3*j+k] = result[i][3*j+k] * multiplyer / mass
                    
    return status, result



[docs]def getEigenvaluesST(alphaVec, betaVec, evn, noEigenVects=0):
    
    """
    Calculates eigenvalues and eigenvectors of symmetric tridiagonal matrix.
    
    """

    if evn >= 0:
        iu = evn + 1
    else:
        iu = 0
    
    if noEigenVects:
        jobz = "N"
    else: 
        jobz = "V"
    
    N = len(alphaVec)
    eigenVectors = np.zeros((N, N), np.float64, order='F')
    
    status = 0
    startTime = time.time()
    
    if iu > 0:
        eigenValues = np.zeros(N, np.float64)
        workspace = np.zeros(N*2, np.float64)
        M = 0
        work = np.zeros(5*N, np.float64)
        iwork = np.zeros(5*N, np.int)
        ifail = np.zeros(N, np.int)
        
        dstevx.dstevx(jobz=jobz, range="I", n=N, d=alphaVec, e=betaVec, vl=0, vu=0, il=1, iu=iu, 
            abstol=0.0, m=M, w=eigenValues, z=eigenVectors, ldz=N, work=work, 
            iwork=iwork, ifail=ifail, info=status)
        
        timeTaken = time.time() - startTime
        log(__name__, "Debug getEigenvaluesST (1): time taken %f s" % timeTaken,3,2)
        
        if status or eigenValues is None:
            print "ALGEBRAERROR: dstevx failed: %d" % status
            sys.stdout.flush()
            if not status:
                status = 31
        
    else:
        
        work = np.zeros(2*N-2, np.float64)
        
        dstev.dstev(jobz=jobz, n=N, d=alphaVec, e=betaVec, z=eigenVectors, ldz=N, work=work, info=status)
        
        timeTaken = time.time() - startTime
        log(__name__, "Debug getEigenvaluesST (2): time taken %f s" % timeTaken,3,2)
        
        if status or alphaVec is None:
            print "ALGEBRAERROR: dstev failed: %d" % status
            sys.stdout.flush()
            if not status:
                status = 30
        
        eigenValues = copy.deepcopy(alphaVec)
    
    if noEigenVects:
        eigenVectors = None
    
    return status, eigenValues, eigenVectors



[docs]def getEigenvaluesSY(hessianMatrix, N, evn, noEigenVects=0):
    
    """
    Calculates eigenvalues and eigenvectors of symmetric matrix.
    
    """
    
    if evn >= 0:
        iu = evn + 1
    else:
        iu = 0
    
    if noEigenVects:
        jobz = "N"
    else: 
        jobz = "V"
    
    status = 0
    info = 0
    
    A = np.reshape(hessianMatrix, (N, N), order='F')
    eigenValues = np.zeros(N, np.float64)
    
    nb = getOptimumBlocksize(N)          
    lwork = (nb+2)*N
    work = np.zeros(max(1, lwork), np.float64)
        
    if iu > 0:
        
        M = 0
        eigenVectors = np.zeros((N, N), np.float64, order='F')
        iwork = np.zeros(5*N, np.int)
        ifail = np.zeros(N, np.int)
        
        dsyevx.dsyevx(jobz=jobz, range="I", uplo="U", n=N, a=hessianMatrix, lda=N, vl=0, vu=0, il=1, iu=iu, 
            abstol=0.0, m=M, w=eigenValues, z=eigenVectors, ldz=N, work=work, lwork=lwork, 
            iwork=iwork, ifail=ifail, info=info)
        
        if info:
            print "ALGEBRAERROR: dsyevx failed: %d" % info
            sys.stdout.flush()
                
    else:
            
        dsyev.dsyev(jobz=jobz, uplo="U", n=N, a=hessianMatrix, lda=N, w=eigenValues, work=work, lwork=lwork, info=info)
        
        if info:
            print "ALGEBRAERROR: dsyevx failed: %d" % info
            sys.stdout.flush()
        
        eigenVectors = A
        
    if noEigenVects:
        eigenVectors = None
            
    if info != 0:
        status = 1

    return status, eigenValues, eigenVectors

[docs]def getSystemsImplicitEigenvalues(system, mainIndexes, includeAtoms, params, evn=-1, noEigenVects=0, 
        includeMasses=0, noOfWorkers=1, jobID=None):
    
    """
    Calculates system implicit eigenvalues and eigenvectors (optionally).
    
    """

    debugInfo = None
    displacement = 0.001
    
    timeSt = time.time()

    status, hessianMatrix = calcHessian(system, mainIndexes, includeAtoms, displacement, 
        noOfWorkers=noOfWorkers, includeMasses=includeMasses, jobID=jobID)
        
    hessianConstTime = time.time() - timeSt
    
    if status:
        log(__name__, "Failed to estimate hessianMatrix.", 0, 2)
        return status, None, None, None
    
    N = len(mainIndexes)*3
    
    timeSt = time.time()
    status, eigenValues, eigenVectors = getEigenvaluesSY(hessianMatrix, N, evn, noEigenVects=noEigenVects)
        
    dsyevTime = time.time() - timeSt  
    
    if params.prefactorDebug:
        hessianFEvals = N*2
        
        debugInfo = AlgebraPerform(dsyevTime=dsyevTime, 
            noHessianWorkers=noOfWorkers, hessianConstTime=hessianConstTime, hessianConstFEvals=hessianFEvals)
    
    if status:
        log(__name__, "Failed to estimate eigenvalues and eigenvectors.", 0, 2)
        
    return status, eigenValues, eigenVectors, debugInfo

            resultVector[atom*3+j] = lev[i*3+j]
