[docs]def commandLineArgs():
    """
    Parse command line arguments
    
    """
    import argparse
    
    parser = argparse.ArgumentParser(description="""
    Update/Compare parameter input file to the latest version.
        (default: OLD = 'lkmcInput.IN', NEW = '<LKMC_Folder>/input/lkmcInput.IN')
    ********************************************************
    - params not in NEW will be ignored;
    - params not in OLD will be added;  
    - param values equal to OLD if exist;
    - all comments will be from NEW. """, formatter_class=argparse.RawTextHelpFormatter)
    
    parser.add_argument('-o','--old', dest='INPUTFILE_old', default="lkmcInput.IN", 
                        help="OLD: The old version of lkmcInput file, where the parameter values will be kept.\n(default = 'lkmcInput.IN')")
    parser.add_argument('-n','--new', dest="INPUTFILE_new", default=os.path.join(os.path.abspath(os.path.dirname(__file__)), "..", "input", "lkmcInput.IN"), 
                        help="NEW: The latest version of lkmcInput file\n(default = '<LKMC_Folder>/input/lkmcInput.IN')")
    parser.add_argument('-p','--print', action = "store_true", dest = "print_flag", default=False, 
                        help="Option to print out params whose values are different in OLD and NEW\n(Only new added params are printed out by default)")
    parser.add_argument('-c','--compare', action = "store_true", dest = "compare_flag", default=False, 
                        help="Option to compare params whose values are different in OLD and NEW\n(OLD is untouched and no output file)")
#     parser.add_argument('-r','--return', action = "store_true", dest = "return_flag", default=False, 
#                         help="Return Mode. Count number of params which are deleted, added, changed, respectively")
    
    return parser.parse_args()

[docs]def run(inputFile_new, inputFile_old, print_flag=False, compare_flag=True, return_flag=True):
    """
    This script will merge old input file into new input file. 
    Params not in new will be ignored, params not in old will be added, param values equal to old if exist, all comments will be from new. 
    
    """
    if return_flag:
        compare_flag = True
    
    if compare_flag:
        print_flag = True
    else:
        outputFolder, outputFile = tempfile.mkstemp()
    
    # start merging
    try:
        f_new = open(inputFile_new, "r")
    except:
        sys.exit(__name__+": ERROR: could not open file: " + inputFile_new)
    try:
        f_old = open(inputFile_old, "r")
    except:
        sys.exit(__name__+": ERROR: could not open file: " + inputFile_old)
    if not compare_flag:
        try:
            f_out = open(outputFile, "w")
        except:
            sys.exit(__name__+": ERROR: could not open file: " + outputFile)
    
    searchlines = f_old.readlines()
    searchlines_n = f_new.readlines()
    
    f_old.close()
    f_new.close()
    
    if not return_flag:
        if compare_flag:
            print "Comparing two input files..."
        else:
            print "Updating input file..."
        if print_flag:
            print """***********************************************************************************
Parameters which have different values in OLD and NEW are listed here: 
(OLD = %r, NEW = %r)
(Values in the output file will be value_in_old, except it's a new added parameter.)
***********************************************************************************

                         Param_Name  :         value_in_new >>         value_in_old
      ------------------------------ : -------------------- >> --------------------"""%(inputFile_old, inputFile_new)
    
    # First find out all parames which are deleted in the new version
    findParamName = 0
    NoDel = 0
    for line in searchlines:
        if line[0] == const.ignorLineSign or line[0:len(const.sectionStartMark)] == const.sectionStartMark or line[0:len( const.sectionEndMark)] == const.sectionEndMark:
            continue
        line.rstrip()
        if not findParamName and line[0] == "%":
            paramName = line[1:-1]
            findParamName = 1
            continue
        elif findParamName:
            findParamName = 0
            value_o = line.rstrip()
            
            #search param in new
            if ("%"+paramName+"\n") not in searchlines_n:
                NoDel += 1
                if print_flag and not return_flag:
                    print ">DEL> %30s : %20r >> %20r" %(paramName, "****<DELETED>****",value_o) 
                elif not return_flag:
                    print "    >>Deleted Parameter: %26r : %20r" %("("+paramName+")", "("+value_o+")")
        else:
            sys.exit("Undefined error in updateParameters: check the format of your input file!")
    
    # keep comments from new, and write to out
    comments = ""
    sectionName = ""
    findParamName = 0
    NoNew = 0
    NoDiff = 0
    for line in searchlines_n:
        if line[0] == const.ignorLineSign:
            comments += line
            continue
            
        line.rstrip()
        
        #find section start, write comments for this section
        if line[0:len(const.sectionStartMark)] == const.sectionStartMark:
            sectionName = line[len(const.sectionStartMark):-1]
            if not compare_flag:
                f_out.write(comments)
                f_out.write(const.sectionStartMark + sectionName + "\n")
            continue
        
        #find section end, clear comments and sectionName for this section
        if line[0:len( const.sectionEndMark)] == const.sectionEndMark:
            if not compare_flag:
                f_out.write(const.sectionEndMark + sectionName + "\n")
            comments = ""
            sectionName = ""
            continue
        
        #read in and write params
        if not findParamName and line[0] == "%":
            paramName = line[1:-1]
            findParamName = 1
            if not compare_flag:
                f_out.write("%" + paramName + "\n")
            continue
        
        elif findParamName:
            findParamName = 0
            value_n = line.rstrip()
            value = None
            
            #search param value in old
            for i, lines in enumerate(searchlines):
                if ("%"+paramName+"\n") in lines:
                    value = searchlines[i+1].rstrip()
                    break
            
            if value is None:
                NoNew += 1
                value = value_n
                if print_flag and not return_flag:
                    print ">NEW> %30s : %20r >> %20r" %(paramName, value_n, "***<NEW PARAM>***")
                    #print ">NEW>", paramName, ":", value_n, ">> None"
                elif not return_flag:
                    #print ">>NEW Parameter:", paramName, ":", value_n
                    print "    >>NEW Parameter: %30r : %20r" %(paramName, value_n)
            elif value_n != value: 
                NoDiff += 1
                if print_flag and not return_flag:
                    #print "     ", paramName, ":", value_n, ">>", value
                    print "      %30s : %20r >> %20r" %(paramName, value_n, value)
            
            if not compare_flag:
                f_out.write(value + "\n")
            
        else:
            sys.exit("Error")
    
    if comments != "":
        if not compare_flag:
            f_out.write(comments)
    if print_flag and not return_flag:
        print "***********************************************************************************"
    
    
    if not compare_flag:
        f_out.close()
        
        # If input file is outdated. Backup and update.
        if NoDel or NoNew:
            import datetime
            backup_file = inputFile_old+"."+datetime.datetime.today().strftime("%Y%m%d%H%M%S")+".backup"
            os.system("mv %r %r"%(inputFile_old, backup_file))
            print "OLD input file is backed up at %r" % backup_file
            os.system("cp %r %r"%(outputFile ,inputFile_old))
            
            os.close(outputFolder)
            #os.unlink(outputFile)
        else:
            print "Your input file is up-to-date. %3d parameters are different from the default. \nRe-run with '-c' to see differences."%NoDiff
            
            os.close(outputFolder)
            os.unlink(outputFile)
    
    if compare_flag and not return_flag:
        print "Comparing Finished. %3d deleted, %3d new, %3d different parameters\n" %(NoDel, NoNew, NoDiff)
    elif not return_flag:
        print "\nParameters are updated. %3d deleted, %3d new, %3d different parameters\n" %(NoDel, NoNew, NoDiff)
        
    if return_flag:
        return NoDel, NoNew, NoDiff

    
if __name__ == "__main__":
    main()
