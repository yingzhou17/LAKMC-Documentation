[docs]    def run(self, latticeIn, step=None, clock=None):
        """
        Calls all of the methods to do one complete deposition simulation.
        
        Parameters
        ----------
        latticeIn : `Lattice.Lattice`
            System lattice to run the deposition on.
        step : int, optional
            Current KMC step, to allow changing into the appropriate working directory.
        clock : float, optional
            Not currently used; for compatibility with other external events.
            
        Returns:
        ----------
        lattice : `Lattice.Lattice`
            Post-deposition lattice (minimised if user requires)
        status : int
            Status report, returned as zero if no errors have been encountered.
        
        Notes
        -----
        The workflow is as follows:
            * Pre-event thermalisation
            * Add deposition atoms
            * Simulating the deposition
            * Post-deposition minimisation
            * Remove sputtered atoms
        
        """
        log(__name__, "Running deposition", 0, 0)
        status = 0
        
        # copy the input lattice
        lattice = copy.deepcopy(latticeIn)
        
        # parameters
        params = self.params
        
        # switch to working directory
        root = self.switchToOutputDirectory(step)
        try:
            # first thermalise the lattice
            velocity = self.thermalise(lattice, params.depoThermalise, params.depoThermoTimeWhole,
                                       params.depoThermoTimeLayer, params.depoThermoType)
            
            # add the deposition atoms
            velocity = self.addDepositionAtoms(lattice, velocity)
            
            # run the MD simulation
            self.runMD(lattice, velocity, params.depoThermoType, params.depoTime)
            
            # minimise the final lattice
            self.minimise(lattice, params.depoMinAfterDepo, params.depoMinUseLAMMPS, params.depoMinFailUseLAMMPS)
            
            # remove sputtered atoms
            lattice.removeSputter(self.refState, params)
        
        finally:
            os.chdir(root)
        
        return status, lattice

    

[docs]    def addDepositionAtoms(self, lattice, velocity):
        """
        Insert cluster(s) into the initial system.
        The inserted cluster(s) are chosen randomly by `readDepoIn()`.
        
        Parameters
        ----------
        velocity : ndarray
            Array containing the velocities of each atom in the system after thermalisation.
            
        Returns
        -------
        velocity : ndarray
            Array containing the velocities of each atom in the system after thermalisation with inserted atoms.
        
        """
        # Read in depo.in file, and make the choice
        restuple = self.readDepoIn(lattice)
        depoDirection, depoCluster, chosenPosition, chosenTheta, chosenPhi, depoVelocity, index = restuple
        selectedCluster = depoCluster.clusters[index]
        
        # add first atom to lattice
        sym = lattice.specieList[selectedCluster.atoms[0].specie]
        lattice.addAtom(sym, chosenPosition, selectedCluster.atoms[0].charge)
        
        # append velocity for first atom
        atomVelocity = np.empty(3, np.float64)
        atomVelocity[0] = depoVelocity * depoDirection[0]
        atomVelocity[1] = depoVelocity * depoDirection[1]
        atomVelocity[2] = depoVelocity * depoDirection[2]
        velocity = np.append(velocity, atomVelocity)
        
        # for clusters contain more than 1 atoms
        if selectedCluster.NAtom:
            for i in xrange(1, selectedCluster.NAtom):
                selectedClusterAtom = selectedCluster.atoms[i]
                theta = math.atan2(math.sqrt(selectedClusterAtom.offset[0] ** 2 + selectedClusterAtom.offset[1] ** 2),
                                   selectedClusterAtom.offset[2])
                phi = math.atan2(selectedClusterAtom.offset[1], selectedClusterAtom.offset[0])
                    
                # Now add our random rotation
                theta += chosenTheta
                phi += chosenPhi
                 
                dist = math.sqrt(selectedClusterAtom.offset[0] ** 2 + selectedClusterAtom.offset[1] ** 2 +
                                 selectedClusterAtom.offset[2] ** 2)
                    
                # Now turn from spherical back to cartesian coords
                subPosition = np.zeros(3, np.float64)
                subPosition[0] = chosenPosition[0] + dist * math.sin(theta) * math.cos(phi)
                subPosition[1] = chosenPosition[1] + dist * math.sin(theta) * math.sin(phi)
                subPosition[2] = chosenPosition[2] + dist * math.cos(theta)
                 
                # add additional atom to lattice
                sym = lattice.specieList[selectedClusterAtom.specie]
                lattice.addAtom(sym, subPosition, selectedClusterAtom.charge)
                 
                # append velocity for atom
                atomVelocity = np.empty(3, np.float64)
                atomVelocity[0] = depoVelocity * depoDirection[0]
                atomVelocity[1] = depoVelocity * depoDirection[1]
                atomVelocity[2] = depoVelocity * depoDirection[2]
                velocity = np.append(velocity, atomVelocity)
        
        return velocity

    

[docs]    def readDepoIn(self, lattice):
        """
        Read information of deposited clusters from file "depo.in".
        
        Returns
        -------
        depoDirection : ndarray
            The chosen direction of velocities of the deposited cluster.
        depoCluster : `DepoClustersObject` object
            This object holds information about the deposited cluster candidates.
        chosenPosition : ndarray
            The chosen position of the deposited cluster to be inserted.
        chosenTheta : float
            The chosen angle of the deposited cluster to be rotated.
        chosenPhi : float
            The chosen angle of the deposited cluster to be rotated.
        depoVelocity : float
            The chosen velocity of the deposited cluster, in Å/fs.
        index : integer
            The chosen cluster's index number (in the `DepoClustersObject` object).
        
        """
        # read in parameters for deposition
        depoParamFile = os.path.join(Globals.MDLocation, 'depo.in')
        
        # pre-define deposition parameters
        depoDirection = np.zeros(3, np.float64)
        depoEnergy = np.zeros(2, np.float64)
        depoWindow = np.zeros(6, np.float64)
        depoCluster = DepoClustersObject()
        
        with open(depoParamFile) as infile:
            # read in primary deposition direction
            infile.readline()
            line = infile.readline().split()
            (depoDirection[0], depoDirection[1], depoDirection[2]) = (float(line[0]), float(line[1]), float(line[2]))
            depoDirection = Vectors.normalise(depoDirection)
            log(__name__, "Deposition direction: (%f,%f,%f)" % (depoDirection[0], depoDirection[1], depoDirection[2]),
                3, 2)
            
            # read in deposition energy
            infile.readline()
            line = infile.readline().split()
            (depoEnergy[0], depoEnergy[1]) = (float(line[0]), float(line[1]))
            log(__name__, "Deposition energy range: (%f,%f)" % (depoEnergy[0], depoEnergy[1]), 3, 2)
            
            # read in deposition species
            infile.readline()
            line = infile.readline().split()
            # check species defined in depo.in
            for i in xrange(1, int(line[0]) + 1):
                if line[i] not in lattice.specieList:
                    log(__name__, "WARNING: Specie %s defined in depo.in is not in input data file" % (line[i]), 1, 2)
                    lattice.addSpecie(line[i])
            depoSpecie = lattice.specieList
            depoSpecieMass = lattice.specieMass
            depoSpecieAMU = lattice.specieMassAMU
            
            if len(depoSpecieMass) != len(depoSpecie):
                log(__name__, "Rebuilding mass list... list length was %d should be %d." % (len(depoSpecieMass),
                                                                                            len(depoSpecie)),
                    1, 2)
                depoSpecieMass = np.empty(len(depoSpecie), np.float64)
                depoSpecieAMU = np.empty(len(depoSpecie), np.float64)
                for i, sym in enumerate(depoSpecie):
                    depoSpecieMass[i] = Atoms.atomicMass(sym)
                    depoSpecieAMU[i] = Atoms.atomicMassAMU(sym)
            
            # read in the departure window
            infile.readline()
            line = infile.readline().split()
            depoWindow[0] = 0.0 if line[0] == "INF" else float(line[0])
            depoWindow[1] = lattice.cellDims[0] if line[1] == "INF" else float(line[1])
            line = infile.readline().split()
            depoWindow[2] = 0.0 if line[0] == "INF" else float(line[0])
            depoWindow[3] = lattice.cellDims[4] if line[1] == "INF" else float(line[1])
            line = infile.readline().split()
            depoWindow[4] = 0.0 if line[0] == "INF" else float(line[0])
            depoWindow[5] = lattice.cellDims[8] if line[1] == "INF" else float(line[1])
            log(__name__, "Deposition departure window: (%f-%f,%f-%f,%f-%f)" % (depoWindow[0], depoWindow[1],
                                                                                depoWindow[2], depoWindow[3],
                                                                                depoWindow[4], depoWindow[5]),
                3, 2)
            
            # read in the deposition atom/cluster information
            infile.readline()
            clusterNum = int(infile.readline().split()[0])
            depoCluster.NCluster = clusterNum
            infile.readline()
            for i in xrange(clusterNum):
                line = infile.readline().split()
                # number of atoms in cluster
                NAtoms = int(line[0])
                # probability of selection
                prob = float(line[1])
                depoCluster.addCluster(NAtoms, prob)
                log(__name__, "New cluster %d: %d atoms" % (i, NAtoms), 3, 2)

                for j in xrange(NAtoms):
                    currentCluster = depoCluster.clusters[i]
                    line = infile.readline().split()
                    if line[0] not in depoSpecie:
                        sys.exit("ERROR: " + line[0] + " is a new specie")
                    # specie index
                    specie = np.where(depoSpecie == line[0])[0][0]
                    # charge
                    charge = float(line[1])
                    currentCluster.addAtom(specie, charge)
                    if j > 0:
                        # secondary atom offset
                        currentCluster.atoms[j].setOffset(float(line[2]), float(line[3]), float(line[4]))
                        log(__name__,
                            "Sub atom: %s, charge: %f, orientation: (%f,%f,%f)" % (lattice.specieList[specie], charge,
                                                                                   currentCluster.atoms[j].offset[0],
                                                                                   currentCluster.atoms[j].offset[1],
                                                                                   currentCluster.atoms[j].offset[2],),
                            3, 3)
                    
                    else:
                        log(__name__, "Primary atom: %s, charge: %f" % (lattice.specieList[specie], charge), 3, 3)
        
        # choose the departure position
        chosenPosition = np.zeros(3, np.float64)
        chosenPosition[0] = random.uniform(depoWindow[0], depoWindow[1])
        chosenPosition[1] = random.uniform(depoWindow[2], depoWindow[3])
        chosenPosition[2] = random.uniform(depoWindow[4], depoWindow[5])
        
        # choose a rotation angle
        chosenTheta = random.uniform(0, 2 * math.pi)
        chosenPhi = random.uniform(0, 2 * math.pi)
        
        # choose a deposition energy
        chosenEnergy = random.uniform(depoEnergy[0], depoEnergy[1])
        
        # choose a cluster to insert
        chosen = random.random()
        index = 0
        prob = depoCluster.clusters[index].prob
        while prob < chosen:
            index += 1
            prob += depoCluster.clusters[index].prob
        log(__name__, "Selected Deposition Cluster %d (%d atoms)" % (index, depoCluster.clusters[index].NAtom), 2, 2)
        log(__name__, "Selected Deposition Energy %f" % chosenEnergy, 2, 2)
        
        depoNAtom = depoCluster.clusters[index].NAtom
        # total mass in LBOMD units (eV, Å, fs)
        totalMass = 0.0
        totalMassAMU = 0.0
        for i in xrange(0, depoNAtom):
            totalMass += depoSpecieMass[depoCluster.clusters[index].atoms[i].specie]
            totalMassAMU += depoSpecieAMU[depoCluster.clusters[index].atoms[i].specie]
        
        # get the velocity from the energy (eV)
        depoVelocity = self.getVelocityFromKE(chosenEnergy, mass=totalMass)
        
        return depoDirection, depoCluster, chosenPosition, chosenTheta, chosenPhi, depoVelocity, index

        """

        ----------
        dim_x : float
        dim_y : float

        self.offset[2] = dim_z




[docs]def main():
    if not len(sys.argv) == 2:
        sys.exit("Usage: Deposition.py inputLatticeFile")
    
    inputState = Lattice.readLattice(sys.argv[1])
    
    params = Input.getLKMCParams()
    Input.readGlobals()
    
    depoRunner = DepositionRunner(params)
    finalLattice, status = depoRunner.run(inputState)
    
    if status:
        print "deposition failed"
    else:
        finalLattice.writeLattice("final.dat")
