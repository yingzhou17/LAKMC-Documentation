[docs]def clientWorker(rank, parameters, resultQ, clientQ, taskStatus, finalsLock, finalsListProxy,
                 finalsListLocal, parentConn, jobQ, jobStatusDict, jobStatusLock, nodeName,
                 ratesListLock, ratesListProxy, barrierListProxy, barrierListLocal, dataDictProxy,
                 clientOutputPath):
    """
    The client worker that gets jobs from the 
    shared job queue, processes them and adds
    result to shared result queue.
    
    """
    # node this worker is running on
    address = Utilities.nodeAddress()
    
    # set log prepend
    name = "Worker-%d" % rank
    Globals.prependLog = "[%s]: " % name
    
    if parameters.clientIndividualOutput:
        # redirect stdout/stderr to a file
        stdoutFile = "%s.worker%d.STDOUT.txt" % (nodeName, rank)
        stderrfile = "%s.worker%d.STDERR.txt" % (nodeName, rank)
        
        if clientOutputPath is not None and os.path.isdir(clientOutputPath):
            stdoutFile = os.path.join(clientOutputPath, stdoutFile)
            stderrfile = os.path.join(clientOutputPath, stderrfile)
        
        # flush current output
        sys.stdout.flush()
        sys.stderr.flush()
        
        # new stdout file
        if Globals.verboseLevel > 1:
            so = file(stdoutFile, 'a+', 0)
        else:
            so = file(stdoutFile, 'a+')
        
        # new stderr file
        se = file(stderrfile, 'a+', 0)
        
        # copy to stdout and stderr
        os.dup2(so.fileno(), sys.stdout.fileno())
        os.dup2(se.fileno(), sys.stderr.fileno())
    
    # ready to begin
    log(__name__, "Ready to work...", 0)
    
    sys.stdout.flush()
    
    # job status
    Globals.jobStatusDict = jobStatusDict
    Globals.jobStatusLock = jobStatusLock
    
    # signal parent to let them know we are awake
    parentConn.send(name)
    
    # wait for parent to tell us to continue
    msg = parentConn.recv()
    log(__name__, "Parent has freed us to proceed (initial) (%s)" % msg, 2)
    
    # main loop - until we are told to exit
    while True:
        # signal parent to let them know we are ready
        parentConn.send(name)
        
        # now we wait for parent to tell us to continue
        log(address, "Waiting on parent", 2)
        msg = parentConn.recv()
        log(address, "Parent has freed us to proceed (%s)" % msg, 2)
        
        # flush stdout
        if parameters.flushSTDOUT:
            sys.stdout.flush()
        
        # get data
        dataDict = dataDictProxy._getvalue()
        
        # check if we are being told to exit
        try:
            dataDict["EXIT"]
        
        except KeyError:
            pass
        
        else:
            log(__name__, "Exiting", 0)
            sys.stdout.flush()
            return
        
        # unique id for current data
        dataID = dataDict["id"]
        
        # job loop - process jobs until told to get new data
        item = None
        while True:
            log(__name__, "Waiting for new job", 0, 0)
            # get job from job queue
            item = jobQ.get()
            
            # data ID
            jobDataID = item.dataID
            
            # clear data?
            if item.clearData and jobDataID == dataID:
                log(__name__, "Breaking job loop to clear data", 1, 0)
                break
            
            elif item.clearData:
                log(__name__, "Tried to clear old data! (%s - %s)" % (jobDataID, dataID), 0, 0)
                item = None
                continue
            
            # check this job corresponds to current data
            if jobDataID is not None and jobDataID != dataID:
                log(__name__, "Job does not correspond to current data (%s - %s)" % (jobDataID, dataID), 0, 0)
                item = None
                continue
            
            # unpack object
            obj = item.obj
            jobID = item.ID
            
            # finals list?
            method = getattr(obj, "setTransitionsLists", None)
            if method is not None:
                method(finalsLock, finalsListProxy, barrierListProxy, localPosList=finalsListLocal,
                       localBarrierList=barrierListLocal)
            
            # finals list?
            method = getattr(obj, "setRatesList", None)
            if method is not None:
                method(ratesListLock, ratesListProxy)
            
            # rank?
            method = getattr(obj, "setRank", None)
            if method is not None:
                method(rank)
            
            # data?
            method = getattr(obj, "setInitialData", None)
            if method is not None:
                status = method(dataDict)
                if status:
                    result = "FAIL: COULD NOT GET DATA FROM DATA DICT"
                    resultQ.put(result)
                    item = None
                    continue
            
            # ID?
            method = getattr(obj, "setJobID", None)
            if method is not None:
                method(jobID)
            
            # unpack job
            method = item.method
            args = item.args
            kwargs = item.kwargs
            
            # get the method from the object
            objMethod = getattr(obj, method, None)
            
            # now run it
            if objMethod is None:
                # failure
                result = "FAIL: UNRECOGNISED METHOD"
            
            else:
                runTime = time.time()
                result = objMethod(*args, **kwargs)
                runTime = time.time() - runTime
                
                if hasattr(result, "setRunTime"):
                    result.setRunTime(runTime)
                
                if hasattr(result, "worker") and result.worker is None:
                    result.worker = "%s:%d" % (address, rank)
            
            # check status of this job
            jobStatus = ClientServerInterface.checkJobStatus(jobID)
            if jobStatus:
                log(__name__, "Breaking to get new job", 0)
                # if the result is complete we may as well send it back???? or not cos server not expecting it????
                item = None
            
            else:
                resultQ.put(result)

[docs]    def daemonise(self):
        """
        Daemonise the client.
        
        We fork once and create our own session, then fork again to detach 
        from the controlling terminal.
        
        """
        # first fork
        try:
            pid = os.fork()
            
            # parent exits
            if pid:
                os._exit(0)
        
        except OSError, e:
            sys.stderr.write("fork #1 failed: %d (%s)\n" % (e.errno, e.strerror))
            sys.exit(1)
                
        # decouple from parent environment
        os.chdir("/")
        os.setsid()
        os.umask(022)
        
        # second fork
        try:
            pid = os.fork()
            
            # parent exits
            if pid:
                os._exit(0)
        
        except OSError, e:
            sys.stderr.write("fork #2 failed: %d (%s)\n" % (e.errno, e.strerror))
            sys.exit(1)
        
        # change to work dir
        os.chdir(self.workDir)    
        
        # node address
        self.address = Utilities.nodeAddress()
        
        # redirect standard file descriptors
        stdoutFile = self.nodeName + ".STDOUT.txt"
        stderrfile = self.nodeName + ".STDERR.txt"
        
        if self.clientOutputPath is not None and os.path.isdir(self.clientOutputPath):
            stdoutFile = os.path.join(self.clientOutputPath, stdoutFile)
            stderrfile = os.path.join(self.clientOutputPath, stderrfile)
        
        sys.stdout.flush()
        sys.stderr.flush()
        si = file("/dev/null", 'r')
        if Globals.verboseLevel > 1:
            so = file(stdoutFile, 'a+', 0)
        else:
            so = file(stdoutFile, 'a+')
        se = file(stderrfile, 'a+', 0)
        os.dup2(si.fileno(), sys.stdin.fileno())
        os.dup2(so.fileno(), sys.stdout.fileno())
        os.dup2(se.fileno(), sys.stderr.fileno())
        
        LKMCClient.clientPID = os.getpid()

        

[docs]    def makeClientManager(self):
        """
        Make sync manager for the client and connect to the queues
        
        """
        class ClientQueueManager(SyncManager):
            pass
    
        # register functions for getting queues
        ClientQueueManager.register("getDataDict")
        ClientQueueManager.register("getJobQ")
        ClientQueueManager.register("getResultQ")
        ClientQueueManager.register("getClientQ")
        ClientQueueManager.register("getStatusQ")
        ClientQueueManager.register("getFinalsLock")
        ClientQueueManager.register("getFinalsList")
        ClientQueueManager.register("getRatesLock")
        ClientQueueManager.register("getRatesList")
        ClientQueueManager.register("getBarriersList")
        
        # create and connect the manager
        self.manager = ClientQueueManager(address=(self.serverAddress, self.serverPort), authkey=self.authkey)
        self.manager.connect()
        
        log(self.address, "Connected to %s:%s" % (self.serverAddress, self.serverPort), 0)
        
        # get the queues
        self.dataDict = self.manager.getDataDict()
        self.jobQ = self.manager.getJobQ()
        self.resultQ = self.manager.getResultQ()
        self.clientQ = self.manager.getClientQ()
        self.statusQ = self.manager.getStatusQ()
        self.finalsLock = self.manager.getFinalsLock()
        self.finalsListProxy = self.manager.getFinalsList()
        self.ratesListLock = self.manager.getRatesLock()
        self.ratesListProxy = self.manager.getRatesList()
        self.barriersListProxy = self.manager.getBarriersList()
        
        # we put our pid into result q for the server to store
        self.resultQ.put([self.nodeName, os.getpgrp()])
        
        # make a local manager for the local copy of finals list
        self.localManager = multiprocessing.Manager()
        self.finalsListLocal = self.localManager.list()
        self.barrierListLocal = self.localManager.list()
        
        # and the dict for job status
        self.jobStatusDict = self.localManager.dict()
        self.jobStatusLock = self.localManager.RLock()
        
        # make dict for data
        self.localDataDict = self.localManager.dict()

    

[docs]    def getWorkerResponses(self, timeout=300, raiseOnError=True):
        """
        Get responses from workers
        
        """
        #TODO: run on multiple threads?
        
        log(self.address, "Checking all workers are responsive (timeout=%d)" % timeout, 1, 2)
        
        missingList = []
        for name, conn in zip(self.workersNames, self.workersConnections):
            if conn.poll(timeout):
                # worker is ready and waiting
                msg = conn.recv()
                log(self.address, "Worker is ready: %s" % msg, 1, 3)
            
            else:
                # worker is not responding
                log(self.address, "Worker has not responded: %s" % name, 1, 3)
                missingList.append(name)
        
        if missingList and raiseOnError:
            raise RuntimeError("Some workers did not respond: %s" % ", ".join(missingList))
        
        return missingList

[docs]    def createWorkerProcess(self, rank):
        """
        Create a new worker process.
        
        Parameters
        ----------
        rank : int
            The rank of the new process
        
        """
        # create pipe for this worker
        conn1, conn2 = multiprocessing.Pipe()
        
        # create worker process
        p = multiprocessing.Process(target=clientWorker, 
                                    args=(rank, self.LKMCParams, self.resultQ, self.clientQ, self.taskStatus, self.finalsLock,
                                          self.finalsListProxy, self.finalsListLocal, conn2, self.jobQ, self.jobStatusDict,
                                          self.jobStatusLock, self.nodeName, self.ratesListLock, self.ratesListProxy,
                                          self.barriersListProxy, self.barrierListLocal, self.localDataDict, self.clientOutputPath), 
                                    name="Worker-%d" % rank)
        
        return p, conn1

    

[docs]    def startWorkers(self):
        """
        Start the workers
        
        """
        # now we start the worker processes, passing them the queues
        self.workers = []
        self.workersNames = []
        self.workersConnections = []
        for i in xrange(self.NWorkers):
            # rank of this process
            rank = self.startRank + i
            
            # create the process and pipe
            p, conn = self.createWorkerProcess(rank)
            
            # start the process
            log(self.address, "Starting Worker-%d" % rank, 2, 0)
            p.start()
            
            # make sure the process started
            if conn.poll(5):
                msg = conn.recv()
                log(self.address, "Worker started: %s" % msg, 2, 1)
            
            else:
                # process did not start correctly!
                # we attempt to restart it
                log(self.address, "Worker failed to start: Worker-%d" % rank, 1)
                log(self.address, "Attempting to restart it...", 1, 1)
                
                # first we terminate the process
                p.terminate()
                p.join()
                
                # then we start another one
                p, conn = self.createWorkerProcess(rank)
                p.start()
                
                if conn.poll(5):
                    msg = conn.recv()
                    log(__name__, "Worker started successfully this time: %s" % msg, 1, 2)
                
                else:
                    # raise exception
                    raise RuntimeError("Could not start Worker-%d" % rank)
            
            # store process
            self.workers.append(p)
            self.workersNames.append("Worker-%d" % rank)
            self.workersConnections.append(conn)
        
        # workers are waiting for us to tell them to continue
        self.tellWorkersToContinue()

        """

        # now join
        for p in self.workers:

[docs]    def watchStatus(self):
        """
        Wait for server to tell us what to do
        
        """
        # main loop
        while True:
            # get job/status from server
            log(__name__, "Head client waiting for command", 0, 0)
            sys.stdout.flush()
            
            status, extraData = self.statusQ.get()
            
            log(__name__, "Client received command: %s" % (status,), 0)
            sys.stdout.flush()
            
            if status == "FREE_WORKERS":
                # free workers
                self.tellWorkersToContinue()
            
            elif status == "FINALISE":
                self.joinWorkers()
                
                self.statusQ.task_done()
                
                return
            
            elif status == "CLEARUNIQUELIST":
                log(self.address, "Clearing local uniqueness list", 0, 1)
                
                self.finalsLock.acquire()
                self.finalsListLocal.__delitem__(slice(0, self.finalsListLocal.__len__()))
                self.barrierListLocal.__delitem__(slice(0, self.barrierListLocal.__len__()))
                self.finalsLock.release()
            
            elif status == "STOPJOB":
                jobID = extraData
                
                log(self.address, "Stopping job: %s" % (str(jobID),), 0, 1)
                
                self.jobStatusLock.acquire()
                
                self.jobStatusDict[jobID] = 1
                
                self.jobStatusLock.release()
            
            elif status == "CLEARJOBSTATUS":
                log(self.address, "Clearing job status dict", 0, 1)
                
                self.jobStatusLock.acquire()
                
                self.jobStatusDict.clear()
                
                self.jobStatusLock.release()
                        
            elif status == "SETJOBIDS":
                log(self.address, "Setting job IDs", 0, 1)
                
                self.jobStatusLock.acquire()
                
                self.jobStatusDict.clear()
                
                jobIDs = extraData
                for jobID in jobIDs:
                    self.jobStatusDict[jobID] = 0
                
                self.jobStatusLock.release()
            
            elif status == "RESETDATA":
                log(self.address, "(Re)setting data", 0, 1)
                
                # check workers are ready/responsive
                self.getWorkerResponses()
                
                if not self.firstTime:
                    # clear job status
                    log(self.address, "Clearing job status dict", 0, 1)
                    self.jobStatusLock.acquire()
                    self.jobStatusDict.clear()
                    self.jobStatusLock.release()
                
                else:
                    self.firstTime = False
                
                # all workers are now waiting for us to tell them to continue
                # now we get data from data q and insert it into localDataDict
                
                keys = self.dataDict.keys()
                log(__name__, "Data keys: %r" % keys, 1, 1)
                self.localDataDict.clear()
                for key in keys:
                    val = self.dataDict[key]
                    self.localDataDict[key] = val
            
            # check in with server
            self.statusQ.task_done()
            log(self.address, "Status Q task_done'd", 2, 1)
            log(self.address, "Waiting for server to task_done clientQ", 0, 1)
            sys.stdout.flush()
            self.clientQ.put("READY:%s" % (self.nodeName,))
            self.clientQ.join()
            log(__name__, "Join client Q freed", 2, 1)

[docs]def main():
    """
    The main client loop.
    First we create a queue manager which connects to the 
    queues on the server.
    We then create "NWorkers" worker processes and pass
    them the queues.
    We then wait for the workers to exit, and then return.
    (We should probably listen on a special queue from the 
    server so that it can tell us to exit, eg if received sigint...)
    
    """
    if len(sys.argv) != 12:
        print "Usage: Client.py NODENAME WORKDIR RANK NUMWORKERS SERVERADDRESS SERVERPORT AUTHKEY INPUTFILE RELDIR SERVERID CLIENTOUTPUTDIR"
        sys.exit(1)
    
    # create client instance
    client = LKMCClient(sys.argv[1], sys.argv[2], int(sys.argv[3]), int(sys.argv[4]), 
                        sys.argv[5], int(sys.argv[6]), sys.argv[7], sys.argv[8], sys.argv[9], 
                        sys.argv[10], sys.argv[11])
    
    # daemonise the main client
    client.daemonise()
    
    # start the sync manager and connect to the server
    client.makeClientManager()
    
    # start the workers
    client.startWorkers()
    
    # check status
    client.watchStatus()
    
    log(__name__, "Finished", 0)
