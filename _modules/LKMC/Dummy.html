[docs]    def writeLattice(self, path):
        latticeFile = open(path, "w")
        latticeFile.write("%d\n" % self.NAtoms)
        latticeFile.write("%f %f %f\n" % (self.cellDims[0], self.cellDims[4], self.cellDims[8]))
        for n in range(self.NAtoms):
            latticeFile.write("%s %f %f %f %f\n" % (self.specieList[self.specie[n]],
                                                    self.pos[3*n], self.pos[3*n+1], self.pos[3*n+2],
                                                    0.0))
        latticeFile.close()

    def __init__(self, trans ):
        self.defectVolumes = []
        self.defectVolumes.append(DummyDV(trans))
        if hasattr(trans, "graphVol"):
            self.defectsObject = DummyDO(trans)

            Natom = np.max(trans.graphVol)+1
            self.pos = np.zeros(3*Natom, np.float64)

            for i in range(len(trans.graphVol)):
                ii = trans.graphVol[i]
                self.pos[3*ii] = trans.dvPos[3*i]
                self.pos[3*ii+1] = trans.dvPos[3*i+1]
                self.pos[3*ii+2] = trans.dvPos[3*i+2]
        else:
            self.defectsObject = None
            self.pos = np.empty(0, np.float64)

        self.newInBuildRev = False

[docs]class DummyDO:
    def __init__(self, trans):
        self.NVac = len(trans.vacancies)
        self.NInt = len(trans.interstitials)
        self.NAnt = len(trans.antisites)
        self.NDefects = self.NAnt + self.NInt + self.NVac
        self.cluster = np.zeros(self.NDefects, np.int32)

        self.vacancies = trans.vacancies
        self.interstitials = trans.interstitials
        self.antisites = trans.antisites
        self.onAntisites = trans.onAntisites

        self.vacPos = np.zeros(3*self.NVac, np.float64)
        self.intPos = np.zeros(3*self.NInt, np.float64)
        self.onAntPos = np.zeros(3*self.NAnt, np.float64)

        for i in range(self.NDefects):
            if i < self.NVac:
                self.vacPos[3*i] = trans.defectPos[3*i]
                self.vacPos[3*i+1] = trans.defectPos[3*i+1]
                self.vacPos[3*i+2] = trans.defectPos[3*i+2]
            elif i < self.NInt+self.NVac:
                ii = i - self.NVac
                self.intPos[3*ii] = trans.defectPos[3*i]
                self.intPos[3*ii+1] = trans.defectPos[3*i+1]
                self.intPos[3*ii+2] = trans.defectPos[3*i+2]
            else:
                ii = i - self.NVac - self.NInt
                self.onAntPos[3*ii] = trans.defectPos[3*i]
                self.onAntPos[3*ii+1] = trans.defectPos[3*i+1]
                self.onAntPos[3*ii+2] = trans.defectPos[3*i+2]
