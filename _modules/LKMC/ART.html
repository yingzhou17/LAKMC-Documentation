        """
        Run the dimer method.
        
        """

[docs]    def runMain(self, inputStateIni, fixSearchObj):
        """
        Main run routine.
        
        """
        lattice = copy.deepcopy(inputStateIni)
        
        climb = True
        climbCount = 0
        artParam = copy.deepcopy(self.params)
        params = artParam
        self.le_rVector = None
        self.le_rVectorFull = None
        self.numConsecNegEVs = 0
        
        # get Lanczos convergence minimisers
        self.saddleFinder = Lanczos.LanczosLBFGSSaddleFinder(self.params)
        self.saddleFinder.setJobID(self.jobID)
        
        # Stores initial positions of Atoms
        initialPos = copy.deepcopy(lattice.pos)
        
        # Calculates initial forces
        status = lattice.calcForce(defectFlag=2, correctTE=True)
        if status:
            log(__name__, "Failed on calcForce", 0)
            return None, None, None, "ART FAILED: CALC FORCE 1"
        
        energyOrig = lattice.totalEnergy
    
        # Checks if the defect list is defined
        if len(lattice.defectList) <= 0:
            log(__name__, "ART Failed: no defects", 0)
            return None, None, None, "ART FAILED: NO DEFECTS"
            
        # Estimates initial, min and max step sizes
        translationStep = artParam.artTranslationStep
        minStepSize = translationStep / 5.0
        maxStepSize = translationStep * 5.0
        
        if artParam.artDebugMode:
            randomNumber = 0.5
        else:
            randomNumber = random.random()
            
        initialStep = translationStep * randomNumber * 2.0
        
        assert lattice.NSubSystem == len(lattice.defectSubSystem)
        
        if type(fixSearchObj) is Utilities.TransitionFix:
            log(__name__, "Fixing search", 2, 1)
            lattice.displacementVector = fixSearchObj.displacementVector
            initialStep = fixSearchObj.stepSize
            self.le_rVectorFull = fixSearchObj.rVector
        
        # Gets the defects subsystem and initiates the first random move
        lattice.initiateDefectDisplacement(initialStep, artParam)
        
        # creating a vector to save the difference between original defect subsystem positions and new ones
        subSystemDiffVec = np.zeros(3*lattice.NSubSystem, np.float64)
            
        # Some initial parameters
        FdotFPrev = 1.0
        FPrevdotFPrevPrev = 0.0
          
        # Climbs up
        stepCount = 0
        while climb:          
            if ClientServerInterface.checkJobStatus(self.jobID):
                return None, None, None, "ART FAILED: JOB STATUS"
            
            # saving forces acting before translation
            FPrev = copy.deepcopy(lattice.force)
    
            # initiating atom movement
            if climbCount > 0:
                Vectors.applyPartialDisplacement(lattice.pos, translationStep * lattice.displacementVector, lattice.defectSubSystem)
            
            # Calculating forces on displaced lattice by translation step
            status = lattice.calcForce(defectFlag=2, correctTE=True)  
            if status:
                log(__name__, "Failed on calcForce", 0)
                return None, None, None, "ART FAILED: CALC FORCE 2"
            
            energy = lattice.totalEnergy
                    
            barrier = energy - energyOrig
            
            # Checks for displacements
            maxMoveAtomIdx, maxMove, avgMove, dist = Vectors.maxMovement(initialPos, lattice.pos, lattice.cellDims)
            
            # Gets the difference between original defect subsystem positions and new ones.
            subSystemDiffVec = Vectors.separationVectorSubSystem(initialPos, lattice.pos, lattice.defectSubSystem, lattice.cellDims, norm=True)
            
            # Calculates the dot product between vectors: position difference vector and force vector acting on defect subsystem atoms
            FdotRP = Vectors.partialDotProduct(lattice.force, subSystemDiffVec, lattice.defectSubSystem)
            
            # Makes new translation vector
            #TODO: write in C
            for i in xrange(lattice.NSubSystem):
                index = lattice.defectSubSystem[i]
                
                lattice.displacementVector[3*i] = lattice.force[3*index] - (1.0 + artParam.artAplha) * FdotRP * subSystemDiffVec[3*i]
                lattice.displacementVector[3*i+1] = lattice.force[3*index+1] - (1.0 + artParam.artAplha) * FdotRP * subSystemDiffVec[3*i+1]
                lattice.displacementVector[3*i+2] = lattice.force[3*index+2] - (1.0 + artParam.artAplha) * FdotRP * subSystemDiffVec[3*i+2]
            
            # Normalizes translation vector
            lattice.displacementVector = Vectors.normalise(lattice.displacementVector)
            
            # Checks if we need to increase/decrease translation step depending whether the forces increased
            FdotFPrev = Vectors.dotProduct(FPrev, lattice.force)
            
            
            absValue = math.fabs(FPrevdotFPrevPrev - FdotFPrev) / FdotFPrev
            
            if (absValue < 0.2):
                translationStep *= 1.2
            else:
                translationStep /= 2.0
            
            if translationStep > maxStepSize:
                translationStep = maxStepSize
            elif translationStep < minStepSize:
                translationStep = minStepSize
            
            FPrev = copy.deepcopy(lattice.force)
            FPrevdotFPrevPrev = FdotFPrev
            
            climbCount += 1
            
            log(__name__, "Step %d. Barrier %f; MaxMoveAtom %d; MaxAtomMove %f; StepSize %f; FdotRP %f" % (climbCount, barrier, maxMoveAtomIdx, 
                                                                                                           maxMove, translationStep, FdotRP), 2, 2)
            
            if barrier > artParam.artMaxBarrier or maxMove > artParam.artMaxMoveDistance or lattice.funcEval > artParam.artMaxFuncEval:
                climb = False
                
                if barrier > artParam.artMaxBarrier:
                    log(__name__, "Failure: barrier too high (%f > %f)" % (barrier, artParam.artMaxBarrier), 1, 3)
                    failReason = "ART FAILED: max barrier too high"
                
                elif maxMove > artParam.artMaxMoveDistance:
                    log(__name__, "Failure: max move too high (%f > %f)" % (maxMove, artParam.artMaxMoveDistance), 1, 3)
                    failReason = "ART FAILED: max move too far"
                
                elif lattice.funcEval > artParam.artMaxFuncEval:
                    log(__name__, "Failure: max move too high (%f > %f)" % (lattice.funcEval, artParam.artMaxFuncEval), 1, 3)
                    failReason = "ART FAILED: max function evaluations exceeded"
                
                else:
                    log(__name__, "Failed: unknown reason (this shouldn't happen)", 1, 3)
                    failReason = "ART FAILED: unknown reason"
                
                return None, None, None, failReason
            
            elif params.artUseLanczos and stepCount > params.artMinStepBeforeLanczos:
                lattice = copy.deepcopy(lattice)
                
                le_mainIndexes = lattice.defectSubSystem
                le_includeAtoms = lattice.buildSubLattice(le_mainIndexes, Globals.subLatticeIncludeRadius)
                
                if self.le_rVector is None and self.le_rVectorFull is not None:
                    log(__name__, "Using generated initial Lanczos Vector", 2, 3)
                    self.le_rVector = Vectors.extractVector3(le_mainIndexes, self.le_rVectorFull)
                
                status, lowestEV, resultVector, _, iterCnt, _, _ = Lanczos.lanczosMethodForASystem(lattice, le_mainIndexes, self.le_rVector, 
                                                                                                   self.params, normEigenVec=1, 
                                                                                                   includeAtoms=le_includeAtoms, returnMode=0)
                
                if status:
                    log(__name__, "Lanczos failed (looking for lowest eigenvalue)", 0, 0)
                    return None, None, None, "ART FAILED: Lanczos looking for lowest EV"
                
                if self.le_rVector is None or len(self.le_rVector) != 3 * len(le_mainIndexes):
                    self.le_rVector = np.empty(3 * len(le_mainIndexes), np.float64)
                
                # store eigenvector to use as initial guess at next step
                self.le_rVector = Vectors.extractVector3(le_mainIndexes, resultVector)
                self.le_rVectorFull = resultVector
                
                log(__name__, "Calculated lowest eigenvalue: %f (%d iterations)" % (lowestEV, iterCnt), 2, 3)
                
                # now we decide whether to start lanczos
                # is the eigenvalue really negative
                if lowestEV < params.artNegEVThreshold:
                    self.numConsecNegEVs += 1
                    
                    # have we seen enough negative eigenvalues in a row
                    if self.numConsecNegEVs >= params.artNumNegBeforeLanczos:
                        
                        
                        log(__name__, "Converging to saddle using Lanczos", 2, 1)
                        
                        # converge to saddle
                        climb = False
                        status = self.convergeToSaddle(lattice)
                        
                        if status:
                            return None, None, None, "ART FAILED: Lanczos converge failed"
                        
                        else:
                            status = lattice.calcForce(correctTE=1)
                            
                            if status:
                                return None, None, None, "ART FAILED: post Lanczos converge calcForce"
                            
                            else:
                                barrier = lattice.totalEnergy - energyOrig
                                return lattice, barrier, None, None
                                
                                log(__name__, "Success: Saddle point found! Barrier %f" % barrier, 2, 1)
                
                else:
                    self.numConsecNegEVs = 0
            
            elif FdotRP > artParam.artFdotRPvalue and maxMove > artParam.artMaxMoveCriteria:
                climb = False
                
                status = lattice.calcForce(correctTE=True)
                
                if status:
                    log(__name__, "Failed on calcForce", 0)
                    return None, None, None, "ART FAILED: CALC FORCE 3"
                
                finalBarrier = lattice.totalEnergy - energyOrig         
                
                finalSep = math.sqrt(math.pow(initialPos[maxMoveAtomIdx*3] - lattice.pos[maxMoveAtomIdx*3], 2) + 
                                     math.pow(initialPos[maxMoveAtomIdx*3+1] - lattice.pos[maxMoveAtomIdx*3+1], 2) + 
                                     math.pow(initialPos[maxMoveAtomIdx*3+2] - lattice.pos[maxMoveAtomIdx*3+2], 2))
                
                log(__name__, "Saddle point found! Barrier %f; Steps used: %d; MaxMoveAtom %d; MaxMoveAtomDistance %f;" % (finalBarrier, climbCount, 
                                                                                                                           maxMoveAtomIdx, finalSep), 1, 3)
                
                # return saddle and barrier and decide what to do with them elsewhere
                return lattice, finalBarrier, None, None
        
            stepCount += 1

    

[docs]    def prepareForConvergence(self, lattice):
        """
        Prepares rVector and index lists.
        
        """
        # not sure this is required!
        lattice.force[:] = 0.0
        
        # prepare main indexes and initial guess at eigenvector
        if self.params.saddleConvergeRadius > 0.0:
            mainIndexes = lattice.buildSubLattice(lattice.defectSubSystem, self.params.saddleConvergeRadius)
            lattice.rVector = Vectors.extractVector3(mainIndexes, self.le_rVectorFull)
        
        else:
            mainIndexes = copy.deepcopy(lattice.defectSubSystem)
            lattice.rVector = self.le_rVector
        
        # prepare include atoms
        includeAtoms = lattice.buildSubLattice(mainIndexes, Globals.subLatticeIncludeRadius)
        
        # store main indexes and include atoms on lattice object
        lattice.mainIndexes = mainIndexes
        lattice.includeAtoms = includeAtoms

    

[docs]    def convergeToSaddle(self, lattice):
        """
        Converge to saddle point using Lanczos/LBFGS
        
        """
        # minimise atoms that were held fixed during ART
        if self.params.lanczosMinOuterFirst:
            log(__name__, "Minimising atoms that were held fixed before Lanczos", 2, 0)

            prms = copy.deepcopy(self.params)
            prms.minimizationType = prms.lanczosMinOuterType
            prms.minimizationCriteria = prms.lanczosMinOuterCriteria
            prms.minimizationTolerance = prms.lanczosMinOuterTol
            prms.minimizationExclusion = 1
            prms.LBFGSTolerance = prms.lanczosMinOuterLBFGSTol
            
            inipos = copy.deepcopy(lattice.pos)
            
            outmin = Minimise.getMinimiser(prms, jobID=self.jobID)
            status = outmin.run(lattice)
            if status:
                log(__name__, "Min outer atoms before Lanczos failed (%d)" % status, 0, 0)
                return 134
            
            sep = Vectors.separation(inipos, lattice.pos, lattice.cellDims)
            log(__name__, "Separation after minimising outer atoms after Lanczos = %f" % sep, 2, 1)
        
        # prepare to converge to saddle
        self.prepareForConvergence(lattice)
        
        # converge to saddle
        status = self.saddleFinder.converge(lattice)
                     
        if status:
            log(__name__, "Failed (Lanczos-LBFGS saddle finder)", 1, 0)
        
        return status

[docs]def commandLineArgs():
    """
    Parse command line arguments
    
    """
    import argparse
    
    parser = argparse.ArgumentParser(description="Run the ART method.")
    
    parser.add_argument('inputFile', help="The input lattice file.")
    parser.add_argument('refFile', help="The reference lattice file.")
    parser.add_argument('-i', dest="paramFile", default="lkmcInput.IN", help="LKMC input file (default=lkmcInput.IN)")
    parser.add_argument('-s', dest="saddleFile", default=None, help="File for storing saddle state to (default=None)")
    parser.add_argument('-P', action="store_true", default=False, dest="profileCode", help="Profile the code using the hotshot module (not for production) (default=False)")
    parser.add_argument('-p', dest="statsFile", default=None, help="File for dumping profiling stats to (default=None)")
    
    return parser.parse_args()

[docs]def main():
    """
    Main routine from command line
    
    """
    progargs = commandLineArgs()
    
    params = Input.getLKMCParams(1, "", progargs.paramFile)
    Input.readGlobals(progargs.paramFile)
    
#     params.printInputParameters()
    
    initialState = Lattice.readLattice(progargs.inputFile)    

    # minimiser
    minimiser = Minimise.getMinimiser(params)
    
    # minimise initial
    status = minimiser.run(initialState)
    if status:
        sys.exit("MIN INI FAILED")
    
    referanceState = Lattice.readLattice(progargs.refFile)
    
    initialState.findDefects(referanceState, params)
    
    initialState.currentDefectVolume = initialState.defectVolumes[0]
    initialState.currentVolumeIndex = 0
    
    runner = ARTNMethod(params)
    
    if progargs.profileCode:
        import Profiling
        
        args = (initialState,)
        kwargs = {}
        stats, result = Profiling.profileMethod(runner.run, args, kwargs, printStats=1, statsFile=progargs.statsFile)
        
        saddle = result[0]
        barrier = result[1]
        time = result[2]
        
    else:
        result = runner.run(initialState)
        
        saddle = result[0]
        barrier = result[1]
        time = result[2]
    
    if saddle is not None:
        print "SUCCESS: BARRIER: %f" % barrier
    
    if saddle is not None and progargs.saddleFile is not None:
        saddle.writeLattice(progargs.saddleFile)
