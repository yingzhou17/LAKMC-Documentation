    For running external events during LKMC
    
    Parameters
    ----------
    params : `Input.InputParams` object
        This object contains all parameters which are read from ``lkmcInput.IN`` by default.
    refState : `Lattice.Lattice` object

    def __init__(self, params, refState, jobID=None):
        # store the parameters and reference state
        self.params = params
        self.refState = refState
        self.jobID = jobID
        
        # required parameters
        self.forceLibrary = None
        self.minAfter = None
        self.minUseLAMMPS = None
        self.minFailUseLAMMPS = None
        self.minStyle = None
        self.maxEval = None
        
        # lammps object
        if self.params.MDType == "LAMMPS":
            self._lmp = Lammps.LammpsInterface(params)
        else:
            self._lmp = None
    

[docs]    def getVelocityFromKE(self, energy, direction=None, mass=None, sym=None):
        """
        Returns the velocity for the given kinetic energy and species.
        
        Parameters
        ----------
        energy : float
            The kinetic energy of the atom in eV.
        direction : `numpy.ndarray`, optional
            A vector of length 3 defining the direction of the atom. If specified the returned value will also be a
            vector of length 3 containing the components of velocity, otherwise the magnitude of the velocity is
            returned
        sym : str[2], optional
            The symbol of the species, eg. 'H_' or "Fe". One of `sym` or `mass` must be specified. Mass is used if both
            are given.
        mass : float, optional
            The total mass of the atom(s) you are calculating the velocity for. One of `sym` or `mass` must be
            specified. Mass is used if both are given.
        
        Returns
        -------
        velocity : float or `numpy.ndarray`
            Depending on the value of `direction` this is the magnitude of the velocity of a vector containing the
            components of velocity. The units depend on the MD library. For LBOMD it will be Å/fs, for LAMMPS it will
            be Å/ps (assuming metal units). Support for other LAMMPS units can also be added.
        
        Raises
        ------
        LammpsUnitsNotSupported
            If the value of the lammpsUnits parameter is not supported.
        
        """
        if sym is None and mass is None:
            raise ValueError("Either sym or mass must be specified")
        
        # parameters
        params = self.params
        
        if mass is None:
            # the mass of the atom, in LBOMD units
            mass = Atoms.atomicMass(sym)
        
        # the magnitude of the velocity, in Å/fs
        velocityMag = math.sqrt(2.0 * energy / mass)
        
        # modify velocity for LAMMPS
        if params.MDType == "LAMMPS":
            if params.lammpsUnits == "metal":
                # convert to Å/ps
                velocityMag *= 1e3
            else:
                raise Lammps.LammpsUnitsNotSupported("LAMMPS '%s' units have not been implemented" % params.lammpsUnits)
        
        # construct result depending on direction kwarg
        if direction is None:
            velocity = velocityMag
        
        else:
            # modulus of direction indices
            direction = np.asarray(direction, dtype=np.float64)
            mod = Vectors.magnitude(direction)
            if mod == 0.0:
                raise RuntimeError("Supplied vector has magnitude zero")
            velocity = np.zeros(3, np.float64)
            velocity[0] = velocityMag * direction[0] / mod
            velocity[1] = velocityMag * direction[1] / mod
            velocity[2] = velocityMag * direction[2] / mod
            
            # check
            mag = Vectors.magnitude(velocity)
            ke = 0.5 * mass * mag * mag
            if params.MDType == "LAMMPS" and params.lammpsUnits == "metal":
                ke /= 1e6
            if math.fabs(ke - energy) > 1e-4:
                raise RuntimeError("KE is wrong (%f not %f as expected) %r" % (ke, energy, velocity))
        
        return velocity

    

[docs]    def minimise(self, lattice, minAfter, minUseLAMMPS, minFailUseLAMMPS):
        """
        Minimise the system.
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The lattice to be minimised
        minAfter : `True` or `False`
            If `True` we perform the minimisation
        minUseLAMMPS : `True` or `False`
            Minimisate the system using LAMMPS
        minFailUseLAMMPS : `True` or `False`
            If the minimisation fails, try to minimise using LAMMPS
        
        """
        params = self.params
        status = True
        if minAfter and minUseLAMMPS and params.MDType == "LAMMPS":
            log(__name__, "Minimising lattice after external event (using LAMMPS)", 2, 0)
            
            # minimise using LAMMPS
            # TODO: test is initialise=False works (it should)
            status = self._lmp.minimise(lattice, initialise=True)
        
        elif minAfter:
            # Minimise using LKMC
            log(__name__, "Minimising lattice after external event", 2, 0)
            
            # run the minimisation
            minimiser = Minimise.getMinimiser(self.params)
            status = minimiser.run(lattice)
            
            # If LKMC failed, try CGSCIPY
            # TODO: this should not be hardcoded
            if status and self.params.minimizationType != "CGSCIPY":
                log(__name__, "Minimisation using LKMC %s failed, try CGSCIPY." % (self.params.minimizationType), 1, 2)
                minimiser = Minimise.ConjugateGradientSciPyMinimiser(self.params)
                status = minimiser.run(lattice)
            
            # minimise using LKMC failed. Try LAMMPS
            if status and minFailUseLAMMPS and params.MDType == "LAMMPS":
                status = self._lmp.minimise(lattice)
        
        if status:
            raise EventMinimisationFailed("Failed to minimise the lattice post external event")

    

[docs]    def runMD(self, lattice, velocity, thermType, runTime):
        """
        Run a molecular dynamics simulation.
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The lattice to run the simulation in.
        velocity : `numpy.ndarray`
            Array containing the initial velocities of the atoms.
        thermType : str
            The type of thermalisation (e.g. BERENDSEN, NO_THERMOSTAT)
        runTime : float
            The run time of the simulation (fs).
        
        Raises
        ------
        ValueError
            If the force library specified in the input parameters is not recognised.
        
        Notes
        -----
        Certain parameters, such as `thermType` will be passed directly to the forces library without being checked. It
        is up to the user to ensure they were specified correctly in the input file.
        
        """
        if self.params.MDType.startswith("LBOMD"):
            self._runMDLbomd(lattice, velocity, thermType, runTime)
        elif self.params.MDType == "LAMMPS":
            self._runMDLammps(lattice, velocity, thermType, runTime)
        else:
            raise ValueError("Unrecognised force library for running MD: '{0}'".format(self.params.MDType))

    
    def _runMDLbomd(self, lattice, velocity, thermType, runTime):
        """
        Run an MD simulation using LBOMD.
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The lattice to run the simulation in.
        velocity : `numpy.ndarray`
            Array containing the initial velocities of the atoms.
        thermType : str
            The type of thermalisation (e.g. BERENDSEN, NO_THERMOSTAT)
        runTime : float
            The run time of the simulation (fs).
        
        Raises
        ------
        EventRunMDFailed
            If running the MD simulation failed.
        
        """
        log(__name__, "Running MD simulation using LBOMD", 1, 0)
        
        # parameters
        temperature = self.params.systemTemperature
        fileOutput = True
        
        # run the MD simulation
        status = Lbomd.runMD(lattice, runTime, temperature, thermType, initialVelocity=velocity, fileOutput=fileOutput)
        if status:
            raise EventRunMDFailed("Lbomd.runMD failed with status: %d" % status)
    
    def _runMDLammps(self, lattice, velocity, thermType, runTime):
        """
        Run an MD simulation using LAMMPS.
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The lattice to run the simulation in.
        velocity : `numpy.ndarray`
            Array containing the initial velocities of the atoms.
        runTime : float
            The run time of the simulation (fs).
        thermType : str
            The type of thermalisation (e.g. BERENDSEN, NO_THERMOSTAT)
        runTime : float
            The run time of the simulation (fs).
        
        Raises
        ------
        EventRunMDFailed
            If running the MD simulation failed.
        
        """
        log(__name__, "Running MD simulation using LAMMPS", 1, 0)
        
        # run LAMMPS
        self._lmp.runMD(lattice, velocity, runTime, thermType)
    

[docs]    def switchToOutputDirectory(self, step=None, outputDirectory=None):
        """
        Create and switch to the output directory.
        
        Parameters
        ----------
        step : int, optional
            Current KMC step, to allow changing into the appropriate working directory.
        outputDirectory : str, optional
            Path to the desired output directory. If specified this overrides the default location.
        
        Returns
        -------
        root : str
            The original working directory.
        
        """
        # Record current working directory
        root = os.getcwd()
        
        # user specified output directory
        if outputDirectory is not None:
            stepPath = outputDirectory
        # default location
        else:
            # determine the path to the output directory
            stepPath = os.path.join(root, self.params.outputDirectory)
            if self.params.eventStoreMDOutput:
                if step is not None:
                    stepPath = os.path.join(stepPath, "Step%d" % step)
                stepPath = os.path.join(stepPath, "md_files")
            else:
                stepPath = os.path.join(stepPath, "last_md_files")
        
        # create the directory if it doesn't exist and switch to it
        log(__name__, "Switching to output directory: '%s'" % stepPath, 2, 0)
        if not os.path.exists(stepPath):
            os.makedirs(stepPath, 0755)
        os.chdir(stepPath)
        
        return root

    

[docs]    def thermalise(self, lattice, thermaliseFlag, fullTime, constrainedTime, thermType):
        """
        Thermalise the lattice if required, otherwise return zeroes for velocities.
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The lattice to be thermalised.
        thermaliseFlag : `True` or `False`
            Whether to thermalise the lattice or not.
        fullTime : float
            Thermalisation time for the full system
        constrainedTime : float
            Thermalisation time for the constrained system
        themType : str
            The type of thermalisation (e.g. BERENDSEN)
        
        Returns
        -------
        velocity : `numpy.ndarray`
            Array containing the velocities of the atoms.
        
        Raises
        ------
        ValueError
            If the force library specified in the input parameters is not recognised.
        
        """
        if thermaliseFlag:
            log(__name__, "Thermalising lattice", 2, 0)
            
            # thermalise the lattice
            if self.params.MDType.startswith("LBOMD"):
                velocity = self._thermaliseLBOMD(lattice, fullTime, constrainedTime, thermType)
            elif self.params.MDType == "LAMMPS":
                velocity = self._thermaliseLAMMPS(lattice, fullTime, constrainedTime)
            else:
                raise ValueError("Unrecognised force library for thermalisation: '{0}'".format(self.params.MDType))
        
        else:
            # set velocities to zero
            velocity = np.zeros(len(lattice.pos), dtype=np.float64)
        
        return velocity

    
    def _thermaliseLBOMD(self, lattice, fullTime, constrainedTime, thermType):
        """
        Thermalise lattice before event
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The lattice to be thermalised.
        fullTime : float
            Thermalisation time for the full system
        constrainedTime : float
            Thermalisation time for the constrained system
        themType : str
            The type of thermalisation (e.g. BERENDSEN)
        
        Returns
        -------
        velocity : `numpy.ndarray`
            Array containing the velocities of each atom in the system after thermalization.
        
        Raises
        ------
        EventThermalisationFailed
            If the lattice could not be thermalised
        
        """
        temperature = self.params.systemTemperature
        log(__name__,
            "Thermalising system to %.0f K using the thermostat type %s." % (temperature, thermType),
            1, 2)
        
        # thermalise
        status, velocity = Lbomd.runThermalisation(lattice, fullTime, constrainedTime, temperature,
                                                   thermType, fileOutput=False)
        
        # raise an exception here
        if status:
            raise EventThermalisationFailed("LBOMD thermalisation failed (%d)" % status)
        
        return velocity
    
    def _thermaliseLAMMPS(self, lattice, fullTime, constrainedTime):
        """
        Thermalise the lattice before simulation.
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The lattice to be thermalised.
        fullTime : float
            Thermalisation time for the full system
        constrainedTime : float
            Thermalisation time for the constrained system
        
        Returns
        -------
        lost : int
            Number of atoms lost during the thermalization.
        
        """
        temperature = self.params.systemTemperature
        log(__name__, "Thermalising system to %.0f K (LAMMPS)." % temperature, 1, 2)
        
        # run the thermalisation
        velocity = self._lmp.thermalise(lattice, fullTime, constrainedTime, temperature)
        
        return velocity
