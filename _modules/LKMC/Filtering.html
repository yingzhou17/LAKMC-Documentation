[docs]def applyHashKeyFilter(initial, final):
    result = 1

    hashKeyBlackList = ['e64a51d_9e49b3c_3e6db7e']

    iniHashLen = len(initial.defectVolumesKeys)
    finHashLen = len(final.defectVolumesKeys)

    for i in range(iniHashLen):
        if initial.defectVolumesKeys[i] in hashKeyBlackList:
            for j in range(finHashLen):
                if initial.defectVolumesKeys[i] == final.defectVolumesKeys[j]:
                    massCentreSep = 0

                    for k in range(3):
                        massCentreSep += (initial.defectVolumes[i].com[k] - final.defectVolumes[j].com[k])**2

                        #initial.graphVolsCentreMass[i][k] - final.graphVolsCentreMass[j][k]

                    if massCentreSep < 9.0:
                        result = 0
                        break

                if not result:
                    break

        if not result:
            break

    return result



[docs]def filterResult(initial, result, params):
    """
    Filter individual result

    """
    accept = True

    #TODO: this is something to do with Tomas' work and should probably be deleted now...
    if initial is not None and result.final is not None:
        accept = applyHashKeyFilter(initial, result.final)

    # minimum barrier
    if result.barrier < params.filterMinimumBarrier and result.barrier is not None:
        accept = False
    if result.reverseBarrier < params.filterMinimumBarrier and result.reverseBarrier is not None:
        accept = False

    return accept
