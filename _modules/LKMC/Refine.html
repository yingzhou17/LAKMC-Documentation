[docs]    def setup(self, initialState=None):
        """
        Setup the transition refiner.

        """

        # minimiser
        minParams = copy.deepcopy(self.params)
        minParams.LBFGSTolerance = self.params.refineFinalMinTol
        self.minimiser = Minimise.getMinimiser(minParams)

        # inclusive minimiser
        inclMinParams = copy.deepcopy(self.params)
        inclMinParams.minimizationExclusion = 1
        inclMinParams.LBFGSTolerance = self.params.refineFinalInclMinTol
        self.minimiserIncl = Minimise.getMinimiser(inclMinParams)

        # exclusive minimiser
        exclMinParams = copy.deepcopy(self.params)
        exclMinParams.minimizationExclusion = 2
        exclMinParams.LBFGSTolerance = self.params.refineFinalExclMinTol
        self.minimiserExcl = Minimise.getMinimiser(exclMinParams)

        # NEB refiner
        self.barrierCalculator = None
        if len(self.params.refineBarrierEstimator.strip()):
            if self.params.refineBarrierEstimator == "NEB":
                self.barrierCalculator = NEB.NEB(self.params)

            elif self.params.refineBarrierEstimator == "String":
                self.barrierCalculator = String.String(self.params)

            else:
                log(__name__, "Warning: unrecognised barrier estimator (refinement): '%s'" % self.params.refineBarrierEstimator, 0, 0)

        # Lanczos refiners
        LanczosParams = Lanczos.initialiseLanczosMinParams(self.params)
        LanczosExclParams = Lanczos.initialiseLanczosMinParams(self.params, subLatticeMd=1)

        self.LanczosRefiner = Lanczos.LanczosLBFGSMinimiser(LanczosParams)
        self.LanczosExclRefiner = Lanczos.LanczosLBFGSMinimiser(LanczosExclParams)

        # Dimer rotation minimiser
        dimeRotMinParams = copy.deepcopy(self.params)
        dimeRotMinParams.dimerTolR = dimeRotMinParams.refineDimerMinTolR
        self.dimerRotMinimiser = Dimer.Dimer(dimeRotMinParams)

        # Saddle finder
        self.saddleFinder = Lanczos.LanczosLBFGSSaddleFinder(self.params)

        # initial state
        self.initialState = initialState

        # create list of hooks
        self.postTransitionHookList = hook_utils.makeHookList("postFoundTransitionHooks", self.params)

        self.setupCompleted = True

            #status = self.NEBRefiner.run(initial, final, saddle)

[docs]    def checkCalcPrefactor(self, barrier):
        """
        Check if we need to calculate the prefactor for this transition.

        """
        if self.ratesListLock is None:
            return None

        checkTime = time.time()

        # get lock on rates list
        self.ratesListLock.acquire()

        ratio = None
        try:
            # get the current total rate
            currentTotalRate = self.ratesListProxy[0]
            print "DEBUGPREFACTOR: CURRENT TOTAL RATE = %f" % currentTotalRate

            # now estimate the rate of this process using standard prefactor
            estimatedEventRate = Rates.calculateRate(barrier, None, None, self.params,
                calcPrefactor=False, jobID=self.jobID)

            print "DEBUGPREFACTOR: estimated event rate = %f" % estimatedEventRate

            # calculate ratio of estimated to total
            if currentTotalRate == 0.0:
                ratio = 1.0
            else:
                ratio = estimatedEventRate / currentTotalRate
            print "DEBUGPREFACTOR: ratio = %g" % ratio

        finally:
            self.ratesListLock.release()

        checkTime = time.time() - checkTime

        print "DEBUGPREFACTOR: time taken to check: %f s" % checkTime

        return ratio, checkTime

[docs]    def createFailResult(self, linkedIdx, storeTimings=True):
        """
        Create fail result.

        """

        refineElapsedTime = time.time() - self.startTime

        if storeTimings:
            self.methodRunList.append("REFINEMENT")
            self.methodTimeList.append(refineElapsedTime)
            self.methodSuccList.append(0)

            timeLog = ClientServerInterface.TimeLog(self.methodRunList,
                                                    self.methodTimeList,
                                                    self.methodSuccList)

        else:
            timeLog = None

        result = ClientServerInterface.Result(self.rank, "BARRIERREFINE", "FAILED",
                                              timing=timeLog, linkedTransIndex=linkedIdx,
                                              currentVolIndex=self.currentVolIndex, jobID=self.jobID)

        return result

                            reuseWorkVolId=None, linkedBarrier=None, calcPrefactor=None, rate=None,
                            saddleSumLogEVs=None):

        """
        Create success result.

        """
        # first run any hooks
        args = ()
        kwargs = {"initial": self.initialState, "saddle": saddle, "final": final, "barrier": refinedBarrier, "params": self.params,

        for hook in self.postTransitionHookList:
            hookres = hook.run(*args, **kwargs)

            if hookres:
                print "DEBUGGING: throwing away transition (hook %s)" % str(hook)
                return self.createFailResult(linkedIdx, storeTimings=False)

        refineElapsedTime = time.time() - self.startTime

        self.methodRunList.append("REFINEMENT")
        self.methodTimeList.append(refineElapsedTime)
        self.methodSuccList.append(1)

        timeLog = ClientServerInterface.TimeLog(self.methodRunList,
            self.methodTimeList, self.methodSuccList)

        result = ClientServerInterface.Result(self.rank, "BARRIERREFINE", "SUCCESS", timing=timeLog, barrier=refinedBarrier, final=final,
            images=images, saddle=saddle, linkedTransIndex=linkedIdx, reuseHashKey=reuseHashKey,
            reuseWorkVolId=reuseWorkVolId, linkedBarrier=linkedBarrier, currentVolIndex=self.currentVolIndex,
            jobID=self.jobID, calcPrefactor=calcPrefactor, rate=rate, reverseBarrier=reverseBarrier, saddleSumLogEVs=saddleSumLogEVs)

        return result
