[docs]def runThermalisation(lattice, fullSystemTime, constrainedSystemTime, temperature, thermostatType, fileOutput=False):
    """
    Thermalise a lattice using the LBOMD interface
    
    'fileOutput' determines whether lbomd output files are written (xyz, failsafe, cascades, thermalise...)
    
    NOTE: this interface is likely to change!
    
    """
    # check interface loaded
    if not lbomdLoaded:
        log(__name__, "Interface not loaded, cannot thermalise lattice", 0, 0)
        return 1, None
    
    # check interface version
    interfaceVersion = getInterfaceVersion()
    if interfaceVersion < 140124:
        log(__name__, "LBOMD interface too old to run thermalisation (%d < %d)" % (interfaceVersion, 140124), 0, 0)
        return 2, None
    
    log(__name__, "Thermalising lattice", 1, 0)
    
    # initialise velocity array
    vel = np.zeros(3 * lattice.NAtoms, np.float64)
    
    # create specie string containing specie list to pass to fortran
    specieList = lattice.specieList
    specieString = "".join(specieList)
    
    # run
    status = LBOMD.python_interface.run_thermalisation(fullSystemTime, constrainedSystemTime, temperature,
                                                       thermostatType, specieString, lattice.specie, lattice.pos,
                                                       lattice.charge, vel, lattice.cellDims[0],
                                                       lattice.cellDims[4], lattice.cellDims[8], fileOutput,
                                                       Globals.MDLocation)
    
    if status:
        print "ERROR: lattice thermalisation failed with status: %d" % status
        vel = None
    
    return status, vel




[docs]def getPKAVelocity(pkaSym, energy, pkaDirection):
    """
    Return PKA velocity for given energy and direction
    
    """
    mass = Atoms.atomicMass(pkaSym)
    
    velocityMag = math.sqrt(2.0 * energy / mass)
    
    # modulus of direction indices
    pkaDirection = np.asarray(pkaDirection, dtype=np.float64)
    mod = Vectors.magnitude(pkaDirection)
    if mod == 0.0:
        log(__name__, "getPKAVelocity error: modulus of pkaDirection == 0", 0, 0)
        return
    
    velocity = np.zeros(3, np.float64)
    velocity[0] = velocityMag * pkaDirection[0] / mod
    velocity[1] = velocityMag * pkaDirection[1] / mod
    velocity[2] = velocityMag * pkaDirection[2] / mod
    
    # check
    mag = Vectors.magnitude(velocity)
    ke = 0.5 * mass * mag * mag
    assert math.fabs(ke - energy) < 1e-4, "getPKAVelocity error: ke is wrong (%f not %f) (%r)" % (ke, energy, velocity)
    
    return velocity




[docs]def addPKA(lattice, energy, pkaDirection, approxPosition=None, approxPositionRadius=None, pkaIndex=None, inputVelocity=None, pkaSymbol=None):
    """
    Add PKA near given coordinates or at atom index.
    
    """
    if inputVelocity is not None and len(inputVelocity) != 3 * lattice.NAtoms:
        log(__name__, "addPKA error: input velocity wrong length (%d != %d)" % (len(inputVelocity), 3 * lattice.NAtoms), 0, 0)
        return
    
    elif inputVelocity is None:
        inputVelocity = np.zeros(3 * lattice.NAtoms, np.float64)
    
    if pkaIndex is not None:
        if pkaIndex >= lattice.NAtoms:
            log(__name__, "addPKA error: specified atom index is out of range (%d >= %d)" % (pkaIndex, lattice.NAtoms), 0, 0)
            return
    
    elif approxPosition is not None and len(approxPosition) == 3:
        if approxPositionRadius is None or approxPositionRadius <= 0:
            closestSep = None
            closestIndex = None
            for i in xrange(lattice.NAtoms):
                if pkaSymbol is not None and lattice.atomSym(i) != pkaSymbol:
                    continue
                
                sep = Vectors.separation(lattice.atomPos(i), approxPosition, lattice.cellDims)
                
                if closestSep is None or sep < closestSep:
                    closestSep = sep
                    closestIndex = i
            
            if closestIndex is None:
                log(__name__, "addPKA error: could not find atom close to position %r (specie %s)" % (approxPosition, pkaSymbol), 0, 0)
                return
            
            pkaIndex = closestIndex
            log(__name__+".addPKA", "Selected atom closest to %r: index %d" % (approxPosition, pkaIndex), 2, 0)
        
        else:
            possibleIndexes = []
            for i in xrange(lattice.NAtoms):
                if pkaSymbol is not None and lattice.atomSym(i) != pkaSymbol:
                    continue
                
                sep = Vectors.separation(lattice.atomPos(i), approxPosition, lattice.cellDims)
                if sep < approxPositionRadius:
                    possibleIndexes.append(i)
            
            if not len(possibleIndexes):
                log(__name__, "addPKA error: could not find atoms within %.1f of position %r" % (approxPositionRadius, approxPosition), 0, 0)
                return
            
            pkaIndex = random.choice(possibleIndexes)
            log(__name__+".addPKA", "Selected random atom near to %r (%.1f): index %d (out of %d)" % (approxPosition, approxPositionRadius,
                                                                                                      pkaIndex, len(possibleIndexes)), 2, 0)
    
    else:
        log(__name__, "addPKA error: you must specify pkaIndex or approxPosition", 0, 0)
        return
    
    log(__name__+".addPKA", "Adding PKA: index %d; energy %f eV; direction (%f, %f, %f)" % (pkaIndex, energy, pkaDirection[0], pkaDirection[1], pkaDirection[2]), 1, 0)
    
    pkaVelocity = getPKAVelocity(lattice.atomSym(pkaIndex), energy, pkaDirection)
    
    # set velocity
    inputVelocity[3*pkaIndex] = pkaVelocity[0]
    inputVelocity[3*pkaIndex+1] = pkaVelocity[1]
    inputVelocity[3*pkaIndex+2] = pkaVelocity[2]
    
    return inputVelocity

[docs]    def callback(self, simTime, currentPos, currentTemperature, numAtoms):
        """
        Callback method
        
        """
        log(__name__, "Defects Cascade Callback...", 3, 0)
        
        # current positions
        self.lattice.pos = currentPos
        
        # find defects
        defectsObj = self.lattice.findDefects(self.ref, self.params, verbose=self.verbose)
        
        # write result
        if self.filename is not None:
            f = open(self.filename, "a")
            f.write("%f, %f, %d, %d, %d, %d\n" % (simTime, currentTemperature, defectsObj.NDefects, defectsObj.NVac, 
                                                  defectsObj.NInt, defectsObj.NAnt))
            f.close()
        
        # send result
        if self.resultQ is not None:
            self.resultQ.put((self.simulationID, simTime, currentTemperature, defectsObj.NDefects, defectsObj.NVac, 
                              defectsObj.NInt, defectsObj.NAnt))

[docs]def runMD(lattice, runTime, temperature, thermostatType, initialVelocity=None, fileOutput=False, writeReference=True, callbackObj=None):
    """
    Run an MD simulation.
    
    'fileOutput' determines whether lbomd output files are written (xyz, failsafe, cascades, thermalise...)
    'writeReference' forces a reference file to be written (this option is ignored if fileOutput=False)
    
    If no initial velocity is passed the simulation will run at 0 K.
    
    NOTES:
        - this interface is likely to change!
        - does not read collisions.IN (you need to set initial velocities to run cascade)
    
    """
    # check interface loaded
    if not lbomdLoaded:
        log(__name__, "Interface not loaded, cannot run MD", 0, 0)
        return
    
    # check interface version
    interfaceVersion = getInterfaceVersion()
    if interfaceVersion < 140808:
        log(__name__, "LBOMD interface too old to run MD (%d < %d)" % (interfaceVersion, 140808), 0, 0)
        return
    
    log(__name__, "Running MD simulation", 1, 0)
    
    # check if velocities passed (correctly)
    if initialVelocity is None or len(initialVelocity) != 3 * lattice.NAtoms:
        log(__name__, "Setting initial velocities to zero", 1, 1)
        
        # initialise velocity array
        vel = np.zeros(3 * lattice.NAtoms, np.float64)
    
    else:
        vel = initialVelocity
    
    # callback
    if callbackObj is None:
        callbackObj = CascadeCallbackDefault()
    
    # create specie string containing specie list to pass to fortran
    specieList = lattice.specieList
    specieString = "".join(specieList)
    
    # run
    status = LBOMD.python_interface.run_md(runTime, temperature, thermostatType, specieString, lattice.specie, lattice.pos, 
                                           lattice.charge, vel, lattice.cellDims[0], lattice.cellDims[4], lattice.cellDims[8], 
                                           fileOutput, writeReference, Globals.MDLocation, callbackObj.callback)
    
    if status:
        print "ERROR: run MD failed with status: %d" % status
    
    return status




[docs]def runDampedMD(lattice, mode, runTime, forceTolerance, initialVelocity=None, fileOutput=False, writeReference=True):
    """
    Run an damped MD simulation.
    
    Parameters
    ----------
    lattice : Lattice.Lattice
        The input lattice object to run the simulation on.
    mode : 0 or 1
        Set to 0 to run for a fixed time or 1 to run until the force tolerance is reached.
    runTime : float
        The time to run the simulation for (assuming `mode` is set to 0).
    forceTolerance : float
        The force tolerance defining the top condition of the simulation (assuming `mode` is set to 1).
    initialVelocity : np.ndarray, optional
        Optional. The initial velocities for the simulation. The default is 0.
    fileOutput : bool
        Determines whether lbomd output files are written (xyz, failsafe, cascades, thermalise...).
    writeReference : bool
        Forces a reference file to be written (ignored if `fileOutput` is False).
    
    Notes
    -----
    This interface is likely to change!
    
    """
    # check interface loaded
    if not lbomdLoaded:
        log(__name__, "Interface not loaded, cannot thermalise lattice", 0, 0)
        return
    
    # check interface version
    interfaceVersion = getInterfaceVersion()
    if interfaceVersion < 160705:
        log(__name__, "LBOMD interface too old to run damped MD (%d < %d)" % (interfaceVersion, 160705), 0, 0)
        return
    
    if mode == 0:
        log(__name__, "Running damped MD simulation for %.1f fs" % runTime, 1, 0)
    else:
        log(__name__, "Running damped MD simulation until force tolerance %e reached" % forceTolerance, 1, 0)
    
    # check if velocities passed (correctly)
    if initialVelocity is None or len(initialVelocity) != 3 * lattice.NAtoms:
        log(__name__, "Setting initial velocities to zero", 1, 1)
        
        # initialise velocity array
        vel = np.zeros(3 * lattice.NAtoms, np.float64)
    
    else:
        vel = initialVelocity
    
    # create specie string containing specie list to pass to fortran
    specieList = lattice.specieList
    specieString = "".join(specieList)
    
    # run
    status = LBOMD.python_interface.run_damped_md(mode, runTime, forceTolerance, specieString, lattice.specie,
                                                  lattice.pos, lattice.charge, vel, lattice.cellDims[0],
                                                  lattice.cellDims[4], lattice.cellDims[8], fileOutput, writeReference,
                                                  Globals.MDLocation)
    
    if status:
        print "ERROR: run damoed MD failed with status: %d" % status
    
    return status




[docs]def readLBOMDRef(filename, storeEnergies=1):
    """
    Read LBOMD reference file.
    
    """
    log(__name__, "reading LBOMD ref: %s" % (filename,), 2)
    
    # check for zip
    zipFlag, filename = Utilities.checkForZipped(filename)
    
    # open file
    f = open(filename)
    
    # get num atoms
    NAtoms = int(f.readline().strip())
    log(__name__, "%d atoms" % (NAtoms,), 2, 1)
    
    # lattice object
    lattice = Lattice.Lattice(NAtoms, storeEnergies=storeEnergies)
    
    # dimensions
    line = f.readline().strip()
    array = line.split()
    lattice.cellDims[0] = float(array[0])
    lattice.cellDims[4] = float(array[1])
    lattice.cellDims[8] = float(array[2])
    
    f.close()
    
    # temporary specie list and counter arrays
    maxNumSpecies = 20 ## if there are more than 20 species these must be changed
    dt = np.dtype((str, 2))
    specieListTemp = np.empty( maxNumSpecies+1, dt ) 
    specieCountTemp = np.zeros( maxNumSpecies+1, np.int32 )
    
    tmpForceArray = np.empty(3, np.float64)
    
    # call c lib
    c_io.readLBOMDRef(filename, lattice.specie, lattice.pos, lattice.charge, lattice.KE, lattice.PE, tmpForceArray, 
                      specieListTemp, specieCountTemp, lattice.maxPos, lattice.minPos, lattice.storeEnergies)
    
    if zipFlag:
        os.remove(filename)
    
    # build specie list and counter in lattice object
    NSpecies = 0
    for i in range(maxNumSpecies):
        if specieListTemp[i] == 'XX':
            break
        else:
            NSpecies += 1
            
    # add species to specie lists
    for i in xrange(NSpecies):
        lattice.addSpecie(specieListTemp[i], count=specieCountTemp[i])
    
        log(__name__, "%d %s (%s) atoms" % (specieCountTemp[i], specieListTemp[i], Atoms.atomName(specieListTemp[i])), 2, 2)
    
    return lattice




[docs]def readLBOMDXYZ(filename, refLattice=None, storeEnergies=1):
    """
    Read LBOMD XYZ file.
    
    """
    log(__name__, "reading LBOMD XYZ: %s" % (filename,), 2)
    
    # check for zip
    zipFlag, filename = Utilities.checkForZipped(filename)
    
    # open file
    f = open(filename)
    
    # get num atoms and time
    NAtoms = int(f.readline().strip())
    log(__name__, "%d atoms" % (NAtoms,), 2, 1)
    
    simTime = float(f.readline().strip())
    log(__name__, "simulation time: %f" % (simTime,), 2, 1)
    
    # read first line to get format
    line = f.readline().strip()
    array = line.split()
    if len(array) == 6:
        xyzformat = 0
        
    elif len(array) == 7:
        xyzformat = 1
    
    else:
        sys.exit("ERROR: unrecognised LBOMD format: %s" % line)
    
    f.close()
    
    # lattice structure
    lattice = Lattice.Lattice(NAtoms, storeEnergies=storeEnergies)
    
    # store time
    lattice.simTime = simTime
    
    tmpForceArray = np.empty(3, np.float64)
    
    # call clib
    c_io.readLBOMDXYZ(filename, lattice.pos, lattice.charge, lattice.KE, lattice.PE, tmpForceArray, 
                      lattice.maxPos, lattice.minPos, xyzformat, lattice.storeEnergies)
    
    if zipFlag:
        os.remove(filename)
    
    if refLattice is not None:
        # copy charge if not included in xyz and specie
        if xyzformat == 0:
            lattice.charge[:] = refLattice.charge[:]
        
        lattice.specie[:] = refLattice.specie[:]
        
        # copy specie arrays from refLattice
        lattice.specieList = copy.deepcopy(refLattice.specieList)
        lattice.specieCount = copy.deepcopy(refLattice.specieCount)
        lattice.specieMass = copy.deepcopy(refLattice.specieMass)
        lattice.specieCovalentRadius = copy.deepcopy(refLattice.specieCovalentRadius)
        lattice.specieRGB = copy.deepcopy(refLattice.specieRGB)
        lattice.cellDims = copy.deepcopy(refLattice.cellDims)
    
    return lattice




[docs]def getLBOMDIdentity(index=0):
    """
    Attempts to find the LBOMD XYZ identity associated with 
    the xyz file with index: ie xxxxxINDEX.xyz
    
    """
    files = glob.glob("*0000.xyz*")
    
    if not len(files):
        return 0
    
    if len(files) > 1:
        print "WARNING: multiple identities: taking first", files
    
    simIdentity = files[0][:5]
    
    return simIdentity




[docs]def findFinalXYZ(identity):
    """
    Find the final XYZ file and index
    corresponding to given identity.
    
    """
    count = 0
    foundFirst = 0
    while count < 10000:
        filename = "%s%04d%s" % (identity, count, '.xyz')
        
        if foundFirst:
            if os.path.exists(filename) or os.path.exists(filename+'.bz2') or os.path.exists(filename+'.gz'):
                currentFile = filename
                currentIndex = count
            
            else:
                break
        
        else:
            if os.path.exists(filename) or os.path.exists(filename+'.bz2') or os.path.exists(filename+'.gz'):
                foundFirst = 1
                currentFile = filename
                currentIndex = count
        
        count += 1
        
    if not foundFirst:
        print "ERROR: could not find xyz files with identity: %s" % identity
        return None, None
        
    finalXYZ = currentFile
    finalIndex = currentIndex
    
    return finalXYZ, finalIndex
