[docs]    def calcForce(self, MDPath=None, defectFlag=0, correctTE=0,
            includeAtoms=np.empty(0, np.int32), mainIndexes=np.empty(0, np.int32)):
       
        """
        Calculates forces using lattice method and modifies
       
        """
        
        if self.modFByMinMode:

            if len(self.mainIndexes):
                
                backupModFByMinMode = self.modFByMinMode
                self.modFByMinMode = 0
                
                _, eigenValue, resultVector, _, _, rVectorTemp, eigevValuesList = Lanczos.lanczosMethodForASystem(self, self.mainIndexes, 
                    None, self.params, normEigenVec=1, includeAtoms=self.includeAtoms, lnegmd=1, evn=3)
                
                self.modFByMinMode = backupModFByMinMode
                
                negCnt = 0
                                
                for i in range(len(eigevValuesList)):
                    if eigevValuesList[i] < 0:
                        negCnt += 1
                    else:
                        break
                
                if negCnt == 1:
                    status = 2
                    print "Only one negative eigenvalue", eigenValue
                    
                elif negCnt == 0:
                    status = 1
                    print "ERROR: All eigenvalues are positive", eigevValuesList[0:5]
                
                elif negCnt > 2:
                    status = 1
                    print "ERROR: More than two negative eigenvalues", eigevValuesList[0:5]
                
                else:
                    
                    super(MinModeLattice, self).calcForce(MDPath=MDPath, defectFlag=defectFlag, 
                        correctTE=correctTE, includeAtoms=self.includeAtoms, mainIndexes=self.mainIndexes)
                    
                    eigenVect = Vectors.normalise(resultVector)
                
                    FdI = np.dot(self.force, eigenVect)

                    self.force -= 2.0 * FdI * eigenVect
                        
                    deltaX = -1.0 * FdI / eigenValue
                    Xenergy = -0.5 * eigenValue * (deltaX ** 2.0)
                        
                    self.totalEnergy += Xenergy * 2.0
                    
                    self.rVector = rVectorTemp
                     
                    status = 0
            else:
                print ": ERROR: Force modification according to min mode is not available while using full system..."
                status = 1
        else:
            status = super(MinModeLattice, self).calcForce(MDPath=MDPath, defectFlag=defectFlag, 
                correctTE=correctTE, includeAtoms=self.includeAtoms, mainIndexes=self.mainIndexes)
            
        return status

[docs]    def evaluateCorrectTE(self, lattice):
        
        """
        Evaluates forces and TE of the whole system.
        
        """
        
        backupModFByMinMode = lattice.modFByMinMode
        
        lattice.modFByMinMode = 0
        
        lattice.calcForce(correctTE=1)
        TE = lattice.totalEnergy
        
        lattice.modFByMinMode = backupModFByMinMode
        
        return TE

[docs]    def runMinMode(self, inputStateIni):
        
        """
        Main procedure.
        
        """
        
        self.setupMinMode(inputStateIni)
        
        system = copy.deepcopy(inputStateIni) 
        
        # Getting initial energy and forces of whole system
        self.initialTE = self.evaluateCorrectTE(system)
        
        backupmodFByMinMode = system.modFByMinMode
        
        system.modFByMinMode = 1
                
        system.params = copy.deepcopy(self.params)
        
        status = self.minimiserMinMode.run(system)
        
        system.modFByMinMode = backupmodFByMinMode
        if not status:
            barrier = self.evaluateCorrectTE(system) - self.initialTE
        else:
            barrier = None
        
        return status, system, barrier



[docs]def convertLattice2MinModeLattice(lattice, params, mainIndexes, includeAtoms, rVector):
        
    """
    Copies all attributes from a lattice object to a minimum mode lattice object.
        
    """
        
    minModeLattice = MinModeLattice(lattice.NAtoms, storeEnergies=lattice.storeEnergies)
    minModeLattice.__dict__ = lattice.__dict__.copy()
    
    minModeLattice.params = copy.deepcopy(params)
    minModeLattice.modFByMinMode = 0
    
    minModeLattice.mainIndexes = copy.deepcopy(mainIndexes)
    minModeLattice.includeAtoms = copy.deepcopy(includeAtoms)
    
    minModeLattice.minModeEigVMethod = params.minModeEigVMethod
    
    minModeLattice.rVector = copy.deepcopy(rVector)
    
    return minModeLattice

[docs]def checkSaddleRank(params, saddle, conv2Rank1Md=0):
    
    """
    Checks the number of negative eigenvalues (saddle's rank). If it greater than 2, returns an error. If it is equal to 
    2, tries to converge to rank 1 saddle with min mode following method. 
    
    """
    
    noNegEigenValues = Prefactor.countNegativeValues(saddle.lanczosEigenVals)

    if noNegEigenValues > 2:
        log(__name__, "Saddle point's rank is higher than 2!", 1, 0)
        
        return 1, saddle
    
    elif noNegEigenValues == 2:
        
        if conv2Rank1Md:
            saddle.calcForce(correctTE=1)
#             barrierBeforeMinMode = saddle.totalEnergy - self.initialEnergy
#             
#             print "Barrier before MIN MODE: %f" % (barrierBeforeMinMode)
    
            if params.saddleConvergeRadius > 0.0:
                mainIndexes = saddle.buildSubLattice(saddle.defectSubSystem, params.saddleConvergeRadius)
            else:
                mainIndexes = copy.deepcopy(saddle.defectSubSystem)
                    
            includeAtoms = saddle.buildSubLattice(mainIndexes, Globals.subLatticeIncludeRadius)
            
            minModeStatus, newSaddle = dealWithRank2Saddle(params, None, saddle, mainIndexes, includeAtoms)
            
            if minModeStatus:
                noNegEigenValues = Prefactor.countNegativeValues(saddle.lanczosEigenVals)
                log(__name__, "After min mode the rank of the saddle is: %d" % (noNegEigenValues), 1, 0)
                 
                status = 1
            
            else:
                status = 0
                saddle.pos = copy.deepcopy(newSaddle.pos)
            
            return status, saddle
        else:
            log(__name__, "Saddle point's rank is 2! Skipping without trying to converge to rank 1.", 1, 0)
        
            return 1, saddle
            
    elif noNegEigenValues == 0:
        log(__name__, "Saddle point's rank is 0! It is not a saddle point!", 1, 0)
        
        return 1, saddle
    else:
        log(__name__, "Saddle point's rank's check is successful! Rank 1.", 1, 0)
        return 0, saddle
