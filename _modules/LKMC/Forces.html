[docs]def calcForce(lattice, MDPath=None, image=None, defectFlag=0, correctTE=False, 
              includeAtoms=np.empty(0, np.int32), mainIndexes=np.empty(0, np.int32)):
    # the path to the directory that contains the MD code files
    # if not passed as argument we assume if to be in the "md" dir in current dir
    if MDPath is None:
        MDPath = Globals.MDLocation
    
    # if exclusion and use sublattice build sublattice (unless it's already been built)
    forceConstructSub = False
    if defectFlag == 2 and Globals.useSubLatticeForceCalc and not len(includeAtoms) and not correctTE:
        includeAtoms = lattice.buildSubLattice(lattice.defectSubSystem, Globals.subLatticeIncludeRadius)
        mainIndexes = lattice.defectSubSystem
        forceConstructSub = True
    
    if defectFlag == 1 and len(includeAtoms):
        log(__name__, "WARNING: defectFlag==1 and includeAtoms is set; resetting to zero", 0, 0)
        includeAtoms = np.empty(0, np.int32)
    
    # call correct function for MDType
    if Globals.MDType == "LBOMDDIRECT" or Globals.MDType == "LBOMD":
        status = _calcForceLBOMDDirect(lattice, MDPath, image, includeAtoms, mainIndexes)  
        
    elif Globals.MDType == "LAMMPS":
        status = _calcForceLAMMPS(lattice, MDPath, image, includeAtoms, mainIndexes)
    
    elif Globals.MDType == "SIESTA":
        status = _calcForceSiesta(lattice, MDPath, image, includeAtoms, mainIndexes)
    
    else:
        sys.exit(__name__+": ERROR: unrecognised MDType: '%s'" % str(Globals.MDType))
    
    if not status:
        # handle inclusion/exclusion of defects here
        if defectFlag == 1:
            status = _excludeDefects(lattice)
        elif defectFlag == 2 and (not len(includeAtoms) or forceConstructSub):
            status = _includeDefects(lattice)
    
    return status

def _calcForceLBOMDDirect(lattice, MDPath, image, includeAtoms, mainIndexes):
    
    if not lbomdLoaded:
        sys.exit("ERROR: LBOMD not imported")
    
    # we work in the MD code dir, then change back to wherever we were originally
    # so store current working dir
    OWD = os.getcwd()
    
    # and change to MD dir
    try:
        os.chdir(MDPath)
    except:
        sys.exit(__name__+": ERROR: could not change to dir: "+MDPath)
    
    log(__name__, "calculating force (LBOMDDIRECT)", 4, 1)
        
    # which image
    if image is None:
        imageFor = -1
    else:
        imageFor = image
    
    # create specie string containing specie list to pass to fortran
    specieList = lattice.specieList
    specieString = "".join(specieList)
    
    try:
        lattice.totalEnergy, lattice.maxForce, lattice.maxForceAtomNo, status = LBOMD.python_interface.calc_force(imageFor, specieString, lattice.specie, 
                                                                                                                  lattice.pos, lattice.charge, lattice.force, 
                                                                                                                  lattice.KE, lattice.PE, lattice.cellDims[0], 
                                                                                                                  lattice.cellDims[4], lattice.cellDims[8], 
                                                                                                                  includeAtoms, mainIndexes)
    except:
        log(__name__, "LBOMDDirect failed", 0, 3)
        return 1
    
    # check status
    if status:
        log(__name__, "LBOMDDirect failed", 0, 3)
        return 1
    
    log(__name__, "total energy is %f eV" % (lattice.totalEnergy), 4, 2)
    log(__name__, "max force is %f eV/A (atom %d)" % (lattice.maxForce, lattice.maxForceAtomNo), 4, 2)
    
    # finally, change back to original working directory
    try:
        os.chdir(OWD)
    except:
        sys.exit(__name__+": ERROR: could not change to dir: "+OWD)
    
    return status

    """
    global lmp
    
    if not lammpsImported:
        sys.exit("ERROR: lammps not imported")
    
#    if image is not None or len(includeAtoms) or len(mainIndexes):

            NAtoms = obj.imageNAtoms

        # check if system has changed or this is first run

        if lmp is None or NAtoms != lmp.get_natoms() or getattr(obj, "resetLammps", 0):
            log(__name__, "Resetting LAMMPS", 4, 2)

            if Globals.PBC[1]:
                pbcy = "p"

                pbcz = "p"

            lmp.command("boundary %s %s %s" % (pbcx, pbcy, pbcz))
            lmp.command("atom_style charge")
            lmp.command("atom_modify sort 0 0") # not sure if this effects performance, makes PE's come out in right order
            lmp.command("atom_modify map array")
            
    #        # check for atom number and positions

    #            for i in xrange(len(obj.force)):
    #                obj.force[i] = 0.0

    #            return 1

#                     os.close(fd)

            # write atom data file
            try:
                # if no AMU info stored, calculate AMU
                if not len(obj.specieMassAMU):
                    obj.specieMassAMU = np.empty(len(obj.specieList), np.float64)

def _calcForceSiesta(obj, MDPath, image, includeAtoms, mainIndexes):
    """
    Calculate forces using SIESTA.
    
    Parameters
    ----------
    obj : {`Lattice.Lattice`, `NEB.NEB`}
        The object we are calculating the forces on.
    MDPAth : str
        Path to directory containing files needed for the MD calculation.
    image : int or None
        If object is a subclass of `Images.Images` then pass the image we are 
        working on; otherwise pass None.
    includeAtoms : 
        Not used!
    mainIndexes : 
        Not used!
    
    Returns
    -------
    status : int
        '0' means success, anything else is a failure.
    
    Notes
    -----
    We make a temporary directory and copy everything from `MDPath` into it. We 
    change to the temporary directory and create an input file for Siesta by:
    
    * writing header containing atom types, cell vectors, atoms positions
    * appending the "siesta.append" file that must be in the `MDPath` directory
      to the end
    
    We then run 'siesta' (it must be in your PATH) and send the output
    to a file, which is then parsed for the total energy. The forces are read in
    from the forces output file (lkmc.FA).
    
    """
    log(__name__, "Calculating force (SIESTA)", 4, 1)
    
    # we work in the MD code dir, then change back to wherever we were originally
    # so store current working dir
    OWD = os.getcwd()
    
    # and change to MD dir
    os.chdir(MDPath)
    
    try:
        # check siesta in file exists
        if not os.path.exists(Globals.siestaInFile):
            print "ERROR: SIESTA in file does not exist: %s" % Globals.siestaInFile
            return 88
        
        # current position
        if image is None:
            currentPos = obj.pos
            NAtoms = obj.NAtoms
        else:
            NAtoms = obj.imageNAtoms
            currentPos = obj.pos[3*image*NAtoms:3*(image+1)*NAtoms]
        
        # temporary directory
        tempdir = tempfile.mkdtemp()
        
        # write SIESTA input file
        try:
            # make sure specie if right type
            obj.specie = np.asarray(obj.specie, dtype=np.int32)
            
            # copy everything to tmp directory
            status = os.system("cp * %s/" % tempdir)
            if status:
                print "Error: could not copy Siesta files to temp directory"
                return 55
            
            # change to temp dir
            os.chdir(tempdir)
            
            # write the file
            #TODO: this should be written in C
            fn = "lkmc.fdf"
            with open(fn, "w") as f:
                # header
                f.write("# Auto-generated by LKMC\n\n")
                f.write("SystemLabel lkmc\n\n")
                f.write("NumberOfAtoms %d\n\n" % NAtoms)
                
                # species
                f.write("NumberOfSpecies %d\n" % len(obj.specieList))
                f.write("%block ChemicalSpeciesLabel\n")
                for i, sym in enumerate(obj.specieList):
                    # atomic number 
                    anum = Atoms.atomicNumber(sym)
                    
                    # we add underscores to end of single char symbols (H => H_, O => O_), 
                    # it looks like SIESTA does not.
                    if sym.endswith("_"):
                        sym = sym[:1]
                    
                    f.write("  %d %d %s\n" % (i + 1, anum, sym))
                f.write("%endblock ChemicalSpeciesLabel\n\n")
                
                # lattice vectors
                f.write("LatticeConstant 1.0 Ang\n") # does this ever change???
                f.write("%block LatticeVectors\n")
                f.write("  %f %f %f\n" % (obj.cellDims[0], obj.cellDims[1], obj.cellDims[2]))
                f.write("  %f %f %f\n" % (obj.cellDims[3], obj.cellDims[4], obj.cellDims[5]))
                f.write("  %f %f %f\n" % (obj.cellDims[6], obj.cellDims[7], obj.cellDims[8]))
                f.write("%endblock LatticeVectors\n\n")
                
                # atom coordinates
                f.write("AtomicCoordinatesFormat Fractional\n") # is there an option to specify absolute???
                f.write("%block AtomicCoordinatesAndAtomicSpecies\n")
                for i in xrange(NAtoms):
                    i3 = 3 * i
                    f.write("%.12f %.12f %.12f %d\n" % (currentPos[i3] / obj.cellDims[0], currentPos[i3+1] / obj.cellDims[4], 
                                                        currentPos[i3+2] / obj.cellDims[8], obj.specie[i] + 1))
                f.write("%endblock AtomicCoordinatesAndAtomicSpecies\n\n")
            
            # append custom section to end of file
            cmd = "cat %s >> %s" % (Globals.siestaInFile, fn)
            log(__name__, "Running: '%s'" % cmd, 4)
            status = os.system(cmd)
            if status:
                print "ERROR: cat cmd failed"
                return 77
            
            # we should now run the siesta code
            outfn = "siesta.out"
            with open(fn) as fin, open(outfn, "w") as fout:
                p = subprocess.Popen(["siesta"], shell=True, stdin=fin, stdout=fout, stderr=subprocess.STDOUT, close_fds=True)
                p.communicate()
                status = p.poll()
            
            if status:
                log(__name__, "Siesta force calc failed: %d (storing output in MDPath)" % status, 0)
                shutil.copy(outfn, os.path.join(OWD, MDPath, outfn))
                return 33
            
            # parse the output to get total energy
            foundTotalEnergy = False
            with open(outfn) as f:
                while not foundTotalEnergy:
                    # read line
                    line = f.readline()
                    if not line:
                        break
                    
                    # look for start of section containing total energy
                    if line.startswith("siesta: Final energy (eV)"):
                        while not foundTotalEnergy:
                            # read line
                            line = f.readline()
                            if not line:
                                break
                            
                            # check if this line contains total energy
                            if "Total =" in line:
                                array = line.split("=")
                                obj.totalEnergy = float(array[1])
                                foundTotalEnergy = True
            
            # check we found energy
            if not foundTotalEnergy:
                print "Error: could not locate total energy in output file"
                shutil.copy(outfn, os.path.join(OWD, MDPath, outfn))
                return 39
            
            # read in forces
            if not os.path.exists("lkmc.FA"):
                print "Error: forces file does not exist"
                return 53
            
            with open("lkmc.FA") as f:
                num = int(f.readline())
                assert num == NAtoms
                
                for count, line in enumerate(f):
                    array = line.split()
                    index = int(array[0]) - 1
                    
                    # correct index
                    if image is None:
                        i3 = 3 * index
                    else:
                        i3 = 3 * (image * num + index)
                    
                    # store forces
                    obj.force[i3  ] = np.float64(array[1])
                    obj.force[i3+1] = np.float64(array[2])
                    obj.force[i3+2] = np.float64(array[3])
                    
                if count + 1 != num:
                    print "Error: forces missing (%d != %d)" % (count, num)
                    shutil.copy("lkmc.FA", os.path.join(OWD, MDPath, "lkmc.FA"))
                    return 56
        
        finally:
            os.chdir(os.path.join(OWD, MDPath))
            shutil.rmtree(tempdir)
        
        # max force atom (may be in SIESTA output???)
        obj.maxForce, obj.maxForceAtomNo = Vectors.calcForceInfNorm(obj.force)
        
    finally:
        # finally, change back to original working directory
        os.chdir(OWD)
    
    return 0

def _includeDefects(lattice, mainIndexes=None):
    """
    Include forces on defect subsystem atoms, zero everything else
    
    """
    log(__name__, "Including defects", 4, 1)
    
    # remove forces on all atoms except those in sub-system
    oldForce = lattice.force
    
    if mainIndexes is None:
        indexes = lattice.defectSubSystem
    else:
        indexes = mainIndexes
    
    lattice.force = Vectors.includeSubVector3(indexes, oldForce)
    
    if lattice.force is None:
        print "ERROR: _includeDefects failed!"
        return 1
    else:
        return 0
    
################################################################################
# exclude defect sub system atoms
################################################################################
def _excludeDefects(lattice, mainIndexes=None):
    """
    Forces on defect subsystem atoms are zeroed
    
    """
    log(__name__, "Excluding defects", 4, 1)
    
    # remove forces acting on defects
    if mainIndexes is None:
        indexes = lattice.defectSubSystem
    else:
        indexes = mainIndexes
    
    status = Vectors.excludeSubVector3(indexes, lattice.force)
    
    if status:
        print "ERROR: _excludeDefects failed! (%d)" % status
    
    return status
