[docs]    def prepareForConvergence(self):
        """
        Prepares rVector and index lists.
        
        """
        # not sure this is required!
        self.lattice.force[:] = 0.0
        
        # prepare main indexes and initial guess at eigenvector
        if self.params.saddleConvergeRadius > 0.0:
            mainIndexes = self.lattice.buildSubLattice(self.lattice.defectSubSystem, self.params.saddleConvergeRadius)
            self.lattice.rVector = Vectors.extractVector3(mainIndexes, self.le_rVectorFull)
        
        else:
            mainIndexes = copy.deepcopy(self.lattice.defectSubSystem)
            self.lattice.rVector = self.le_rVector
        
        # prepare include atoms
        includeAtoms = self.lattice.buildSubLattice(mainIndexes, Globals.subLatticeIncludeRadius)
        
        # store main indexes and include atoms on lattice object
        self.lattice.mainIndexes = mainIndexes
        self.lattice.includeAtoms = includeAtoms

    

[docs]    def convergeToSaddle(self):
        """
        Converge to saddle point using Lanczos/LBFGS
        
        """
        convTime = time.time()
        
        # prepare for convergence
        self.prepareForConvergence()
        
        # converge to saddle
        status = self.saddleFinder.converge(self.lattice)
        
        self.totalFEvals += self.saddleFinder.forceCalls
        
        if status:
            log(__name__, "Failed (Lanczos-LBFGS saddle finder)", 1, 0)
            if self.saddleFinder.failReason is not None:
                self.failReason = self.saddleFinder.failReason
            else:
                self.failReason = "RAT: Lanczos saddle finder failed"
            self.lattice = None
            self.barrier = None
        
        else:
            pass
        
        self.lanczosConvergeTime = time.time() - convTime

    

[docs]    def checkIfConverged(self):
        """
        Checks if saddle point is reached or the search went off..
        
        """
        params = self.params
        
        # Checking the displacement
        maxMoveAtomIdx, maxMove, _, _ = Vectors.maxMovement(self.initialPos, self.lattice.pos, self.lattice.cellDims)
        
        self.FdotRP = Vectors.dotProduct(Vectors.normalise(self.force), 
                                         Vectors.normalise(Vectors.separationVector(self.initialPos, self.lattice.pos, 
                                                                                    self.lattice.cellDims)))
        
        # calculate barrier every few steps...
        fmag = Vectors.magnitude(self.force)
        if self.transCount % self.params.ratCheckBarrierInterval == 0:
            checkBarrierTime = time.time()
            self.lattice.calcForce(correctTE=True)
            self.barrier = self.lattice.totalEnergy - self.energyOrig
            currentBarrier = self.barrier
            self.checkBarrierTime += time.time() - checkBarrierTime
            log(__name__, "Step %d: MMAtm %d; MM %lf; Fmag %lf; FdotRP %lf; TFeval %d; Trans Step %f; Barrier %lf" % (self.transCount, maxMoveAtomIdx, maxMove,
                                                                                                                      fmag, self.FdotRP, self.totalStepCount, 
                                                                                                                      self.transStep, currentBarrier), 2, 2)
        else:
            currentBarrier = None
            log(__name__, "Step %d: MMAtm %d; MM %lf; Fmag %lf; FdotRP %lf; TFeval %d; Trans Step %f" % (self.transCount, maxMoveAtomIdx, maxMove, 
                                                                                                         fmag, self.FdotRP, self.totalStepCount, 
                                                                                                         self.transStep), 2, 2)
        
        # first check if run for too long, etc...
        if ((currentBarrier is not None and currentBarrier > self.params.ratMaxBarrier) or maxMove > self.params.ratMaxMoveDistance or 
                self.forceEvals > self.params.ratMaxFuncEval or self.transCount > self.params.ratMaxSteps):
            
            if currentBarrier is not None and currentBarrier > self.params.ratMaxBarrier:
                log(__name__, "Failure: barrier too high (%f > %f)" % (currentBarrier, self.params.ratMaxBarrier), 1, 3)
                self.failReason = "RAT: Barrier too high"
            
            elif maxMove > self.params.ratMaxMoveDistance:
                log(__name__, "Failure: max move too high (%f > %f)" % (maxMove, self.params.ratMaxMoveDistance), 1, 3)
                self.failReason = "RAT: Max move too high"
            
            elif self.forceEvals > self.params.ratMaxFuncEval:
                log(__name__, "Failure:  too many funcevals (%d > %d)" % (self.forceEvals, self.params.ratMaxFuncEval), 1, 3)
                self.failReason = "RAT: Max function evaluations exceeded"
            
            elif self.transCount > self.params.ratMaxSteps:
                log(__name__, "Failure:  too many steps (%d > %d)" % (self.transCount, self.params.ratMaxSteps), 1, 3)
                self.failReason = "RAT: Max steps exceeded"
            
            else:
                log(__name__, "Failed: unknown reason (this shouldn't happen)", 1, 3)
                self.failReason = "RAT: Unknown (this should not happen)"
            
            self.climb = False
            self.saddleFoundFail = True
            self.lattice = None
            self.barrier = None
            return
        
        # checking lowest eigenvalue...
        if params.ratUseLanczos and self.transCount > params.ratMinStepBeforeLanczos:
            lancCheckTime = time.time()
            
            lattice = copy.deepcopy(self.lattice)
            
            le_mainIndexes = lattice.defectSubSystem
            le_includeAtoms = lattice.buildSubLattice(le_mainIndexes, Globals.subLatticeIncludeRadius)
            
            # fix
            if self.fixSearches and self.le_rVector is None:
                log(__name__, "Using generated random vector for initial Lanczos vector", 2, 3)
                self.le_rVector = Vectors.extractVector3(le_mainIndexes, self.le_rVectorFull)
            
            status, lowestEV, resultVector, FEvalTmp, iterCnt, _, _ = Lanczos.lanczosMethodForASystem(lattice, le_mainIndexes, self.le_rVector, 
                                                                                                      self.params, normEigenVec=1, 
                                                                                                      includeAtoms=le_includeAtoms, returnMode=0)
            
            self.totalFEvals += FEvalTmp
            
            if status:
                log(__name__, "Lanczos failed (looking for lowest eigenvalue)", 0, 0)
                self.failReason = "RAT: Lanczos failed (lowest eigenvalue check)"
                self.climb = False
                self.lattice = None
                self.barrier = None
                return
            
            # store eigenvector to use as initial guess at next step
            self.le_rVector = Vectors.extractVector3(le_mainIndexes, resultVector)
            self.le_rVectorFull = resultVector
            
            log(__name__, "Calculated lowest eigenvalue: %f (%d iterations)" % (lowestEV, iterCnt), 2, 3)
            
            self.lanczosCheckLowestTime += time.time() - lancCheckTime
            
            # now we decide whether to start lanczos
            # is the eigenvalue treally negative
            if lowestEV < params.ratNegEVThreshold:
                self.numConsecNegEVs += 1
                
                # have we seen enough negative eigenvalues in a row
                if self.numConsecNegEVs >= params.ratNumNegBeforeLanczos:
                    self.climb = False
                    
                    # step back test
                    if self.params.ratStepBackTestBeforeLanczos:
                        status = self.stepBackTest()
                        if status:
                            self.lattice = None
                            self.barrier = None
                            return
                    
                    # minimise outer atoms
                    if self.params.lanczosMinOuterFirst:
                        status = self.minimiseFixedAtoms()
                        if status:
                            log(__name__, "Min outer atoms before Lanczos failed (%d)" % status, 0, 0)
                            self.failReason = "RAT: Minimise outer atoms before Lanczos failed"
                            self.lattice = None
                            self.barrier = None
                            return
                    
                    # converge to saddle
                    log(__name__, "Converging to saddle using Lanczos", 2, 1)
                    self.convergeToSaddle()
                    
                    if self.lattice is not None:
                        status = self.lattice.calcForce(correctTE=True)
                        self.forceEvals += 1
                        
                        if status:
                            self.failReason = "RAT: Final force calculation failed"
                            self.lattice = None
                            self.barrier = None
                            return
                        
                        else:
                            self.barrier = self.lattice.totalEnergy - self.energyOrig
                            log(__name__, "Success: Saddle point found! Barrier %f" % self.barrier, 2, 1)
                            return
            
            else:
                self.numConsecNegEVs = 0
        
        # Checking for stopping criteria
        elif (not self.params.ratUseLanczos) and self.FdotRP > self.params.ratFdotRPvalue and maxMove > self.params.ratMaxMoveCriteria:
            self.climb = False

            status = self.lattice.calcForce(correctTE=True)
            self.forceEvals += 1
            if status:
                self.failReason = "RAT: Final force calculation failed"
                self.lattice = None
                self.barrier = None
                return
            
            self.barrier = self.lattice.totalEnergy - self.energyOrig
                   
            finalSep = math.sqrt(math.pow(self.initialPos[maxMoveAtomIdx*3] - self.lattice.pos[maxMoveAtomIdx*3], 2) + 
                                 math.pow(self.initialPos[maxMoveAtomIdx*3+1] - self.lattice.pos[maxMoveAtomIdx*3+1], 2) + 
                                 math.pow(self.initialPos[maxMoveAtomIdx*3+2] - self.lattice.pos[maxMoveAtomIdx*3+2], 2))
                
            log(__name__, 
                "SUCCESS: Saddle point found! Barrier %f; Steps used %d; MaxMoveAtom %d; MaxMoveAtomDistance %f;" % (self.barrier, 
                                                                                                                     self.totalStepCount, 
                                                                                                                     maxMoveAtomIdx, finalSep), 
                1, 3)

    

[docs]    def minimiseFixedAtoms(self):
        """
        Minimise atoms that were held fixed during RAT
        
        """
        # minimise atoms that are normally held fixed during RAT
        log(__name__, "Minimising atoms that were held fixed before Lanczos", 2, 0)

        prms = copy.deepcopy(self.params)
        prms.minimizationType = prms.lanczosMinOuterType
        prms.minimizationCriteria = prms.lanczosMinOuterCriteria
        prms.minimizationTolerance = prms.lanczosMinOuterTol
        prms.minimizationExclusion = 1
        prms.LBFGSTolerance = prms.lanczosMinOuterLBFGSTol
        
        inipos = copy.deepcopy(self.lattice.pos)
        
        outmin = Minimise.getMinimiser(prms, jobID=self.jobID)
        status = outmin.run(self.lattice)
        if status:
            return status
        
        sep = Vectors.separation(inipos, self.lattice.pos, self.lattice.cellDims)
        log(__name__, "Separation after minimising outer atoms = %f" % sep, 2, 1)

[docs]    def relax(self):
        """
        Relaxation step
        
        """
        relaxTime = time.time()
        
        relax = True
        relaxCount = 0
        relaxStepIter = self.relaxStep
        self.FdotRP = 0
        FdotFPrev = 1.0
        FPrevdotFPrevPrev = 0.0
        
        # remove parallel component of true force
        self.removeParallelForceComp()
        
        # relaxation loop
        while relax and relaxCount < self.params.ratMaxRelaxSteps:
            relaxCount += 1
            self.totalStepCount += 1
                            
            # Normalizing force vector
            normForce = Vectors.normalise(self.lattice.force)
                
            # Relaxation step to the direction of the force
            Vectors.addVectorsInplace(self.lattice.pos, Vectors.scaleVector(normForce, relaxStepIter))

            # force calculation (sub-lattice)
            status = self.lattice.calcForce(defectFlag=2)
            self.forceEvals += 1
            if status:
                self.failReason = "RAT: Relaxation force calculation failed"
                return 4
            self.force = copy.deepcopy(self.lattice.force)

            # leave perpendicular component of true force
            self.removeParallelForceComp()
     
            FdotFPrev = Vectors.dotProduct(self.lattice.force, self.prevForce)
                
            # Recalculating relaxation step size
            absValue = math.fabs(FPrevdotFPrevPrev - FdotFPrev) / FdotFPrev
            
            if absValue < 0.5:
                relaxStepIter *= 1.2
            else:
                relaxStepIter /= 2.0
                
            if relaxStepIter >= self.relaxStepMax:
                relaxStepIter = self.relaxStepMax
            elif relaxStepIter <= self.relaxStepMin:
                relaxStepIter = self.relaxStepMin
                                      
            FPrevdotFPrevPrev = FdotFPrev
                
            log(__name__, "Step %d. RelaxStep %d; RelaxStepLen %lf; FdotFPrev %lf; TFeval %d;" % (self.transCount, relaxCount, relaxStepIter, FdotFPrev, self.totalStepCount), 3, 2)
            
            # Stop relaxation, as soon as force change its direction
            if FdotFPrev < 0:                   
                log(__name__, "Step %d. RelaxStep %d. Relaxation stopping criteria reached. Returning to translation. TFeval %d" % (self.transCount, relaxCount, self.totalStepCount), 3, 2)                
                relax = False

        # Reevaluating translation step
        if relaxCount < 3:
            self.transStep *= 2.0 
        else:
            self.transStep /= 2.0
            
        if self.transStep > self.transStepMax:
            self.transStep = self.transStepMax
        elif self.transStep < self.transStepMin:
            self.transStep = self.transStepMin
        
        self.relaxTime += time.time() - relaxTime
        
        if self.saddleFoundFail:
            return 1
        else:
            return 0

        self.lattice.displacementVector = Vectors.normalise(self.lattice.displacementVector)

       

[docs]    def removeParallelForceComp(self):
        """
        Removes component of the force that is parallel to the translation vector
        
        """
        FdotN = Vectors.partialDotProduct(self.lattice.force, self.lattice.displacementVector, self.lattice.defectSubSystem)
        
        vector = Vectors.scaleVector(self.lattice.displacementVector, -1.0 * FdotN)
        
        Vectors.applyPartialDisplacement(self.lattice.force, vector, self.lattice.defectSubSystem)

        """

        if self.params.ratMinOuterAtomsInterval > 0 and self.transCount % self.params.ratMinOuterAtomsInterval == 0:
            log(__name__, "Step %d. Minimise outer atoms" % (self.transCount), 3, 2)
            
            # minimise outer atoms
            status = self.minimiseFixedAtoms()
            if status:
                self.failReason = "RAT: Minimise outer atoms during RAT failed"
                log(__name__, "Minimise outer atoms during RAT failed", 1, 1)
                self.lattice = None
                self.barrier = None
                return status
            
            # need to force calc
            status = self.lattice.calcForce(defectFlag=2)
            if status:
                self.failReason = "RAT: calc force after min outer failed"
                log(__name__, "Calc force after min outer failed", 1, 1)
                self.lattice = None
                self.barrier = None
                return status

    

[docs]    def runMain(self, inputState, fixSearchObj):
        """
        The main RAT routine
        
        """
        totalTime = time.time()
        
        log(__name__, "Running RAT method...", 2, 1)
        
        # not sure if working with whole lattice force calc
        if not self.params.useSubLatticeForceCalc and self.params.ratUseLanczos:
            log(__name__, "Warning: RAT only tested with subLatticeForceCalc (Please delete this is you confirm it works with wholeLattice!)", 0, 0)
        
        # initial setup
        self.setup(inputState, fixSearchObj)
        
        assert self.lattice is not None
        assert self.initialState is inputState
        
        if self.ratDebugOutput:
            self.lattice.writeLattice(os.path.join("rat.output", "rat_step%d.dat" % 0))
        
        # initial displacement
        self.displace()
        
        if self.ratDebugOutput:
            self.lattice.writeLattice(os.path.join("rat.output", "rat_step%d.dat" % 1))
        
        # main loop
        while self.climb:            
            # counters 
            self.transCount += 1
            self.totalStepCount += 1      
            
            # Stores previous positions/forces of atoms
            self.prevPos = copy.deepcopy(self.lattice.pos)
            self.prevForce = copy.deepcopy(self.lattice.force)
            
            # translation
            status = self.translate()
            if status:
                break
            
            # relaxation
            status = self.relax()
            if status:
                break
            
            # minimise outer atoms
            status = self.checkMinimiseOuterAtoms()
            if status:
                break
            
            # new displacement vector
            self.remakeDisplVetor()
            
            # check if converged to saddle point
            self.checkIfConverged()
            
            if self.ratDebugOutput and self.lattice is not None:
                self.lattice.writeLattice(os.path.join("rat.output", "rat_step%d.dat" % (self.transCount + 1,)))
        
        # debugging (probably need separate param)
        if self.params.ratDebugMode:
            barrier = self.barrier
            if barrier is None:
                barrier = -9999
            debugInfo = Utilities.SaddleSearchDebugInfo("RAT", barrier, 0.0, self.transCount, self.forceEvals)
        else:
            debugInfo = None
        
        totalTime = time.time() - totalTime
        
        # number of force calls
        self.totalFEvals += self.forceEvals
        
        # create timing dict
        ratTimingDict = {
            "RAT total time": totalTime,
            "RAT translate time": self.translateTime,
            "RAT relax time": self.relaxTime,
            "Lanczos check time": self.lanczosCheckLowestTime,
            "Lanczos converge time (incl 0s)": self.lanczosConvergeTime,
            "RAT step count": self.transCount,
            "RAT total step count": self.totalStepCount,
            "RAT total force calls": self.totalFEvals,
            "RAT check barrier time": self.checkBarrierTime,
        }
        
        # exclude zeros (ie failed before lanczos converge to saddle)
        if self.lanczosConvergeTime > 0.0:
            ratTimingDict["Lanczos converge time (excl 0s)"] = self.lanczosConvergeTime
        
        # include time step back if relevant
        if self.params.ratStepBackTestBeforeLanczos:
            ratTimingDict["RAT step back before Lanczos"] = self.timeStepBack
        
        # release some refs
        if self.lattice is not None:
            self.lattice.displacementVector = None
            self.lattice.rVector = None
            self.lattice.mainIndexes = None
            self.lattice.includeAtoms = None
        self.initialState = None
        
        return self.lattice, self.barrier, debugInfo, self.failReason, ratTimingDict

[docs]    def setup(self, initialState, fixSearchObj):
        """
        Setup RAT for saddle search
        
        """
        # initialise
        self.climb = True
        self.transCount = 0
        self.totalStepCount = 0
        self.forceEvals = 0
        self.le_rVector = None
        self.le_rVectorFull = None
        self.numConsecNegEVs = 0
        self.fixSearches = False
        self.failReason = None
        self.ratDebugOutput = False
        
        # stats
        self.lanczosCheckLowestTime = 0.0
        self.lanczosConvergeTime = 0.0
        self.translateTime = 0.0
        self.relaxTime = 0.0
        self.totalFEvals = 0
        self.timeStepBack = 0.0
        self.checkBarrierTime = 0.0
        
        # make a copy of the lattice and initial positions
        self.initialState = initialState
        self.lattice = copy.deepcopy(initialState)
        self.initialPos = copy.deepcopy(initialState.pos)
        
        # set some params (no sure where *5 came from...)
        self.relaxStep = self.params.ratRelaxationStep
        self.relaxStepMin = self.relaxStep #/ 5.0
        self.relaxStepMax = self.relaxStep * 5.0
        
        # again, not sure where *6 came from
        self.transStep = self.relaxStep * 6.0
        self.transStepMin = self.relaxStepMin
        self.transStepMax = self.relaxStepMax * 6.0
        
        if type(fixSearchObj) is Utilities.TransitionFix:
            log(__name__, "Fixing search (index %d)" % fixSearchObj.index, 2, 1)
            self.lattice.displacementVector = fixSearchObj.displacementVector
            assert math.fabs(Vectors.magnitude(fixSearchObj.displacementVector) - 1.0) < 1e-7
            self.transStepIni = fixSearchObj.stepSize
            self.fixSearches = True
            self.le_rVectorFull = fixSearchObj.rVector
        
        else:
            # fix random number if debugging
            if self.params.ratDebugMode:
                randomNumber = 0.5
            else:
                randomNumber = random.random()
            
            # initial step size
            self.transStepIni = self.transStep * randomNumber * 2.0
        
        log(__name__, "Initial step size: %.10f" % self.transStepIni, 2, 3)
        
        # calc initial energy
        initialState.calcForce(correctTE=True)
        self.forceEvals += 1
        self.energyOrig = initialState.totalEnergy
        log(__name__, "Initial energy: %f" % self.energyOrig, 2, 3)
        
        # initialise
        self.energy = None
        self.prevPos = None
        self.prevForce = None
        self.saddleFoundFail = False
        self.barrier = 0.0
        self.force = None
        self.FdotRP = None
        
        # get Lanczos convergence minimisers
        self.saddleFinder = Lanczos.LanczosLBFGSSaddleFinder(self.params)
        self.saddleFinder.setJobID(self.jobID)
        
        if os.path.exists("rat.output"):
            self.ratDebugOutput = True



[docs]    def translate(self):
        """
        Translation
        
        """
        # check job status
        if ClientServerInterface.checkJobStatus(self.jobID):
            return 2
        
        transTime = time.time()
        
        # translate
        if self.transCount > 1:
            log(__name__, "Step %d. Translation" % (self.transCount), 3, 2)
            
            Vectors.applyPartialDisplacement(self.lattice.pos, Vectors.scaleVector(self.lattice.displacementVector, self.transStep), 
                                             self.lattice.defectSubSystem)
            
            # force calculation (sub-lattice)
            status = self.lattice.calcForce(defectFlag=2)
            self.forceEvals += 1
            if status:
                self.failReason = "RAT: Translation force calculation failed"
                return 3
            self.force = copy.deepcopy(self.lattice.force)
        
        self.translateTime += time.time() - transTime
        
        return 0

        """
        Step back and check if connected to initial
        
        """
        log(__name__, "Running step back test", 1, 1)
        
        timeStepBack = time.time()
        
        initialBackwards = Lattice.pushSystem(self.initialState, self.lattice, self.params, 0)
        initialBackwards.currentDefectVolume = copy.deepcopy(self.initialState.currentDefectVolume)
        initialBackwards.defectSubSystemFull = copy.deepcopy(self.initialState.defectSubSystemFull)
        initialBackwards.defectSubSystem = copy.deepcopy(self.initialState.defectSubSystem)
        initialBackwards.defectSubSystemSmall = copy.deepcopy(self.initialState.defectSubSystemSmall)
        
        stepBackParams = copy.deepcopy(self.params)
#         stepBackParams.minimizationExclusion = 2
        
        normalMinimiser = Minimise.getMinimiser(stepBackParams, self.jobID)
        minStatus = normalMinimiser.run(initialBackwards)
        
        status = 0
        if not minStatus:
            sep = Vectors.separation(self.initialPos, initialBackwards.pos, self.initialState.cellDims)
            
            if sep > self.params.ratStepBackTestCriteria:
                log(__name__, "Step back test failed: separation too big (%f > %f)" % (sep, self.params.ratStepBackTestCriteria), 1, 2)
                self.failReason = "RAT: Step back test failed: separation too big"
                status = 1
            
            else:
                log(__name__, "Step back test passed (%f < %f)" % (sep, self.params.ratStepBackTestCriteria), 1, 2)
        
        else:
            log(__name__, "Minimisation failed during step back test (%d); skipping test" % minStatus, 1, 2)
        
        self.timeStepBack = time.time() - timeStepBack
        
        return status
