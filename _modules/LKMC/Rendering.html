[docs]    def build(self, indexes, lattice):
        """
        Creates the object
        
        """
        self.cellDims = copy.deepcopy(lattice.cellDims)
        
        self.specieMass = copy.deepcopy(lattice.specieMass)
        self.specieCovalentRadius = copy.deepcopy(lattice.specieCovalentRadius)
        self.specieRGB = copy.deepcopy(lattice.specieRGB)
        
        self.NAtoms = len(indexes)
        
        self.pos = np.empty(3 * self.NAtoms, np.float64)
        self.specie = np.empty(self.NAtoms, np.int32)
        
        for i in xrange(self.NAtoms):
            index = indexes[i]
            
            self.specie[i] = lattice.specie[index]
            self.pos[3*i] = lattice.pos[3*index]
            self.pos[3*i+1] = lattice.pos[3*index+1]
            self.pos[3*i+2] = lattice.pos[3*index+2]

        

[docs]    def output(self, filename, storePOV=0, renderPOV=True, outdir='.'):
        """
        Output the object.
        filename: filename without prefix
        storePOV: store the povray file
        renderPOV: render the povray file
        
        """
        if Globals.povrayPath is None:
            renderPOV = False
        
        if storePOV:
            # write povray file in current directory
            povfile = filename + ".pov"
        
        else:
            # write pov to temporary file
            povFileDesc, povfile = tempfile.mkstemp(dir="/tmp")
        
        try:
            fpov = open(povfile, "w")
            
            # write header
            self.writePOVHeader(fpov)
            
            # write cell
            self.writeCellOutline(fpov)
            
            # write atoms
            for i in xrange(self.NAtoms):
                spec = self.specie[i]
                string = "sphere { <%f,%f,%f>, %f pigment { color rgb <%f,%f,%f> } finish { ambient %f phong %f } }\n" % (- self.pos[3*i], 
                                                    self.pos[3*i+1], self.pos[3*i+2], self.specieCovalentRadius[spec], self.specieRGB[spec][0],
                                                    self.specieRGB[spec][1], self.specieRGB[spec][2], 0.25, 0.9)
                fpov.write(string)
            
            fpov.close()
            if not storePOV:
                os.close(povFileDesc)
            
            cwd = os.getcwd()

            if os.path.exists(outdir):
                dirChanged = 1
                os.chdir(outdir)
            else:
                dirChanged = 0
            
            # render
            if renderPOV:
                print "RUNNING POVRAY %s" % filename
                command = "'%s' -I%s -D +A +W%d +H%d +O%s" % (Globals.povrayPath, povfile, 800, 600, filename+".png")
#                print "COMMAND: %s" % command
#                os.system(command)
                output, stderr, status = Utilities.runSubProcess(command)
                if status:
                    print "POVRAY FAILED"
                    print stderr
            
            if dirChanged:
                os.chdir(cwd)
        
        finally:
            if not storePOV:
                os.unlink(povfile)

    

[docs]    def writePOVHeader(self, fpov):
        """
        Write header to povray file
        
        """
        povDims = [-self.cellDims[0], self.cellDims[4], self.cellDims[8]]
        
        if self.cellDims[4] > self.cellDims[0]:
            povCamLocation = [0.9 * self.cellDims[4]]
        else:
            povCamLocation = [self.cellDims[0] - 0.1 * self.cellDims[4]]
        povCamLocation.append(1.3 * povDims[1])
        povCamLocation.append(povDims[2] / 2)
        
        povLightSource = [0]
        povLightSource.append(1.2 * povDims[1])
        povLightSource.append(povDims[2] / 2)
        
        povCamLookAt = [povDims[0] / 2]
        povCamLookAt.append(povDims[1] / 2)
        povCamLookAt.append(povDims[2] / 2)
        
        fpov.write("camera { location <"+str(povCamLocation[0])+","+str(povCamLocation[1])+","+str(povCamLocation[2])+">\n")
        fpov.write("         look_at <"+str(povCamLookAt[0])+","+str(povCamLookAt[1])+","+str(povCamLookAt[2])+"> }\n")
        fpov.write("light_source { <"+str(povLightSource[0])+","+str(povLightSource[1])+","+str(povLightSource[2])+"> color rgb <1,1,1> }\n")
        fpov.write("background { color rgb <1,1,1> }\n\n")

[docs]    def build(self, vacancies, interstitials, antisites, onAntisites, input, ref, NClusters=0, cluster=[], overlayMode=False):
        """
        Create the defects object
        
        """
        # Building a list for overlaying defect numbers according to their specie
        if overlayMode:
            self.overlaySpecieList = input.specieList
            self.overlaySpecieCount = len(input.specieList)
            
            self.overlayVacTot = 0
            self.overlayIntTot = 0
            self.overlayAntTot = 0
            
            self.overlayVacBySpecie = np.zeros(self.overlaySpecieCount, np.int32)
            self.overlayIntBySpecie = np.zeros(self.overlaySpecieCount, np.int32)
            self.overlayAntBySpecie = np.zeros(self.overlaySpecieCount, np.int32)
        
        self.cellDims = copy.deepcopy(ref.cellDims)
        
        self.inputMass = input.specieMass
        self.inputCovRad = input.specieCovalentRadius
        self.inputRGB = input.specieRGB
        self.refMass = ref.specieMass
        self.refCovRad = ref.specieCovalentRadius
        self.refRGB = ref.specieRGB
        
        self.vacancies = vacancies
        self.interstitials = interstitials
        self.antisites = antisites
        self.onAntisites = onAntisites
        self.NVac = len(vacancies)
        self.NInt = len(interstitials)
        self.NAnt = len(antisites)
        self.NDefects = self.NVac + self.NInt + self.NAnt
        
        # build list of vacancies
        self.vacSpecie = np.empty(self.NVac, np.int32)
        self.vacPos = np.empty(3 * self.NVac, np.float64)
        for i in xrange(len(vacancies)):
            index = vacancies[i]
            
            self.vacSpecie[i] = ref.specie[index]
            self.vacPos[3*i] = ref.pos[3*index]
            self.vacPos[3*i+1] = ref.pos[3*index+1]
            self.vacPos[3*i+2] = ref.pos[3*index+2]
            
            if overlayMode:
                self.overlayVacTot += 1
                self.overlayVacBySpecie[self.vacSpecie[i]] += 1
        
        # build list of interstitials
        self.intSpecie = np.empty(self.NInt, np.int32)
        self.intPos = np.empty(3 * self.NInt, np.float64)
        for i in xrange(len(interstitials)):
            index = interstitials[i]
            
            self.intSpecie[i] = input.specie[index]
            self.intPos[3*i] = input.pos[3*index]
            self.intPos[3*i+1] = input.pos[3*index+1]
            self.intPos[3*i+2] = input.pos[3*index+2]
            
            if overlayMode:
                self.overlayIntTot += 1
                self.overlayIntBySpecie[self.intSpecie[i]] += 1
        
        # build list of antisites
        self.antSpecie = np.empty(self.NAnt, np.int32)
        self.antPos = np.empty(3 * self.NAnt, np.float64)
        self.onAntPos = np.empty(3*self.NAnt, np.float64)
        self.onAntSpecie = np.empty(3 * self.NAnt, np.int32)
        for i in xrange(len(antisites)):
            index = antisites[i]
            index2 = onAntisites[i]
            
            self.antSpecie[i] = ref.specie[index]
            self.onAntSpecie[i] = input.specie[index2]
            self.antPos[3*i] = ref.pos[3*index]
            self.antPos[3*i+1] = ref.pos[3*index+1]
            self.antPos[3*i+2] = ref.pos[3*index+2]
            self.onAntPos[3*i] = input.pos[3*index2]
            self.onAntPos[3*i+1] = input.pos[3*index2+1]
            self.onAntPos[3*i+2] = input.pos[3*index2+2]
            
            if overlayMode:
                self.overlayAntTot += 1
                self.overlayAntBySpecie[self.antSpecie[i]] += 1
        
        # store cluster array
        self.NClusters = NClusters
        self.cluster = cluster

[docs]    def renderClusters(self, filename, storePOV, renderPOV):
        """
        Render clusters of defects to separate files
        
        """
        if Globals.povrayPath is None:
            renderPOV = False
        
        for i in xrange(self.NClusters):
            if storePOV:
                # write povray file in current directory
                povfile = "%s_cluster%d.pov" % (filename, i)
            
            else:
                # write pov to temporary file
                povFileDesc, povfile = tempfile.mkstemp(dir="/tmp")
            
            fpov = open(povfile, "w")
            
            # write header
            self.writePOVHeader(fpov)
            
            # write cell
            self.writeCellOutline(fpov)
            
            # render vacancies in this cluster
            for j in xrange(self.NVac):
                # is this in interstitial part of this cluster
                if self.cluster[j] == i:
                    spec = self.vacSpecie[j]
                    covRad = self.refCovRad[spec]
                    string = "box { <%f,%f,%f>,<%f,%f,%f> pigment { color rgbt <%f,%f,%f,%f> } finish { diffuse %f ambient %f phong %f } }\n" % (
                                                        - self.vacPos[3*j] - covRad, self.vacPos[3*j+1] - covRad, self.vacPos[3*j+2] - covRad, 
                                                        - self.vacPos[3*j] + covRad, self.vacPos[3*j+1] + covRad, self.vacPos[3*j+2] + covRad,
                                                        self.refRGB[spec][0], self.refRGB[spec][1], self.refRGB[spec][2], 0.2, 0.4, 0.25, 0.9)
                    fpov.write(string)
            
            # render interstitials in this cluster
            for j in xrange(self.NInt):
                # is this in interstitial part of this cluster
                if self.cluster[self.NVac+j] == i:
                    spec = self.intSpecie[j]
                    string = "sphere { <%f,%f,%f>, %f pigment { color rgb <%f,%f,%f> } finish { ambient %f phong %f } }\n" % (- self.intPos[3*j], 
                                                        self.intPos[3*j+1], self.intPos[3*j+2], self.inputCovRad[spec], self.inputRGB[spec][0],
                                                        self.inputRGB[spec][1], self.inputRGB[spec][2], 0.25, 0.9)
                    fpov.write(string)
            
            # render antisites in this cluster
            for j in xrange(self.NAnt):
                # is this in interstitial part of this cluster
                if self.cluster[self.NVac+self.NInt+j] == i:
                    spec1 = self.antSpecie[j]
                    spec2 = self.onAntSpecie[j]
                    covRad = self.inputCovRad[spec2]
                    
                    # render box outline
                    cornera = [0]*3
                    cornerb = [0]*3
                    cornera[0] =  - self.antPos[3*j] - covRad
                    cornera[1] =  self.antPos[3*j+1] - covRad
                    cornera[2] =  self.antPos[3*j+2] - covRad
                    cornerb[0] =  - self.antPos[3*j] + covRad
                    cornerb[1] =  self.antPos[3*j+1] + covRad
                    cornerb[2] =  self.antPos[3*j+2] + covRad
                    _renderCellOutline(fpov, cornera, cornerb, self.refRGB[spec1], 0.1)
                    
                    # render atom
                    string = "sphere { <%f,%f,%f>, %f pigment { color rgb <%f,%f,%f> } finish { ambient %f phong %f } }\n" % (- self.antPos[3*j], 
                                                        self.antPos[3*j+1], self.antPos[3*j+2], covRad, self.inputRGB[spec2][0],
                                                        self.inputRGB[spec2][1], self.inputRGB[spec2][2], 0.25, 0.9)
                    fpov.write(string)
            
            fpov.close()
            if not storePOV:
                os.close(povFileDesc)
            
            # render
            if renderPOV:
                command = "'%s' -I%s -D +A +W%d +H%d +O%s" % (Globals.povrayPath, povfile, 800, 600, "%s_cluster%d.png" % (filename, i))
    #            print "COMMAND", command
                output, stderr, status = Utilities.runSubProcess(command)
                if status:
                    print "POVRAY FAILED"
            
            if not storePOV:
                os.unlink(povfile)

def _renderCellOutline( filehandle, a, b, rgb, rad ):
    
    filehandle.write("#declare R = "+str(rad)+";\n")
    filehandle.write("#declare myObject = union {\n")
    filehandle.write("    sphere { <"+str(a[0])+","+str(a[1])+","+str(a[2])+">, R }\n")
    filehandle.write("    sphere { <"+str(b[0])+","+str(a[1])+","+str(a[2])+">, R }\n")
    filehandle.write("    sphere { <"+str(a[0])+","+str(a[1])+","+str(b[2])+">, R }\n")
    filehandle.write("    sphere { <"+str(b[0])+","+str(a[1])+","+str(b[2])+">, R }\n")
    filehandle.write("    sphere { <"+str(a[0])+","+str(b[1])+","+str(a[2])+">, R }\n")
    filehandle.write("    sphere { <"+str(b[0])+","+str(b[1])+","+str(a[2])+">, R }\n")
    filehandle.write("    sphere { <"+str(a[0])+","+str(b[1])+","+str(b[2])+">, R }\n")
    filehandle.write("    sphere { <"+str(b[0])+","+str(b[1])+","+str(b[2])+">, R }\n")
    filehandle.write("    cylinder { <"+str(a[0])+","+str(a[1])+","+str(a[2])+">, <"+str(b[0])+","+str(a[1])+","+str(a[2])+">, R }\n")
    filehandle.write("    cylinder { <"+str(a[0])+","+str(a[1])+","+str(b[2])+">, <"+str(b[0])+","+str(a[1])+","+str(b[2])+">, R }\n")
