[docs]    def run(self, latticeIn, step=None, clock=None):
        """
        Calls all of the methods to do one complete cascade simulation.
        
        Parameters
        ----------
        latticeIn : `Lattice.Lattice`
            System lattice to run a cascade in.
        step : int, optional
            Current KMC step, to allow changing into the appropriate working directory.
        clock : float, optional
            System time of KMC simulation, used to record time of last cascade event, facilitating simulation resume.
            
        Returns:
        ----------
        lattice : `Lattice.Lattice`
            Post-cascade lattice (minimised if user requires)
        status : int
            Status report, returned as zero if no errors have been encountered.
        
        Notes
        -----
        The workflow is as follows:
            * Pre-event thermalisation
            * Generating PKA attributes
            * Adding the PKA
            * Simulating the cascade
            * Post-cascade minimisation
        
        """
        log(__name__, "Running Cascade", 0, 0)
        
        # local ref to params
        params = self.params
        
        # copy the input lattice
        lattice = copy.deepcopy(latticeIn)
        
        # switch to working directory
        root = self.switchToOutputDirectory(step)
        try:
            # initial status
            status = 0
            
            # thermalise lattice and get velocities
            velocity = self.thermalise(lattice, params.cascThermalise, params.cascThermoTimeWhole,
                                       params.cascThermoTimeLayer, params.cascThermoType)
                    
            # Create PKA attributes for new cascade
            pkaVector = self.getPKAVector()
            pkaEnergy = self.getPKAEnergy()
            pkaIndex = self.getPKAIndex(lattice)
            
            # Record PKA attributes in current working directory
            self.writePKAMetadata(lattice, pkaEnergy, pkaVector, pkaIndex, clock, step)
            
            # Add PKA to lattice system (modifies the velocity array to add the PKA)
            self.setPKAVelocity(lattice, velocity, pkaIndex, pkaEnergy, pkaVector)
            
            # Run molecular dynamics simulation
            self.runMD(lattice, velocity, params.cascThermoType, params.cascDuration)
            
            # Minimise if required
            self.minimise(lattice, params.cascMinAfterCasc, params.cascMinUseLAMMPS, params.cascMinFailUseLAMMPS)
        
        finally:
            # Change back to original working directory
            os.chdir(root)
        
        # record start time of cascade
        cascTimePath = os.path.join(root, params.outputDirectory, "lastCascTime.OUT")
        with open(cascTimePath, "w") as timeOut:
            if clock is not None:
                timeOut.write("%.15f\n" % clock)
            else:
                timeOut.write("0.0\n")
            if step is not None:
                timeOut.write("%d\n" % step)
        
        # Return post-cascade lattice & status
        return status, lattice

    

[docs]    def setPKAVelocity(self, lattice, velocity, pkaIndex, pkaEnergy, pkaVector):
        """Set the velocity of the PKA."""
        # velocity of PKA atom
        pkaVelocity = self.getVelocityFromKE(pkaEnergy, direction=pkaVector, sym=lattice.atomSym(pkaIndex))
        
        # set velocity
        velocity[3 * pkaIndex] = pkaVelocity[0]
        velocity[3 * pkaIndex + 1] = pkaVelocity[1]
        velocity[3 * pkaIndex + 2] = pkaVelocity[2]

    

[docs]    def writePKAMetadata(self, lattice, pkaEnergy, pkaVector, pkaIndex, clock, step):
        """Write file containing metadata for this event."""
        with open("pkaAttributes.OUT", "w") as pkaOut:
            if step is not None:
                pkaOut.write("Step %d\n" % step)
            if clock is not None:
                pkaOut.write("Time: %.15f s\n" % clock)
            pkaOut.write("=" * 80 + "\n")
            pkaOut.write("PKA Attributes:\n")
            pkaOut.write("=" * 80 + "\n")
            pkaOut.write("Energy: %.3f\n" % pkaEnergy)
            pkaOut.write("Orientation: %f %f %f\n" % (pkaVector[0], pkaVector[1], pkaVector[2]))
            pkaOut.write("Index: %d\n" % pkaIndex)
            pkaOut.write("Symbol: %s\n" % lattice.atomSym(pkaIndex))
            pkaOut.write("Location: %f %f %f\n" % (lattice.pos[3 * pkaIndex], lattice.pos[3 * pkaIndex + 1],
                                                   lattice.pos[3 * pkaIndex + 2]))

    

[docs]    def getPKAVector(self):
        """
        Creates a PKA vector, either randomly or using a pre-set value.
        
        Returns
        -------
        pkaVector: `numpy.ndarray`
            Cartesian three vector indicating a point on the unit sphere; used to describe the PKA direction.
        
        Notes
        -----
        If a random PKA vector is required it is chosen from a probability distribution evenly distributed over the
        unit sphere [1]_.
        
        References
        ----------
        .. [1] E.W. Weisstein. Sphere Point Picking. From MathWorld-A Wolfram Web Resource.
           http://mathworld.wolfram.com/SpherePointPicking.html
        
        """
        # create vector for storing result
        pkaVector = np.zeros(3)
        
        # Random PKA vector
        if self.params.cascUseRandPKAVector:
            u = random.uniform(0, 1)
            v = random.uniform(0, 1)
            t = 2.0 * math.pi * u
            p = math.acos(2.0 * v - 1.0)
            pkaVector[0] = math.sin(p) * math.cos(t)
            pkaVector[1] = math.sin(p) * math.sin(t)
            pkaVector[2] = math.cos(p)
            log(__name__, "PKA vector generated, [%f, %f, %f]" % (pkaVector[0], pkaVector[1], pkaVector[2]), 0, 2)
        
        # Retrieve PKA vector pre-set in lkmcInput.IN
        else:
            vector = self.params.cascPKAVector
            pkaVector = np.asarray(map(float, vector.split(",")))
            if len(pkaVector) != 3:
                raise ValueError("User specified PKA Vector not of length 3 -- a Cartesian three-vector is required!")
            log(__name__, "Using pre-set PKA vector, [%f, %f, %f]" % (pkaVector[0], pkaVector[1], pkaVector[2]), 0, 2)
            
        return pkaVector

    

[docs]    def getPKAEnergy(self):
        """
        Set the PKA energy to be used in the collision cascade.
        
        Returns
        -------
        energy: float
            A randomly generated PKA energy (units: eV)
        
        Raises
        ------
        IOError
            If there is an error reading the probability file.
        
        Notes
        -----
        If one wishes to use a probabilisticly generated (i.e. non-fixed) PKA energy, this function will provide a
        value. These are generated from a user supplied input file describing the probability distribution, the path of
        which is specified by `params.cascPKAProbFile`.
        
        """
        pkaEnergy = 0.0
        if self.params.cascUseProbEnergies:
            probabilityFilePath = os.path.join(Globals.MDLocation, self.params.cascPKAProbFile)
            with open(probabilityFilePath, "r") as probabilityFile:
                # read the number of lines
                line = probabilityFile.readline().strip()
                if len(line.split(",")) != 1:
                    raise IOError("The first line of the probability file should contain the number of rows only")
                num = int(line)
                
                # allocate array
                probabilities = np.zeros((num, 2), dtype=np.float64)
                
                # read lines
                for count, line in enumerate(probabilityFile):
                    if count == num:
                        break
                    array = line.split(",")
                    probabilities[count][0] = float(array[0])
                    probabilities[count][1] = float(array[1])
            
            # check we read all the lines
            if count + 1 < num:
                raise IOError("Missing lines in cascade probability distribution file (%d < %d)" % (count + 1, num))
            
            # extent of data
            minVal = probabilities[0][1]
            maxVal = probabilities[-1][1]
        
            # to prevent very small chance of returning a PKA energy of 0 eV.
            # TODO: Look at this more closely some point.
            while True:
                position = random.uniform(minVal, maxVal)
                for x in xrange(count - 1):
                    if (position >= probabilities[x][1] and position <= probabilities[x + 1][1]):
                        pkaEnergy = probabilities[x][0]
                        break
                if pkaEnergy > 0.0:
                    break
            
            log(__name__, "PKA energy generated, %.3f eV" % (pkaEnergy), 0, 2)
        
        else:
            pkaEnergy = self.params.cascPKAEnergy
            log(__name__, "Using preset PKA energy, %.3f eV" % (pkaEnergy), 0, 2)
            
        return pkaEnergy

    

[docs]    def getPKAIndex(self, lattice):
        """
        Selects an atom index for the PKA from the user's specification in params.cascPKASpecies
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            System lattice to choose a PKA from.
        
        Returns
        ----------
        pkaIndex : int
            The lattice index of the selected atom.
        pkaSymbol : str
            The atomic symbol of the selected atom.
            
        """
        # parameters
        pkaSpecies = self.params.cascPKASpecies
        
        # process
        pkaIndex = None
        possiblePKAs = None
        if pkaSpecies == "ALL":
            pkaIndex = random.randint(0, lattice.NAtoms - 1)
        else:
            possiblePKAs = lattice.getSpecieIndices(pkaSpecies)
            if len(possiblePKAs):
                pkaIndex = possiblePKAs[random.randint(0, len(possiblePKAs)) - 1]
            else:
                raise RuntimeError("Failed to find any atoms of your specified PKA species.")
        
        atomName = Atoms.atomName(lattice.atomSym(pkaIndex))
        log(__name__,
            "The selected PKA is an %s atom with index %d, located at [%f %f %f]" % (atomName, pkaIndex,
                                                                                     lattice.pos[3 * pkaIndex],
                                                                                     lattice.pos[3 * pkaIndex + 1],
                                                                                     lattice.pos[3 * pkaIndex + 2]),
            0, 2)
        
        return pkaIndex
