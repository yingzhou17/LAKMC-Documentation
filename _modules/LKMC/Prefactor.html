[docs]    def calculate(self, currentVolIndex):
        """
        Uses implicit eigenvalues calculation method to find the initial eigenvalues.
        
        """
        
        # class properties for statistics
        self.startTime = time.time()
        
        self.methodRunList = []
        self.methodTimeList = []
        self.methodSuccList = []
        
        # building sub-systems
        mainIndexes = self.initialState.buildSubLattice(self.initialState.defectVolumes[currentVolIndex].searchInitialVol, 
            self.params.prefactorRadius)
        includeAtoms = self.initialState.buildSubLattice(mainIndexes, Globals.subLatticeIncludeRadius)
         
        jobStatus = ClientServerInterface.checkJobStatus(self.jobID)
        
        if jobStatus:
            return self.createFailResult(currentVolIndex=currentVolIndex)
                
        # calculating eigenvalues
        statusInt, eigenValues, _, _ = Algebra.getSystemsImplicitEigenvalues(self.initialState, 
            mainIndexes, includeAtoms, self.params, evn=-1, noEigenVects=1, includeMasses=1, 
            noOfWorkers=1, jobID=self.jobID)
        
        if statusInt:
            return self.createFailResult(currentVolIndex=currentVolIndex)
        
        if countNegativeValues(eigenValues) != 0:
            return self.createFailResult(currentVolIndex=currentVolIndex)
        
        initialSumLogEVs = eigenvalues2Frequencies(eigenValues)
        
        return self.createSuccessResult(currentVolIndex=currentVolIndex, initialSumLogEVs=initialSumLogEVs)

            return 1
        
        mainIndexes = saddleLattice.buildSubLattice(saddleLattice.defectSubSystem, params.saddleConvergeRadius)
    else:
        mainIndexes = saddleLattice.rVectorIndexes
                
    includeAtoms = saddleLattice.buildSubLattice(mainIndexes, Globals.subLatticeIncludeRadius)

    if saddleLattice.lanczosEigenVals is not None:

[docs]def countNegativeValues(array):
    
    """
    Counts the number of negative values in the array. Assumes that values are in ascending order
    
    """
    
    cnt = 0
    for i in range(len(array)):
        if array[i] < 0.0:
            cnt += 1
        else:
            break
    
    return cnt    



[docs]def eigenvalues2Frequencies(eigenValues, ignoreFirst=None):
    """
    Converts array of eigenvalues to a sum of log((ev[i])/twopi)
    
    """
    
    logSum = 0.0
    
    for i in range(len(eigenValues)):
        if ignoreFirst and i == 0:
            continue
        else:
            logSum += math.log(math.sqrt(eigenValues[i])/Constants.twoPi)
    
    return logSum

    

[docs]def estimatePrefactor(initialSumLogEVs, saddleSumLogEVs):
    
    """
    Estimates prefactor by using pre-calculated eigenvalues from the initial and saddle states.
    
    Eigenvalues are expressed as such:  initialSumLogEVs = log(ev1) + log(ev2) + .. log(evN)
                                        saddleSumLogEVs = log(ev2) + log(ev3) + .. log(evN)
    """
    
    if ((initialSumLogEVs is None) or (saddleSumLogEVs is None)):        
        log(__name__, "Warning: something is wrong with the eigenvalues sums!", 2, 2)
        return -1 
    
    prefactor = math.exp(initialSumLogEVs - saddleSumLogEVs)

    prefactor *= 1e15
    
    return prefactor



[docs]def estimatePrefactorOld(initialEigenVals, saddleEigenVals):
    
    """
    Estimates prefactor by using pre-calculated eigenvalues from the initial and saddle states.
    """
    
    negCnt = countNegativeValues(saddleEigenVals)
    if negCnt != 1:
        log(__name__, "Warning: saddle point's rank = %d" % (negCnt), 2, 2)
        return -1 
    
    noEV = len(initialEigenVals)
    if noEV != len(saddleEigenVals):
        log(__name__, "Warning: the lengths of eigenvalue arrays do not match!", 2, 2)
        return -1 
        
    prefactor = 1.0
    
    for i in range(noEV-1):
        idx = noEV-i-1
        
        prefactor *= math.sqrt(initialEigenVals[idx])
        prefactor /= math.sqrt(saddleEigenVals[idx])

        #print prefactor, initialEigenVals[idx],  saddleEigenVals[idx]
        
    prefactor *= math.sqrt(initialEigenVals[0])
    prefactor /= math.pi * 2
    prefactor *= 1e15
    
    return prefactor
