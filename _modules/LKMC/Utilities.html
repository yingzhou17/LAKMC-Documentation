[docs]    def write(self, fileName, append=False):
        """
        Write the text to file
        
        Parameters
        ----------
        fileName : str
            The name of the file to write the text to
        append : bool, optional
            Whether the text should be appended to the file. Default is False.
        
        """
        d = os.path.dirname(fileName)
        if d:
            if not os.path.exists(d):
                os.makedirs(d)

        if append:
            mode = "a"
        else:
            mode = "w"
        
        with open(fileName, mode) as f:
            if self.textAsLines:
                f.write("\n".join(self.text))
                
                # make sure file ends in newline
                f.write("\n")
            
            else:
                f.write(self.text)

[docs]def runDreadnaut(graphFile=None, graphString=None):
    """
    Run dreadnaut on the given graph
    
    """
    if graphFile is None and graphString is None:
        return None
    
    dreadnaut = getPathToDreadnaut()
        
    if graphFile is not None:
        command = "%s < %s" % (dreadnaut, graphFile)
    else:
        command = "echo '%s' | %s" % (graphString, dreadnaut)
    
    output, stderr, status = runSubProcess(command)
    if status:
        print "WARNING: dreadnaut FAILED"
        print stderr
        return None
    
    array = output.strip().split("\n")
    line = array[-1].strip()
        
    if line[:1] == "[" and line[-1:] == "]":
        line = string.replace(line, " ", "_")
        line = string.replace(line, "[", "")
        line = string.replace(line, "]", "")
    
    else:
        print "WARNING: dreadnaut FAILED"
        line = None
    
    return line

[docs]def fileLineCount(filename):
    """
    Return line count of given file
    
    """
    if not os.path.exists(filename) and not os.path.exists(filename+".gz") and not os.path.exists(filename+".bz2"):
        lineCount = -1
    
    else:
        zipFlag, filename = checkForZipped(filename)
        
        command = "wc -l %s" % (filename,)
        stdout, stderr, status = runSubProcess(command)
        
        try:
            array = stdout.split()
            lineCount = int(array[0])
        
        except:
            lineCount = -1
        
        if zipFlag:
            os.remove(filename)
    
    return lineCount

[docs]def runSubProcess(command, verbose=0):
    """
    Run command using subprocess module.
    Return tuple containing STDOUT, STDERR, STATUS
    Caller can decide what to do if status is true
    
    """
    if verbose:
        print command
    
    process = subprocess.Popen(command, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    output, stderr = process.communicate()
    status = process.poll()
    
    return (output, stderr, status)

[docs]def checkForZipped(filePath):
    """
    Unzip specified file if required.
    
    Parameters
    ----------
    filePath : str
        Path to the file
    
    Returns
    -------
    zipFlag : bool
        True if the file was unzipped, false otherwise
    
    Raises
    ------
    IOError
        If the filePath or a zipped version of it could not be found
    
    """
    if os.path.exists(filePath) and (filePath.endswith('.gz') or filePath.endswith('.bz2')):
        if filePath.endswith('.gz'):
            log(__name__, 'GZIP...', 3)
            filePath = filePath[:-3]
            os.system('gzip -cd ' + filePath + '.gz > ' + filePath)
        
        elif filePath.endswith('.bz2'):
            log(__name__,'BZIP...', 3)
            filePath = filePath[:-4]
            os.system('bzip2 -d -k ' + filePath + '.bz2')
        
        zipFlag = True
    
    elif os.path.exists(filePath):
        zipFlag = False
    
    elif os.path.exists(filePath + '.bz2'):
        log(__name__,'BZIP exists', 3)
        os.system('bzip2 -d -k ' + filePath + '.bz2')
        zipFlag = True
    
    elif os.path.exists(filePath + '.gz'):
        log(__name__, 'GZIP exists', 3)
        os.system('gzip -cd ' + filePath + '.gz > ' + filePath)
        zipFlag = True
    
    else:
        raise IOError("Utilities.checkForZipped: Could not find file: '%s'" % filePath)
    
    return zipFlag, filePath

[docs]def convertStrToType(strValue, typeName):
    try:
        if typeName == 'str':
            return strValue.strip()
        elif typeName == 'int':
            return int(strValue)
        elif typeName == 'float':
            return float(strValue)
        else:
            sys.exit(__name__ + ": convertStrToType: ERROR: undefined type: " + typeName)
    except:
        sys.exit(__name__ + ": convertStrToType: ERROR: cannot convert: " + strValue + " to " + typeName)

[docs]def log(caller, message, level, indent=0):
    """
    Log output to screen
    
    level:
        0: essential output only (minimal)
        1: normal output level for production run
        2: minimal debugging output
        3: extra debugging output
        4: huge amount of output: all force calls, etc
    
    indent:
        how much to indent the output by (defaults to zero)
    
    """
    # only print if less than the level set in Input module
    if level <= Globals.verboseLevel:
        now = datetime.datetime.now().strftime("%d/%m/%y, %H:%M:%S")
        ind = ""
        for _ in xrange(indent):
            ind += "  "
        print "[%s]: %s%s%s >> %s" % (now, Globals.prependLog, ind, caller, message)

[docs]def makeBarrierCriteria(inputString, maxBarrier):
    iterList = []
    enerList = []
    cnt = 0
    
    splits = inputString.split(",")
    
    for split in splits:
        criteria = split.split("-")
        if len(criteria) == 2:
            iteration = convertStrToType(criteria[0], 'int')
            multiplyer = convertStrToType(criteria[1], 'float')
            
            if iteration > 0 and multiplyer > 0 and maxBarrier > 0:
                cnt += 1
                
                energy = maxBarrier * multiplyer
                
                iterList.append(iteration)
                enerList.append(energy)
                                
    return cnt, iterList, enerList

[docs]def which(program):
    """
    Check if the given path or program exists and is executable.
    
    Parameters
    ----------
    program : str
        Path to a program or the name of a program
    
    Returns
    -------
    exe_file : str
        The path to call the program or None if it doesn't exist
    
    Notes
    -----
    See http://stackoverflow.com/questions/377017/test-if-executable-exists-in-python
    
    """
    fpath, fname = os.path.split(program)
    if fpath:
        if is_exe(program):
            return program
    else:
        for path in os.environ["PATH"].split(os.pathsep):
            path = path.strip('"')
            exe_file = os.path.join(path, program)
            if is_exe(exe_file):
                return exe_file

    return None
