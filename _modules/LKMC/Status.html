[docs]    def resetSearches(self, currentTime, currentState=None, combinedList=[]):
        """
        Reset the searches start time

        """
        self.NVolumes = 0
        self.NVolumesTotal = 0
        self.hashKeys = {}
        self.searchType = {}
        self.NTotal = {}
        self.NComplete = {}
        self.NSuccess = {}
        self.NFail = {}
        self.NDuplicate = {}

        self.transMovedVolumes = False
        self.NMovedTo = {}
        self.NMovedFrom = {}

        if currentState is not None:
            self.NVolumes = len(currentState.defectVolumesKeys)
            self.NVolumesTotal = self.NVolumes

            for i, dv in enumerate(currentState.defectVolumes):
                self.hashKeys[i] = dv.graphKey
                self.searchType[i] = ""
                self.NTotal[i] = 0
                self.NComplete[i] = 0
                self.NSuccess[i] = 0
                self.NFail[i] = 0
                self.NDuplicate[i] = 0
                if dv.combinedVol:
                    self.isCombinedVol[i] = "Y"
                else:
                    self.isCombinedVol[i] = "N"
                self.NMovedTo[i] = 0
                self.NMovedFrom[i] = 0

        self.NCombined = len(combinedList)
        for volIndex in combinedList:
            self.hashKeys[volIndex] = "COMBINED VOLUME"
            self.searchType[volIndex] = "SEARCH"
            self.NTotal[volIndex] = 0
            self.NComplete[volIndex] = 0
            self.NSuccess[volIndex] = 0
            self.NFail[volIndex] = 0
            self.NDuplicate[volIndex] = 0

            self.NVolumesTotal += 1

        self.searchesStartTime = currentTime



[docs]    def updateSearchInfo(self, searchList=[], reuseList=[],reuseAndSearchList=[], NSearches={},fixedRefineNum=0):
        """
        Update info about searches

        """
        for volIndex in searchList:
            self.searchType[volIndex] = "SEARCH"

        for volIndex in reuseList:
            self.searchType[volIndex] = "REUSE"

        for volIndex in reuseAndSearchList:
            self.searchType[volIndex] = "REUSE + SEARCH"

        for volIndex, N in NSearches.items():
            self.NTotal[volIndex] = N - fixedRefineNum

[docs]    def updateSearches(self, NCompleted, NSuccessful, NUnique, result=None, status=None, storeVolIndex=None):
        """
        Update number of searches.

        kwarg status:
          * 0 => Failed during search
          * 1 => Duplicate
          * 2 => Failed during filtering
          * 3 => Success

        """
        if result is not None and status is not None:
            resultVolIndex = result.currentVolIndex

            # if storeVolIndex is not the same as volIndex then we handle this differently
            if storeVolIndex is not None and storeVolIndex != resultVolIndex:
                self.transMovedVolumes = True

                self.NMovedFrom[resultVolIndex] += 1
                self.NMovedTo[storeVolIndex] += 1

                volIndex = storeVolIndex

            else:
                volIndex = resultVolIndex

            # add to number completed (original volume always)
            self.NComplete[resultVolIndex] += 1

            # add to other counters (stored volume)
            if status == 3:
                self.NSuccess[volIndex] += 1
            elif status == 1:
                self.NDuplicate[volIndex] += 1
            else:
                self.NFail[volIndex] += 1



[docs]    def writeStatus(self, filename=None, outdir=None):
        """
        Write status to file

        """
        if filename is None:
            filename = self.filename

        currentTime = time.time()

        lines = []
        nl = lines.append

        # header
        nl("<!DOCTYPE html>")
        nl("<html>")
        nl("<head>")

        # refresh every 5 secs
        nl('<META HTTP-EQUIV="REFRESH" CONTENT="5">')

        # title
        nl("<title>LKMC Status</title>")
        nl("</head>")

        # body
        nl("<body>")
        nl("<h1>LKMC Status</h1>")

        # KMC status section
        nl("<h2>KMC job status</h2>")
        nl("<p>Current step: %d</p>" % (self.KMCStep,))
        nl("<p>Current simulation time: %g s</p>" % (self.simTime,))
        nl("<p>Run time: %g s</p>" % (currentTime - self.KMCStartTime,))

        # transitions/volumes section
        nl("<h2>Transition finding status</h2>")

        if self.NVolumes:
            # table
            nl('<table border="1">')

            # header row
            nl("<tr>")

            # cell
            nl("<th>Volume &#35;</th>")
            nl("<th>Key</th>")
            nl("<th>CV</th>")
            nl("<th>Action</th>")
            nl("<th>Total &#35;</th>")
            nl("<th>Completed</th>")
            nl("<th>Success</th>")
            nl("<th>Duplicate</th>")
            nl("<th>Failure</th>")

            if self.transMovedVolumes:
                nl("<th>Moved from</th>")
                nl("<th>Moved to</th>")

            nl("</tr>")

            # content rows
            for i in xrange(self.NVolumesTotal):
                # row
                nl("<tr>")

                nl("<td>%d</td>" % i)
                nl("<td>%s</td>" % self.hashKeys[i])
                nl("<td>%s</td>" % self.isCombinedVol[i])
                nl("<td>%s</td>" % self.searchType[i])
                nl("<td>%d</td>" % self.NTotal[i])
                nl("<td>%d</td>" % self.NComplete[i])
                nl("<td>%d</td>" % self.NSuccess[i])
                nl("<td>%d</td>" % self.NDuplicate[i])
                nl("<td>%d</td>" % self.NFail[i])

                if self.transMovedVolumes:
                    nl("<td>%d</td>" % self.NMovedFrom[i])
                    nl("<td>%d</td>" % self.NMovedTo[i])

                nl("</tr>")

            nl("</table>")

            nl("<p>Run time: %g s</p>" % (currentTime - self.searchesStartTime,))

        if self.reuseStats is not None:
            nl("<h2>Reuse statistics</h2>")

            # table
            nl('<table border="1">')

            # header row
            nl("<tr>")

            # cell
            nl("<th>Volume key</th>")
            nl("<th>Times reused</th>")

            nl("</tr>")

            for key, value in self.reuseStats.items():
                # row
                nl("<tr>")

                nl("<td>%s</td>" % key)
                nl("<td>%d</td>" % value)

                nl("</tr>")

            nl("</table>")

        # footer
        nl("</body>")
        nl("</html>")
        nl("")

        # create text object
        textObj = Utilities.TextObject(lines, textAsLines=True)

        # send to writer
        item = self.writer.writerItem("WRITE", dataType="object", data=[textObj,], outdir=outdir)
        method = "write"
        args = (filename,)
        kwargs = {}
        item.commands.append([method, args, kwargs])
        self.writer.add(item)
