[docs]    def converge(self, lattice, searchMd=1, looseIterMax=0):
        """
        The idea is to approach saddle state with loose tolerance L-BFGS-B + Lanczos and than uses quite tight tolerance
        for convergence. Also restart of L-BFGS-B is recommended every 10-20 steps.
        
        """
        self.setup(searchMd, looseIterMax)
        self.failReason = None
        
        log(__name__, "First we do couple of steps of Lanczos - L-BFGS-B...", 2, 2)

        # first we run loose tolerance sub-lattice finder to approach the saddle state
        status = self.lanczosLooseSubMinimiser.run(lattice)
        
        self.forceCalls = lattice.NMinLanczosFE
        
        log(__name__, "Now we do Lanczos - L-BFGS-B with restarts every %d steps" % (self.params.LBFGSMaxIter), 2, 2)
         
        itern = 0
        while not status and (itern == 0 or lattice.LBFGSConvTask == 3):
            status = self.lanczosSubMinimiser.run(lattice)
            
            self.forceCalls += lattice.NMinLanczosFE
            
            if not status and lattice.LBFGSConvTask == 2 and self.params.LBFGSLanczosConvgOnlyByTolerance:
                
                log(__name__, "Convergence by _FACTR*EPSMCH is not allowed!!!", 1, 1)
                
                # if this case occurs during searches, give up on direction
                if searchMd:
                    status = 1                    
                else:
                    # otherwise, try again
                    status = 2
                     
            itern += 1
        
        lattice.mainIndexes = None
        lattice.includeAtoms = None
        
        lattice.rVector = copy.deepcopy(self.lanczosSubMinimiser.rVector)
        lattice.rVectorIndexes = copy.deepcopy(self.lanczosSubMinimiser.mainIndexesFinal)
        
        return status

            includeAtoms=np.empty(0, np.int32), mainIndexes=np.empty(0, np.int32)):
        
        self.params.lanczosMaxIter
        
        if len(includeAtoms) and len(mainIndexes):
            mainIdx = mainIndexes
            inclIdx = includeAtoms
        
        elif len(self.includeAtoms) and len(self.mainIndexes):
            mainIdx = self.mainIndexes
            inclIdx = self.includeAtoms
        
        else:
            print "PROBLEM!!!"
            return 1
        
        status = super(LanczosLattice, self).calcForce(MDPath=MDPath, defectFlag=defectFlag, correctTE=correctTE, 
            includeAtoms=inclIdx, mainIndexes=mainIdx)
        
        if self.invertForceComponent:
            
            iFCBackup = self.invertForceComponent
            self.invertForceComponent = 0
            
            status, lle, llev, fEvals, iterCnt, self.rVector, _ = lanczosMethodForASystem(self, 
                mainIdx, self.rVector, self.params, normEigenVec=0, includeAtoms=inclIdx)
            
            self.invertForceComponent = iFCBackup
        
            if status:
                self.force = np.zeros(3 * self.NAtoms, np.float64)
            else:                
                log(__name__, "Lowest eigenvalue %f..." % (lle), 1, 1) 
                                   
                if (lle >= self.params.lanczosNegExtCrit):

                else:
                    newForce, deltaE = lanczosMinModNewForce(self.force, llev, lle)
                    
                    self.force = newForce
                    self.totalEnergy += deltaE

        return status

                log(__name__, "Lowest eigenvalue %f..." % (lle), 1, 1)

                    self.object.force = np.zeros(3 * self.object.NAtoms, np.float64)
                    self.status = 5
            
            self.mainIndexesFinal = mainIndexes
        
        return self.status

                self.object.force = np.zeros(3 * self.object.NAtoms, np.float64)

                self.status = 5
        
        self.mainIndexesFinal = mainIndexes
        
        return (self.object.totalEnergy, self.object.force * -1.0)

                if ((lle >= self.params.lanczosNegExtCrit) and (self.NForceCalls != 1)):
                    self.object.force = np.zeros(3 * self.object.NAtoms, np.float64)
                    log(__name__, "Lowest eigenvalue is positive (%f) (> %f)..." % (lle, self.params.lanczosNegExtCrit), 1, 1)
                    self.status = 1
                    
                else:
                    FdI = np.dot(self.object.force, llev)
                    
                    # inverting the force along the lowest eigenvector direction
                    self.object.force -= FdI * llev + self.forceAlpha * llev
                    
                    # Using Newton's method in optimization to calculate delta x
                    # deltaX = -1.0 * FdI / lle (-1.0 removed to save time)
                    deltaX = FdI / lle
                    deltaX2 = deltaX * deltaX
                    
                    deltaE = -0.5 * (lle - self.forceAlpha) * deltaX2
                    
                    self.object.totalEnergy += deltaE 
                    
        self.previousEnergy = self.object.totalEnergy
        
        if self.NForceCalls == 0:
            self.iniEnergy = self.object.totalEnergy

                self.status = 5
        
        self.mainIndexesFinal = mainIndexes
        
        return (self.object.totalEnergy, self.object.force * -1.0)



[docs]def calcEigenValues(system, mainIndexes, includeAtoms, displVector, params, lnegmd=0, evn=None, noEigenVects=0, jobID=None):
    """
    Calculates system's eigenvalues using Lanczos method.
    
    """
    timeSt = time.time()
    lattice = copy.copy(system)
    
    minTol = params.lanczosConvTol
    iterMax = min(params.lanczosMaxIter, len(mainIndexes)*3)
        
    result = 1
    iterCnt = 0
    forceEvalSum = 0
    eigenValues = None
    eigenVectors = None
    
    if params.useSubLatticeForceCalc:
        if not len(includeAtoms):
            includeAtoms = lattice.buildSubLattice(mainIndexes, params.subLatticeIncludeRadius)
                    
    if displVector is None:
        if params.debugLanczos:
            rdmSeed = 10
        else:
            rdmSeed = None

        displVector = Vectors.normalise(Vectors.randomVector(len(mainIndexes)*3, randomSeed=rdmSeed))
    
    if evn is None:
        evn = params.lanczosConvByEigenVNo
    
    r = copy.deepcopy(displVector)
    sigma = Vectors.magnitude(r)

    eigenOld = 1000.0
    eigenDiff = 0.0
    evectOld = None
        
    alphaVecTmp = []
    sigmaVecTmp = []
    qVecTmp = []
    
    q = 0
    z = None

    while (iterCnt < iterMax) and (iterCnt < len(displVector)):
        
        # check job status to stop
        if iterCnt % 2 == 0 and jobID is not None:
            if ClientServerInterface.checkJobStatus(jobID):
                timeElapsed = time.time() - timeSt
                log(__name__, "eigenvalue calculation terminated due to job status",2,2)
                return 1, eigenValues, None, None, iterCnt, forceEvalSum, timeElapsed
        
        if sigma == 0:
            timeElapsed = time.time() - timeSt
            return 1, eigenValues, None, None, iterCnt, forceEvalSum, timeElapsed
        
        qOld = copy.copy(q)
        q = r / sigma
    
        if len(qVecTmp) > 0:
            nq = np.array(qVecTmp,dtype=float).T[:,-iterMax::]
            
            if not np.isfinite(q).all() or not np.isfinite(nq).all():
                timeElapsed = time.time() - timeSt
                return 1, eigenValues, None, None, iterCnt, forceEvalSum, timeElapsed

            q = q - np.dot(nq, linalg.lstsq(nq,q)[0])

        status, u, fEval = calcHessianVectProduct(lattice, mainIndexes, includeAtoms, q, params)
        forceEvalSum += fEval
        
        if status:
            timeElapsed = time.time() - timeSt
            return 1, eigenValues, None, None, iterCnt, forceEvalSum, timeElapsed

        r = u - sigma*qOld
        
        if  len(q) != len(r):
            timeElapsed = time.time() - timeSt
            return 1, eigenValues, None, None, iterCnt, forceEvalSum, timeElapsed
        
        alpha = np.dot(q, r)
        alphaVecTmp.append(alpha)
        
        qVecTmp.append(q)
        
        r = r - alpha*q
        
        sigma = Vectors.magnitude(r)
      
        if iterCnt > params.lanczosMinIter:
            alphaVec = np.array(alphaVecTmp, np.float64)
            betaVec = np.array(sigmaVecTmp, np.float64)
            
            status, eigenValues, eigenVectors = Algebra.getEigenvaluesST(alphaVec, betaVec, -1, noEigenVects=noEigenVects)
            
            if status:
                timeElapsed = time.time() - timeSt
                return 1, eigenValues, None, None, iterCnt, forceEvalSum, timeElapsed
            
            #eigenDiff = eigenOld-eigenValues[evn]
            eigenDiff = math.fabs((eigenOld-eigenValues[evn])/eigenOld)

            eigenOld = eigenValues[evn]
                        
            if(eigenDiff < minTol):
                iterCnt += 1
                result = 0
                break
        
        sigmaVecTmp.append(sigma)
        
        iterCnt += 1
            
    lneidx = 0
    
    if lnegmd:
        for i in range(0, evn+1):
            if eigenValues[i] >= 0.0:
                break
            lneidx = i
    
    if eigenValues is None:
        timeElapsed = time.time() - timeSt
        return 1, eigenValues, None, None, iterCnt, forceEvalSum, timeElapsed
    else:
        eigenValue = eigenValues[lneidx]
    
    if noEigenVects:
        evectLB = None
        eigenValues = None
    else:
        evect = eigenVectors[:, lneidx]
        
        # Orthogonal basis of Lanczos vectors
        nq = np.array(qVecTmp, np.float64).T
        
        # Transforming lowest eigenvalue vector to Lanczos orthogonal basis
        evectLB = np.dot(nq, evect.T)
        
    timeElapsed = time.time() - timeSt
    
    return result, eigenValues, eigenValue, evectLB, iterCnt, forceEvalSum, timeElapsed



[docs]def calcHessianVectProduct(lattice, mainIndexes, includeAtoms, displVectorNormalised, parameters):
    """
     Method approximates a dot product between a Hessian matrix and a vector of a 
        given system.
        lattice: input lattice
        mainIndexes: list of atoms that should be included in Hessian matrix calculation
        displVectorNormalised: a random normalised vector
        parameters: input parameters from lkmcInput.IN
     Returns:
        hessianU: a dot product of a hessian matrix and a normalized vector of a 
            given system.
        forEvalCnt: number of force evaluations used to calculate hessian matrix
    
    REMARKS: be sure that the passed lattice has an evaluated force vector
    """
    wholeSystemMd = 0
    status = 0
    try:
        mainIndexesLen = len(mainIndexes)
    except:
        mainIndexesLen = 0
    
    if mainIndexesLen == 0:
        wholeSystemMd = 1
        mainIndexesLen = lattice.NAtoms
    
    hessianU = np.zeros(3 * mainIndexesLen, np.float64)
    
    tempLattice = copy.deepcopy(lattice)
    
    tempDispl = parameters.hessianDispl
    
    # what to use for main indexes
    if parameters.lanczosLocaliseHessian:
        mainIndexesUse = tempLattice.defectSubSystem
    else:
        mainIndexesUse = mainIndexes
    
    # do step left first if needed
    if parameters.hessianApproxMode != 1:
        tempDispl *= -1.0
        
        displVector = tempDispl * displVectorNormalised
        
        tempLattice.initiateAtomsDisplacement(mainIndexes, displVector)
        
        if parameters.useSubLatticeForceCalc:
            status = tempLattice.calcForce(mainIndexes=mainIndexesUse, includeAtoms=includeAtoms)
        else:
            status = tempLattice.calcForce()
        
        if status:
            return status, hessianU, 0
        
        leftStepForce = copy.deepcopy(tempLattice.force)
        
        tempDispl = 2.0 * parameters.hessianDispl
        
    displVector = tempDispl * displVectorNormalised
    tempLattice.initiateAtomsDisplacement(mainIndexes, displVector)
    
    if parameters.useSubLatticeForceCalc:
        status = tempLattice.calcForce(mainIndexes=mainIndexesUse, includeAtoms=includeAtoms)
    else:
        status = tempLattice.calcForce()
        
    if status:
        return status, hessianU, 0
    
    if parameters.hessianApproxMode == 1:
        forEvalCnt = 1
        
        if wholeSystemMd:
            for i in range(3 * mainIndexesLen):
                hessianU[i] = (lattice.force[i] - tempLattice.force[i]) / parameters.hessianDispl
        else:
            c_lanczos.calcHessianUSubSystem(3, parameters.hessianDispl, mainIndexesLen, mainIndexes, 
                                            hessianU, lattice.force, tempLattice.force)
    else:
        forEvalCnt = 2
        
        if wholeSystemMd:
            for i in range(3 * mainIndexesLen):
                hessianU[i] = (leftStepForce[i] - tempLattice.force[i]) / (2.0*parameters.hessianDispl)
        else:
            c_lanczos.calcHessianUSubSystem(3, 2.0*parameters.hessianDispl, mainIndexesLen, mainIndexes, 
                                            hessianU, leftStepForce, tempLattice.force)
        
    return status, hessianU, forEvalCnt

[docs]def lanczosMinModNewForce(force, llev, lle):
    
    """
    Modifies the force and the energy of the system for climing up using the eigenvalue and the 
    eigenvector from the lanczos method.
    
    """
    
    FdI = np.dot(force, llev)
                    
    # inverting the force along the lowest eigenvector direction
    newForce = force - 2.0 * FdI * llev
                    
    # Using Newton's method in optimization to calculate delta x
    # deltaX = -1.0 * FdI / lle (-1.0 removed to save time)
    deltaX = FdI / lle
    deltaX2 = deltaX * deltaX
                    
    # Energy change using Taylor series: 
    # deltaE = -0.5 * lle * deltaX2 (-0.5 removed to save time)
    deltaE = lle * deltaX2
                    
    # Adding twice us much due to inverse of the force
    # totalEnergy += deltaE * 2.0 (+= 2.0* removed to save time)  
    
    return newForce, -deltaE
