[docs]def histogram(filename, data, xlabel, ylabel, title=None, grid=0, textsize=24, binWidth=0.05, ymax=None, xmax=None):
    """
    Plot distribution of barriers.
    
    """
    if not plotImported:
        return
    
    pars = matplotlib.figure.SubplotParams(left=0.14, bottom=0.12, right=None, top=None, wspace=None, hspace=None)
    fig = plt.figure(figsize=(8,6),subplotpars=pars)
    ax = fig.add_subplot(111)
    
    NDatas = len(data)
    
    maxData = max(data)
    minData = min(data)
    
    xmin = 0.0
    if xmax is None:
        xmax = 0.0
        while xmax < maxData:
            xmax += binWidth
    
    # number of bins
    NBins = math.ceil((xmax - xmin) / binWidth)
    
    # histogram
    n, bins, patches = ax.hist(data, NBins, range=(xmin, xmax))
    
    plt.xlabel(xlabel, fontsize=textsize)
    plt.ylabel(ylabel, fontsize=textsize)
    
    if title is not None:
        plt.title(title)
    
    if grid:
        plt.grid()
    
    if ymax is not None:
        plt.ylim(ymax=ymax)
    
    for tick in ax.xaxis.get_major_ticks():
        tick.label1.set_fontsize(textsize)
    for tick in ax.yaxis.get_major_ticks():
        tick.label1.set_fontsize(textsize)
    
    plt.savefig(filename+".png")
    
    plt.clf()
    plt.close()

[docs]def stackedBarChart(filename, xvals, y1avals, y1bvals, y2vals, y1label="", y2label="", y1aleg="", y1bleg="", y2leg="", title=None, grid=1, textsize=20, align="edge", width=0.35):
    
    if not plotImported:
        return
    
    ind = np.arange(len(xvals))
    
    pars = matplotlib.figure.SubplotParams(left=0.14, bottom=0.12, right=None, top=None, wspace=None, hspace=None)

    ax1 = fig.add_subplot(111)
    
    p1 = plt.bar(ind, y1avals, width, color='r')
    p2 = plt.bar(ind, y1bvals, width, color='y', bottom=y1avals)
    
    ax2 = ax1.twinx()
    
    p3 = plt.bar(ind+width, y2vals, width, color='b')
    
    plt.xticks(ind+width, xvals, rotation='vertical')

    plt.legend( (p1[0], p2[0], p3[0]), (y1aleg, y1bleg, y2leg), shadow=True, borderaxespad=0.)
    
    ax1.set_ylabel(y1label, fontsize=textsize)
    ax2.set_ylabel(y2label, fontsize=textsize)

[docs]def linePlot(filename, xvals, xlabel, yvals, ylabel, title=None, label=None, grid=1, textsize=20, useInterp=1, yerror=None,
             xlimmin=None, xlimmax=None, ylimmin=None, ylimmax=None, ymax=None, tight=True):
    """
    Do a line plot of data using stineman interpolation.
    'ymax' should be specified as the index in the arrays of the maximum
                (so that interpolation doesn't change it).
    'label' goes in the legend if you specify it.
    Saves graph in 'filename'.pdf.
    
    """
    if not plotImported:
        return
    
    pars = matplotlib.figure.SubplotParams(left=0.14, bottom=0.12, right=None, top=None, wspace=None, hspace=None)
    fig = plt.figure(figsize=(8,6),subplotpars=pars)
    ax = fig.add_subplot(111)
    yp = None
    
    if ymax is not None:
        # make sure the max value is the max after interpolation
        length = len(xvals)
        
        # interpolate and plot up to max
        if useInterp:
            xi = scipy.linspace( xvals[0], xvals[ymax], 100 )
            yi = pylab.stineman_interp(xi, xvals[0:ymax+1], yvals[0:ymax+1], yp)
            ax.plot(xi, yi, '-', linewidth=2, color="blue")
        
            # interpolate and plot from max
            xi = scipy.linspace(xvals[ymax], xvals[-1], 100)
            yi = pylab.stineman_interp(xi, xvals[ymax:length], yvals[ymax:length], yp)
            ax.plot(xi, yi, '-', linewidth=2, color="blue")
        
        # plot points
        ax.errorbar(xvals, yvals, fmt='o', yerr=yerror, linewidth=2, color="red", ecolor="blue", label=label)
    
    else:
        # interpolate and plot
        if useInterp:
            xi = scipy.linspace( xvals[0], xvals[-1], 100 )
            yi = pylab.stineman_interp(xi, xvals, yvals, yp)
        
            ax.plot(xi, yi, '-', linewidth=2, color="blue", label="Stineman")
            
        ax.errorbar(xvals, yvals, fmt='o', yerr=yerror, linewidth=2, color="red", ecolor="blue", label=label)
    
    plt.xlabel(xlabel, fontsize=textsize)
    plt.ylabel(ylabel, fontsize=textsize)
    if title is not None:
        plt.title(title)
    
    for tick in ax.xaxis.get_major_ticks():
        tick.label1.set_fontsize(textsize)
    for tick in ax.yaxis.get_major_ticks():
        tick.label1.set_fontsize(textsize)
    
    if label is not None:
        leg = plt.legend(loc='best')
        for t in leg.get_texts():
            t.set_fontsize(textsize)
    
    if xlimmin is not None:
        plt.xlim(xmin=xlimmin)
    
    if xlimmax is not None:
        plt.xlim(xmax=xlimmax)
    
    if ylimmin is not None:
        plt.ylim(ymin=ylimmin)
    
    if ylimmax is not None:
        plt.ylim(ymax=ylimmax)
    
    if grid:
        plt.grid()
    
    if tight:
        plt.tight_layout()
    
    plt.savefig(filename+".png")
    
    plt.clf()
    plt.close()

[docs]def multiLinePlot( filename, xvalsList, xlabel, yvalsList, ylabel, labels, ymax=0, ymin=0, title=None, grid=1, textsize=20, colours=None, useInterp=1 ):
    """
    Do a line plot of multiple sets of data.
    Data is interpolated using Stineman interpolation
    'xvalsList': list of arrays containing xvals for each set of data
    'yvalsList': corresponding list of arrays containing y values
    'labels': list containing a label for each set of data
    'ymax': boolean, if true will interpolate up to and down
        from the max value
    'ymin': boolean, equivalent to ymax
    Saves graph in 'filename'.pdf.
    
    """
    if not plotImported:
        return
    
    pars = matplotlib.figure.SubplotParams(left=0.14, bottom=0.12, right=None, top=None, wspace=None, hspace=None)
    fig = plt.figure(figsize=(8,6),subplotpars=pars)
    ax = fig.add_subplot(111)
    yp = None
    
    if colours is None:
        colours = ["red", "blue", "green", "yellow", "orange", "black"]
    
    for plotcount in xrange(len(xvalsList)):
        xvals = xvalsList[plotcount]
        yvals = yvalsList[plotcount]
        label = labels[plotcount]
        
        if ymax or ymin:
            if ymax:
                index = np.argmax(yvals)
            elif ymin:
                index = np.argmin(yvals)
            
            # make sure the max value is the max after interpolation
            length = len(xvals)
            
            if useInterp:
                # interpolate and plot up to max
                xi = scipy.linspace( xvals[0], xvals[index], 100 )
                yi = pylab.stineman_interp(xi, xvals[0:ymax+1], yvals[0:index+1], yp)
                ax.plot(xi, yi, '-', linewidth=2, color=colours[plotcount % len(colours)])
                
                # interpolate and plot from max
                xi = scipy.linspace(xvals[ymax], xvals[-1], 100)
                yi = pylab.stineman_interp(xi, xvals[index:length], yvals[index:length], yp)
                ax.plot(xi, yi, '-', linewidth=2, color=colours[plotcount % len(colours)])
            
            # plot points
            ax.plot(xvals, yvals, 'o', linewidth=2, color=colours[plotcount % len(colours)], label=label)
        
        else:
            if useInterp:
                # interpolate and plot
                xi = scipy.linspace( xvals[0], xvals[-1], 100 )
                yi = pylab.stineman_interp(xi, xvals, yvals, yp)
                ax.plot(xi, yi, '-', linewidth=2, color=colours[plotcount % len(colours)])
            
            ax.plot(xvals, yvals, 'o', linewidth=2, color=colours[plotcount % len(colours)], label=label)
    
    plt.xlabel(xlabel, fontsize=textsize)
    plt.ylabel(ylabel, fontsize=textsize)
    if title is not None:
        plt.title(title)
    
    for tick in ax.xaxis.get_major_ticks():
        tick.label1.set_fontsize(textsize)
    for tick in ax.yaxis.get_major_ticks():
        tick.label1.set_fontsize(textsize)
    
    leg = plt.legend(loc='best')
    for t in leg.get_texts():
        t.set_fontsize(textsize)
    
    if grid:
        plt.grid()
    
    plt.savefig(filename+".png")
    
    plt.clf()
    plt.close()
