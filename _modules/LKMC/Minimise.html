[docs]def getMinimiser(params, jobID=None):
    """
    Return the minimiser set in the `params` object.
    
    Parameters
    ----------
    params : `Input.InputParams`
        Parameters instance.
    jobID : `uuid`
        Unique identifier to be passed to the minimiser. Used for stopping
        repeating jobs on clients.
    
    Returns
    -------
    minimiser : 
        The selected minimiser object. A child of `BaseMinimiser`.
    
    Raises
    ------
    NotImplementedError
        If the requested minimiser is not recognised.
    
    """
    name = params.minimizationType
    
    # return required minimiser
    if name == "LBFGS":
        minimiser = LBFGSMinimiser(params)
        minimiser.setJobID(jobID)
    
    elif name ==  "LBFGSMinMode":
        minimiser = LBFGSMinimiser(params)
        minimiser.setJobID(jobID)
    
    elif name == "CG":
        minimiser = ConjugateGradientMinimiser(params)
        minimiser.setJobID(jobID)
    
    elif name == "SD":
        minimiser = SteepestDescentMinimiser(params)
        minimiser.setJobID(jobID)
    
    elif name == "CGSCIPY":
        minimiser = ConjugateGradientSciPyMinimiser(params)
        minimiser.setJobID(jobID)
    
    else:
        raise NotImplementedError("Minimiser %s not implemented" % name)
    
    return minimiser

[docs]def getLineSearcher(params, calcForce):
    """
    Return the line searcher set in the `params` object.
    
    Parameters
    ----------
    params : `Input.InputParams`
        Parameters instance.
    calcForce : callable
        The method the line searcher should use to calculate forces,
        for example `BaseMinimiser.calcForce`.
    
    Returns
    -------
    lineSearcher : 
        The selected line searcher object.
    
    Raises
    ------
    NotImplementedError
        If the requested line searcher is not recognised.
    
    """
    name = params.minimizationLineSearch
    
    if name == "secant":
        lineSearcher = SecantLineSearcher(params, calcForce)

    elif name == "modNewton":
        lineSearcher = ModifiedNewtonLineSearcher(params, calcForce)
        
    elif name == "oneStepper":
        lineSearcher = OneStepLineSearcher(params, calcForce)
    
    elif name == "scipy":
        lineSearcher = ScipyLineSearcher(params, calcForce)
        
    else:
        raise NotImplementedError("Line searcher %s not implemented" % name)
    
    return lineSearcher

        """
        Run the minimiser.
        
        Parameters
        ----------
        obj : 
            An object that can be minimised, eg. `Lattice.Lattice` or 
            `NEB.NEB`.
        
        Returns
        -------
        status : integer
            0 means success, anything else is a failure.
        
        """
        # initialise some counters on object
        obj.minimisationTime = 0.0
        obj.startTime = time.time() 
        obj.NMinFE = 0
        obj.NMinWholeFEvals = 0
        obj.NMinExclFEvals = 0
        
        # run main minimiser
        status = self.runMain(obj)
        
        # finalise counters
        if self.params.timeMinimization:
            obj.minimisationTime = time.time() - obj.startTime
        else:
            obj.minimisationTime = -1.0
                
        if hasattr(obj, "wrapAtoms"):
            obj.wrapAtoms()
        
        return status

[docs]    def calcForce(self, obj, excl, params, includeAtoms=None, mainIndexes=None):
        """
        Force calculation.
        
        Parameters
        ----------
        obj : 
            The object being minimised.
        excl : {0, 1, 2}
            The defect flag to set when calling `calcForce` on `obj`.
        params : `Input.InputParams`
            The parameters instance.
        mainIndexes : ndarray, optional
            An array containing the indexes of the atoms that we need to 
            accurately determine the forces for.
        includeAtoms : ndarray, optional
            An array containing the indexes of the atoms to include during
            the force calculation (for a sub-lattice force calculation). 
            The forces on these atoms will not be correct but including 
            them ensures the forces on the `mainIndexes` atoms are.
        
        Returns
        -------
        status : integer
            0 mean success, anything else means a failure.
        
        """
        if Globals.taskStatus is not None:
            if Globals.taskStatus.value:
                return 111
        
        exclFEvalCount = params.minimizationExclusionReverseEveryFEval
        
        if exclFEvalCount > 0:
            if obj.NMinFE % exclFEvalCount == 0:
                if excl == 0 or excl == 1:
                    exclCalcForce = 2
                else:
                    exclCalcForce = 0
            else:
                exclCalcForce = excl
        else:
            exclCalcForce = excl
        
        if includeAtoms is not None and mainIndexes is not None and params.useSubLatticeForceCalc:
            status = obj.calcForce(defectFlag=exclCalcForce, includeAtoms=includeAtoms, mainIndexes=mainIndexes)
        else:
            status = obj.calcForce(defectFlag=exclCalcForce)
        
        if (excl == 2 and Globals.useSubLatticeForceCalc):
            obj.NMinWholeFEvals += 1
        else:
            obj.NMinExclFEvals += 1
        
        obj.NMinFE += 1
        
        elapsedTime = time.time() - obj.startTime
        
        if elapsedTime > params.minimisationMaxRunTime:
            log(__name__, "Max runtime is exceeded =  %f < %f" % (params.minimisationMaxRunTime, elapsedTime), 2, 1)
            status = 1
        
        if params.NEBUseBarrierCriteria:
            if params.NEBBarrCritCnt > 0:
                
                if params.NEBBarrCritCnt > 0:
                    #log(__name__, "After %d steps barrier is %f. MAX BARRIER %f." % (lattice.NMinFE, lattice.getBarrier(), params.NEBBarrList[0]), 2, 1)
                
                    if params.NEBBarrIterList[0] == obj.NMinFE:
                        if obj.getBarrier() > params.NEBBarrList[0]:
                            log(__name__, "After %d steps barrier is higher than %f. BYE BYE." % (obj.NMinFE, params.NEBBarrList[0]), 2, 1)
                            status = 1
                            
                        else:
                            params.NEBBarrCritCnt -= 1
                            params.NEBBarrIterList.pop(0)
                            params.NEBBarrList.pop(0)   
        
        return status

[docs]    def runMain(self, obj):
        """
        Run minimiser: minimises "obj" in place.
        
        """
        # params
        self.criteria = self.params.minimizationCriteria
        self.tol = self.params.minimizationTolerance
        self.maxIter = self.params.minimisationMaxIterations
        self.minDebug = self.params.minimisationDebug
        self.noOutput = self.params.minimisationNoOutput
        self.exlusion = self.params.minimizationExclusion
        
        step = 0.2
        
        if not self.noOutput:
            log(__name__,"Steepest descent minimiser. Tolerance is %f (%s)" % (self.tol, self.criteria), 1)
        
        # store initial position
        initialPos = copy.deepcopy(obj.pos)
        
        tol = self.tol
        crit = self.criteria
        maxIter = self.maxIter
        minDebug = self.minDebug
        params = self.params
                
        # initial force calc
        status = self.calcForce(obj, self.exlusion, self.params)
        if status:
            obj.pos = initialPos
            return status
        
        # initial criteria value
        critVal = self.tol + 10
        
        energy = obj.totalEnergy
        minStepsEnergy.append(energy)
        origEnergy = energy
        magF = Vectors.magnitude(obj.force)
        if crit == "force":
            critVal = magF
        obj.NMinimisationSteps = 0
        
        # main minimisation loop
        count = 0
        
        if not self.noOutput:
            log(__name__, "STEP %5d: E %#E; FMAG %#E; FMXA %#E (%6d); FMXE %#E" % (obj.NMinimisationSteps, energy, magF, obj.maxForce, obj.maxForceAtomNo, 
                                                                                   np.max(np.absolute(obj.force))), 2, 1)
        
        while (critVal > tol and (count < maxIter)):
            energyOld = energy
        
            # store current force
            FOld = copy.deepcopy(obj.force)
            magFOld = magF
            
            # search direction
            direction = Vectors.normalise(obj.force)
            
            # move forwards
            obj.pos += step * direction
            
            # force calculation
            status = self.calcForce(obj, self.exlusion, self.params)
            if status:
                obj.pos = initialPos
                return status
            
            # calc energy change and magnitude of force
            energy = obj.totalEnergy
            EChange = energyOld - energy
            magF = Vectors.magnitude(obj.force)
            
            # step size
            forceDot = Vectors.dotProduct(obj.force, FOld)
            if forceDot < 0:
                step /= 2
            
            else:
                step *= 1.2
            
            # determine stop value
            if crit == "energy":
                critVal = abs(EChange)
            elif crit == "force_element":
                critVal = np.max(np.absolute(obj.force))
            else:
                critVal = obj.maxForce
            
            obj.NMinimisationSteps += 1
            if not params.minimisationNoOutput:
                    log(__name__, "STEP %5d: E %#E; dE %#E; FMAG %#E; FMXA %#E (%6d); FMXE %#E; FEVAL %d" % (obj.NMinimisationSteps, energy, -EChange, magF, 
                                                                                                             obj.maxForce, obj.maxForceAtomNo, 
                                                                                                             np.max(np.absolute(obj.force)), obj.NMinFE), 2, 1)
            
            if self.jobID is not None and count % 20 == 0:
                if ClientServerInterface.checkJobStatus(self.jobID):
                    print "TERMINATING SD DUE TO JOB STATUS"
                    obj.pos = initialPos
                    return 1
                        
            count += 1
        
        if count == maxIter:
            if not params.minimisationNoOutput:
                log(__name__, "Minimisation terminated due to max iterations reached: %d, %d" % (count, maxIter), 1)
            
            # In some cases it is needed to do minimisation for couple of steps and get the output
            if params.minimisationRequestFinal:
                return 0
            else:
                obj.pos = initialPos
                return 1
        
        if not params.minimisationNoOutput:    
            log(__name__, "Minimised: total energy change of %lf eV in %d steps (%d force evaluations)" % (energy - origEnergy, obj.NMinimisationSteps, obj.NMinFE), 1, 1)
        
        return 0

        """
        Run minimiser: minimises "obj" in place.
        
        """
        # params
        self.criteria = self.params.minimizationCriteria
        self.tol = self.params.minimizationTolerance
        self.maxIter = self.params.minimisationMaxIterations
        self.minDebug = self.params.minimisationDebug
        self.noOutput = self.params.minimisationNoOutput
        self.exlusion = self.params.minimizationExclusion
        
        if not self.noOutput:
            log(__name__,"Conjugate gradient minimiser. Tolerance is %f (%s)" % (self.tol, self.criteria), 1)
        
        # line searcher
        lineSearcher = getLineSearcher(self.params, self.calcForce)
        
        # store initial position
        initialPos = copy.deepcopy(obj.pos)
        
        tol = self.tol
        crit = self.criteria
        maxIter = self.maxIter
        minDebug = self.minDebug
        params = self.params
                
        # initial force calc
        if not self.params.useSubLatticeForceCalc:
#            self.includeAtoms = np.asarray([], dtype=np.int32)
#            self.mainIndexes = np.asarray([], dtype=np.int32)
            status = self.calcForce(obj, self.exlusion, self.params)
        else:
            status = self.calcForce(obj, self.exlusion, self.params, includeAtoms=self.includeAtoms, mainIndexes=self.mainIndexes)
        
        if status:
            if params.minimisationRequestFinal:
                return 0
            else:
                obj.pos = initialPos
                return status
        
        # initial search direction
        direction = Vectors.normalise(obj.force)
        
        # initial criteria value
        critVal = self.tol + 10
        
        energy = obj.totalEnergy
        minStepsEnergy.append(energy)
        origEnergy = energy
        magF = Vectors.magnitude(obj.force)
    #    if crit == "force":
    #        critVal = magF
        obj.NMinimisationSteps = 0
        
        # main minimisation loop
        count = 0
        
        if not self.noOutput:
            log(__name__, "STEP %5d: E %#E; FMAG %#E; FMXA %#E (%6d); FMXE %#E" % (obj.NMinimisationSteps, energy, magF, obj.maxForce, obj.maxForceAtomNo, 
                                                                                   np.max(np.absolute(obj.force))), 2, 1)
        
        while critVal > tol and count < maxIter:
            energyOld = energy
        
            # store current force
            FOld = copy.deepcopy(obj.force)
            magFOld = magF
            
            # line search
            reset = lineSearcher.run(obj, direction)
            
            # if the line search failed we reset to steepest descent
            if reset:
                if reset == -1:
                    obj.pos = initialPos
                    return 1
                
                if reset == -2:
                    return 0
                
                if obj.NMinimisationSteps == 0:
                    print "ERROR: trying to reset on first step, something is definitely wrong"
                    obj.pos = initialPos
                    return 1
                
                if not params.minimisationNoOutput:
                    log(__name__, "Resetting to steepest descent", 2, 2)
                
                direction = Vectors.normalise(obj.force)
            
            else:
                # choose new direction (Polak Ribiere method)
                gamma = Vectors.dotProduct(obj.force - FOld, obj.force)
                gamma = gamma / (magFOld * magFOld)
                
                # new direction (differs for rotational minisation)
                if obj.stepType == "rotation":
                    direction = Vectors.normalise(obj.force + gamma * Vectors.magnitude(direction) * Vectors.normalise(obj.theta))
                
                else:
                    direction = Vectors.normalise(obj.force + gamma * direction)
            
                # calc energy change and magnitude of force
                energy = obj.totalEnergy
                EChange = energyOld - energy
                magF = Vectors.magnitude(obj.force)
                
                # determine stop value
                if crit == "energy":
                    critVal = abs(EChange)
                elif crit == "force_element":
                    critVal = np.max(np.absolute(obj.force))
                else:
                    critVal = obj.maxForce
                                        
                minStepsEnergy.append(energy)
            
                obj.NMinimisationSteps += 1
                
                if not params.minimisationNoOutput:
                    log(__name__, "STEP %5d: E %#E; dE %#E; FMAG %#E; FMXA %#E (%6d); FMXE %#E; FEVAL %d" % (obj.NMinimisationSteps, energy, -EChange, magF, 
                                                                                                             obj.maxForce, obj.maxForceAtomNo, 
                                                                                                             np.max(np.absolute(obj.force)), obj.NMinFE), 2, 1)
                
                if minDebug:
                    obj.writeLattice( "relax" + str(obj.NMinimisationSteps) + ".dat")
                
                # reset to steepest descent if we have exceeded a certain number of steps
                # need to add this to the parameter file
                if obj.NMinimisationSteps % self.params.minimisationResetCGInterval == 0 and obj.NMinimisationSteps != 0:
                    if not params.minimisationNoOutput:
                        log(__name__, "Resetting to steepest descent (step %d)" % obj.NMinimisationSteps, 2, 2)
                        
                    direction = Vectors.normalise(obj.force)
            
            if self.jobID is not None and count % 10 == 0:
                if ClientServerInterface.checkJobStatus(self.jobID):
                    log(__name__, "Terminating CG due to job status", 0, 0)
                    obj.pos = initialPos
                    return 1
            
            count += 1
        
        if count == maxIter:
            if not params.minimisationNoOutput:
                log(__name__, "Minimisation terminated due to max iterations reached: %d, %d" % (count, maxIter), 1)
            
            # In some cases it is needed to do minimisation for couple of steps and get the output
            if params.minimisationRequestFinal:
                return 0
            else:
                obj.pos = initialPos
                return 1
        
        if not params.minimisationNoOutput:    
            log(__name__, "Minimised: total energy change of %lf eV in %d steps (%d force evaluations)" % (energy - origEnergy, obj.NMinimisationSteps, obj.NMinFE), 1)
        
        return 0

        """
        Run minimiser
        
        """
        # parameters
        gtol = self.params.minimizationTolerance
        self.excl = self.params.minimizationExclusion
        maxiter = self.params.minimisationMaxIterations
        
        log(__name__, "SciPy CG minimiser. Tol is %g" % (gtol,), 1)
        
        # store initial position
        self.initialPos = copy.deepcopy(obj.pos)
        initialPos = self.initialPos
        
        # store reference to object
        self.object = obj
        
        # initialise
        self.status = 0
        self.NForceCalls = 0
        self.iterationCount = 0
        self.x = None
        
        # run
        try:
            res = optimize.fmin_cg(self.f, obj.pos, fprime=self.fprime, gtol=gtol, full_output=True, 
                                   callback=self.postIterationCallback, maxiter=maxiter, disp=False)
        
        except MinimisationTerminated:
            log(__name__, "Minimisation terminated: %s" % sys.exc_info()[1], 2)
            obj.pos = initialPos
            obj.force[:] = 0.0
            obj.totalEnergy = 0.0
            return self.status
        
#         xopt = res[0]
#         fopt = res[1]
#         func_calls = res[2]
#         grad_calls = res[3]
        warnflag = res[4]
        
        if self.status:
            log(__name__, "Minimisation failed (%d)" % (self.status,), 1, 1)
            obj.pos = initialPos
            status = 1
        
        else:
            if warnflag:
                status = 2
                
                if warnflag == 1:
                    log(__name__, "Minimisation terminated due to max iterations reached: %d" % (self.iterationCount,), 1, 1)
                elif warnflag == 2:
                    log(__name__, "Minimisation terminated due to gradient/function calls not changing", 1, 1)
                    status = 0
                else:
                    log(__name__, "Minimisation terminated for unknown reason; warnflag = %d" % (warnflag,), 1, 1)
            
            else:
                log(__name__, "Minimised: total energy change of %f eV, MAG F is %f (%d force evaluations)" % (obj.totalEnergy - self.iniEnergy, 
                                                                                                               Vectors.magnitude(obj.force), 
                                                                                                               self.NForceCalls), 1, 1)
                status = 0
        
        # clear refs
        self.x = None
        self.object = None
        self.initialPos = None
        
        return status

[docs]    def calcForceCG(self, x):
        """
        Do force calculation
        
        """
        self.object.pos = x
        self.x = x
        
        self.status = self.calcForce(self.object, self.excl, self.params)
        
        if self.status:
            log(__name__, "Error: SciPy CG calc force failed (%d)" % self.status, 0, 0)
            raise MinimisationTerminated("Force calc failed (%d)" % self.status)
        
        if self.NForceCalls == 0:
            self.iniEnergy = self.object.totalEnergy
            self.previousEnergy = self.iniEnergy
            log(__name__, "Initial energy is %f" % (self.iniEnergy,), 2, 1)
        
        # check job status
        if self.NForceCalls % 5 == 0:
            if ClientServerInterface.checkJobStatus(self.jobID):
                log(__name__, "SciPy CG stopping due to job status", 1, 0)
                self.status = 33
                raise MinimisationTerminated("Stopping minimisation for job status")
        
        self.NForceCalls += 1

        """
        Function evaluation
        
        """
        sep = 1
        if self.x is not None:
            sep = Vectors.subtractVectors(x, self.x)
            sep = Vectors.magnitude(sep)
        
        if sep == 0:
            pass
        
        else:
            self.calcForceCG(x)
        
        return self.object.totalEnergy

    

[docs]    def fprime(self, x):
        """
        Gradient evaluation
        
        """
        sep = 1
        if self.x is not None:
            sep = Vectors.subtractVectors(x, self.x)
            sep = Vectors.magnitude(sep)
        
        if sep == 0:
            pass
        
        else:
            self.calcForceCG(x)
        
        return -1.0 * self.object.force

        self.status = 0
        self.previousEnergy = None
        self.iterationCount = 0

        self.object = obj

            iprint = 1

            iprint = -1
        
        else:

        # run minimiser
        try:
            x, f, d = optimize.fmin_l_bfgs_b(self.calcForceLBFGS, obj.pos, iprint=iprint,

            return self.status

            obj.LBFGSConvTask = 1

            obj.LBFGSConvTask = 3
            
        elif "EVALUATIONS EXCEEDS LIMIT" in d['task']:
            obj.LBFGSConvTask = 4

            obj.LBFGSConvTask = 0

        if self.status:
            obj.pos = initialPos
            status = 1

                status = 1

                status = 0

            status = 23
            log(__name__, "Minimisation failed due to machine precision stop criteria", 1)

        self.object = None

[docs]    def fallbackMinimisation(self, obj, statusIn=None):
        """
        Fallback if minimisation failed or terminated due to machine 
        precision condition. Either rerun LBFGS or run another 
        method - CG or SD
        
        """
        if self.noFallBack:
            if statusIn is None:
                return self.status
            else:
                return statusIn
        
        # list of minimisers to fall back to
        minimisersStr = self.params.LBFGSFallbackMinimisers
        minimisersList = minimisersStr.split("+")
        minimisersList = [m for m in minimisersList if len(m.strip())]
        
        if not len(minimisersList):
            if statusIn is None:
                return self.status
            else:
                return statusIn
        
        # run them until success
        for count, minimiserName in enumerate(minimisersList):
            log(__name__, "LBFGS: Falling back to '%s' minimiser (%d/%d)" % (minimiserName, count+1, len(minimisersList)), 1)
            
            # parameters
            minParam = copy.deepcopy(self.params)
            minParam.minimizationType = minimiserName
            minParam.minimizationCriteria = "force_element"
            minParam.minimizationTolerance = minParam.LBFGSTolerance
            minParam.LBFGSFallbackMinimisers = ""
            minParam.LBFGSFailOnMachinePrecision = 1
            
            # minimiser
            minimiser = getMinimiser(minParam, jobID=self.jobID)
            
            # run it
            status = minimiser.run(obj)
            
            # stop if success
            if not status:
                break
        
        if not status:
            obj.LBFGSConvTask = 1
        
        return status

        """
        Calculate forces.
        
        This method should be sub-classed if required.
        
        """
        self.object.pos = x
        self.previousEnergy = self.object.totalEnergy
        
        self.status = self.calcForce(self.object, self.excl, self.params)
        
        if self.status:
            log(__name__, "Error: L-BFGS calc force failed (%d)" % self.status, 0, 0)
            raise MinimisationTerminated("Force calc failed (%d)" % self.status)
        
        if self.NForceCalls == 0:
            self.iniEnergy = self.object.totalEnergy
            log(__name__, "Initial energy is %f" % (self.iniEnergy), 2, 1)
        
        # check job status
        if self.NForceCalls % 5 == 0:
            if ClientServerInterface.checkJobStatus(self.jobID):
                log(__name__, "L-BFGS stopping due to job status", 1, 0)
                self.status = 33
                raise MinimisationTerminated("Stopping minimisation for job status")
        
        self.NForceCalls += 1
        
        return (self.object.totalEnergy, self.object.force * -1)
