    def __init__(self, params, debug=False):
        if not LAMMPS_IMPORTED:
            raise LammpsNotInstalled("Could not import the LAMMPS Python wrapper")
        self._lmp = None
        self._storeEnergies = False
        self._debug = debug
        self._params = params
    
    def __del__(self):
        if self._lmp is not None:
            self._lmp.close()
    
    def _initialiseLammps(self, obj, numAtoms, currentPos, velocity=None, screenOutput=False, logOutput=False):
        """
        Initialise the LAMMPS interface.
        
        Parameters
        ----------
        obj : `Lattice.Lattice`-like object
            The object that is going to be used with LAMMPS. Must be like the `Lattice.Lattice` object.
        numAtoms : int
            The number of atoms.
        currentPos : `numpy.ndarray`
            The positions of the atoms.
        velocity : `numpy.ndarray`, optional
            If specified then initialise the atoms with the given velocities.
        screenOutput : `True` or `False`, optional
            If False (the default) then disable LAMMPS screen output. This will be overriden by the `debug` argument to
            the initialiser.
        logOutput : `True` or `False`, optional
            If False (the default) then disable LAMMPS log file output.
        
        Raises
        ------
        LammpsUnitsNotSupported
            If the value of the lammpsUnits parameter is not supported.
        IOError
            If the LAMMPS input file cannot be opened.
        
        """
        # first close the current instance
        if self._lmp is not None:
            self._lmp.close()
        
        # create a new instance
        args = []
        if not screenOutput and not self._debug:
            args.append("-screen")
            args.append("none")
        if not logOutput:
            args.append("-log")
            args.append("none")
        self._lmp = lammps.lammps(cmdargs=args)
        
        # boundary conditions
        pbcx = "p" if Globals.PBC[0] else "f"
        pbcy = "p" if Globals.PBC[1] else "f"
        pbcz = "p" if Globals.PBC[2] else "f"
        
        # check LAMMPS units parameter
        knownValues = ('metal',)
        if Globals.lammpsUnits not in knownValues:
            raise LammpsUnitsNotSupported("LAMMPS units are '%s'; should be one of %r" % (Globals.lammpsUnits,
                                                                                          knownValues))
        
        # initialise
        self._lmp.command("units %s" % Globals.lammpsUnits)
        self._lmp.command("dimension 3")
        self._lmp.command("boundary %s %s %s" % (pbcx, pbcy, pbcz))
        self._lmp.command("atom_style charge")
        self._lmp.command("atom_modify sort 0 0")  # makes PE's come out in right order
        self._lmp.command("atom_modify map array")
        
        # if no AMU info stored, calculate AMU
        if not len(obj.specieMassAMU):
            obj.specieMassAMU = np.empty(len(obj.specieList), np.float64)
            for i in xrange(len(obj.specieList)):
                obj.specieMassAMU[i] = Atoms.atomicMassAMU(obj.specieList[i])
        
        # ensure the specie array is the correct type
        # TODO: is this still necessary?
        obj.specie = np.asarray(obj.specie, dtype=np.int32)
        
        # temporary file for atom data
        with tempfile.NamedTemporaryFile() as tmpf:
            # the name of the temporary file
            fn = tmpf.name
            
            # write the data file
            c_io.writeLammpsAtomDataFile(fn, numAtoms, len(obj.specieList), currentPos, obj.specie, obj.charge,
                                         obj.specieMassAMU, obj.cellDims, 0, velocity)
            
            # tell lammps to read the data
            self._lmp.command("read_data {0}".format(fn))
        
        # we need to be in Globals.MDLocation for this part
        _cwd = os.getcwd()
        os.chdir(Globals.MDLocation)
        try:
            # tell LAMMPS about the potential
            if not os.path.exists(Globals.lammpsInFile):
                raise IOError("Could not locate LAMMPS input file: '%s'" % Globals.lammpsInFile)
            with open(Globals.lammpsInFile) as fh:
                for line in fh:
                    self._lmp.command(line.strip())
        finally:
            os.chdir(_cwd)
        
        # warn about lost atoms
        self._lmp.command("thermo_modify lost warn")
    
    def _checkForAtomsLost(self, numAtoms, currentPos, obj):
        """
        Check if atoms have been lost.
        
        """
        if numAtoms != self._lmp.get_natoms():
            obj.totalEnergy = 0.0
            obj.force[:] = 0.0
            log(__name__, "Lammps calcForce failed due to atoms lost! (%d->%d)" % (numAtoms, self._lmp.get_natoms()),
                2, 2)
            
            lmp_ids = self._lmp.extract_atom("id", 0)
            for atomid in range(0, numAtoms):
                if lmp_ids[atomid] != atomid + 1:
                    log(__name__, "Atom No. %d Lost! (%f %f %f)" % (atomid + 1, currentPos[3 * atomid],
                                                                    currentPos[3 * atomid + 1],
                                                                    currentPos[3 * atomid + 2]), 2, 3)
                    break
            
            return 1
    

[docs]    def thermalise(self, lattice, fullTime, layerTime, temperature):
        """
        Thermalise the given lattice.
        
        """
        # initialse lammps
        self._initialiseLammps(lattice, lattice.NAtoms, lattice.pos, screenOutput=True)
        
        # initialize lattice with velocity at desired temperature
        self._lmp.command("velocity moving create %f %d dist gaussian" % (temperature, random.randint(0, 999999)))
        
        # thermalize whole system
        self._lmp.command("fix 2 moving temp/berendsen {0} {0} 1".format(temperature))
        self._lmp.command("run %d post no" % fullTime)
        
        # then thermalize the thermal region only
        self._lmp.command("unfix 2")
        self._lmp.command("fix 4 thermal temp/berendsen {0} {0} 1".format(temperature))
        self._lmp.command("run %d post no" % layerTime)
        
        # get data back from LAMMPS and store in the lattice
        self.extractData(lattice)
        
        # extract the velocities too
        lmpvel = self._lmp.gather_atoms("v", 1, 3)
        velocity = np.ctypeslib.as_array(lmpvel)
        
        return velocity

    

[docs]    def minimise(self, lattice, initialise=True):
        """
        Run a minimisation using LAMMPS.
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The lattice object to be minimised.
        initialise : `True` or `False`, optional
            If `True` (default) then initialise LAMMPS before running the minimisation. If the lattice has not been
            modified since the last time LAMMPS was called then you do not need to initialise.
        
        """
        log(__name__, "Minimising lattice using LAMMPS", 1, 0)
        
        if initialise:
            # initialse lammps
            self._initialiseLammps(lattice, lattice.NAtoms, lattice.pos)
        
        # prepare to minimise
        params = self._params
        self._lmp.command("min_style %s" % params.depoMinStyle)
        if params.minimizationCriteria == "force" or params.minimizationCriteria == "force_element":
            ftol = params.minimizationTolerance
            etol = 0
        elif params.minimizationCriteria == "energy":
            ftol = 0
            etol = params.minimizationTolerance
        else:
            raise RuntimeError("Unknown minimization criteria: %s" % params.minimizationCriteria)
        maxiter = params.minimisationMaxIterations
        
        # run the minimisation
        self._lmp.command("minimize %f %f %d %d" % (etol, ftol, maxiter, params.depoMinMaxEval))
        
        # get data back from LAMMPS and store in the lattice
        self.extractData(lattice)

    

[docs]    def runMD(self, lattice, velocity, runTime, thermostat):
        """
        Run a molecular dynamics simulation.
        
        """
        log(__name__, "Running LAMMPS MD simulation", 1, 0)
        
        params = self._params
        
        # initialse lammps
        self._initialiseLammps(lattice, lattice.NAtoms, lattice.pos, velocity=velocity, screenOutput=True)
        
        # finish setting up lammps
        # set up compute for temperature of unfixed region
        self._lmp.command("compute temperature moving temp")
        
        # choose the ensemble
        self._lmp.command("fix 0 all nve")
        
        # thermal layer
        if thermostat != "NO_THERMOSTAT":
            self._lmp.command("fix 4 thermal temp/berendsen {0} {0} 1".format(params.systemTemperature))
        
        # thermodynamic info (may change it if not need)
        thermoFreq = 100
        self._lmp.command("thermo %d" % thermoFreq)
        self._lmp.command("thermo_style custom step time c_temperature etotal ke pe")
        self._lmp.command("thermo_modify lost warn")
        # dump file
        dumpFn = "dump*.dat"
        dumpFreq = 250
        self._lmp.command("dump dumpdat all custom %d %s element x y z q" % (dumpFreq, dumpFn))
        elementString = " ".join(lattice.specieList)
        self._lmp.command("dump_modify dumpdat element %s" % elementString)
        
        # are we using a variable time step or not
        if params.lammpsMinDt == params.lammpsMaxDt:
            print "fixed dt..."
            
            # set timestep
            dt = params.lammpsMinDt
            if Globals.lammpsUnits == "metal":
                dt *= 1e-3
            self._lmp.command("timestep %f" % dt)
            
            # run simulation
            numSteps = math.ceil(runTime / params.lammpsMinDt)
            self._lmp.command("run %d post no" % numSteps)
        
        else:
            print "variable dt..."
            minDt = params.lammpsMinDt
            maxDt = params.lammpsMaxDt
            if Globals.lammpsUnits == "metal":
                minDt *= 1e-3
                maxDt *= 1e-3
            self._lmp.command("fix vardt all dt/reset %d %f %f %f units box" % (params.lammpsUpdateDtSteps, minDt,
                                                                                maxDt, params.lammpsMaxAtomMoveDt))
            
            # run simulation
            maxNumSteps = math.ceil(runTime / params.lammpsMinDt)
            runTimeLmp = runTime
            if Globals.lammpsUnits == "metal":
                runTimeLmp *= 1e-3
            self._lmp.command("variable ctim equal time")
            self._lmp.command("run %d every 10 \"if '${ctim} > %f' then 'quit'\"" % (maxNumSteps, runTimeLmp))
        
        # get data back from LAMMPS and store in the lattice
        self.extractData(lattice)

        
        # TODO: write out final lattice incase it wasn't written!?
    

[docs]    def extractData(self, lattice):
        """
        Get data back from LAMMPS and store it.
        
        Parameters
        ----------
        lattice : `Lattice.Lattice`
            The `Lattice` object that the LAMMPS data will be stored in.
        
        Notes
        -----
        We assume that `lattice` is the `Lattice` that was used to run LAMMPS. If the number of atoms in LAMMPS differs
        to the number in `lattice` we remove extra atoms in `lattice`.
        
        """
        # deal with the case where we have lost atoms (LAMMPS removes sputtered atoms completely)
        iniNumAtoms = lattice.NAtoms
        currentNumAtoms = self._lmp.get_natoms()
        numLost = currentNumAtoms - iniNumAtoms
        if currentNumAtoms < iniNumAtoms:
            # atoms were lost
            log(__name__, "%d atom(s) lost during LAMMPS MD simulation" % numLost, 2, 0)
            
            # resize arrays
            lattice.pos.resize(3 * currentNumAtoms, refcheck=False)
            lattice.force.resize(3 * currentNumAtoms, refcheck=False)
            lattice.specie.resize(currentNumAtoms, refcheck=False)
            lattice.charge.resize(currentNumAtoms, refcheck=False)
            
            # copy data
            lattice.pos[:] = self._lmp.gather_atoms("x", 1, 3)[:]
            lattice.force[:] = self._lmp.gather_atoms("f", 1, 3)[:]
            lattice.specie[:] = self._lmp.gather_atoms("type", 0, 1)[:] - 1
            lattice.charge[:] = self._lmp.gather_atoms("q", 1, 1)[:]
            
            # handle species arrays
            for i in xrange(len(lattice.specieList) - 1, -1, -1):
                self._lmp.command("group %s type %d" % (lattice.specieList[i], i + 1))
                self._lmp.command("variable %s equal count(%s)" % (lattice.specieList[i], lattice.specieList[i]))
                lattice.specieCount[i] = int(self._lmp.extract_variable("%s" % lattice.specieList[i], 0, 0))
                if lattice.specieCount[i] == 0 and not lattice.specieListForced:
                    lattice.removeSpecie(i)
            
            # set min/max pos
            for i in xrange(3):
                lattice.minPos[i] = np.min(lattice.pos[i::3])
                lattice.maxPos[i] = np.max(lattice.pos[i::3])
        
        elif iniNumAtoms < currentNumAtoms:
            # this should never happen
            raise RuntimeError("Gained atoms during LAMMPS MD simulation")
        
        else:
            # no atoms lost, just copy data
            lattice.pos[:] = self._lmp.gather_atoms("x", 1, 3)[:]
            lattice.force[:] = self._lmp.gather_atoms("f", 1, 3)[:]
            lattice.charge[:] = self._lmp.gather_atoms("q", 1, 1)[:]
        
        # total energy and force magnitude
        lattice.totalEnergy = self._lmp.extract_compute("thermo_pe", 0, 0)
        lattice.maxForce, lattice.maxForceAtomNo = Vectors.calcForceInfNorm(lattice.force)

    

[docs]    def calcForce(self, obj, image=None, includeAtoms=None, mainIndexes=None):
        """
        Calculate forces on the system.
        
        Parameters
        ----------
        
        
        """
        # current position
        if image is None:
            currentPos = obj.pos
            numAtoms = obj.NAtoms
        else:
            numAtoms = obj.imageNAtoms
            currentPos = obj.pos[3 * image * numAtoms:3 * (image + 1) * numAtoms]
        
        # reset lammps if required
        lmpNone = self._lmp is None
        numAtomsDiffer = numAtoms != self._lmp.get_natoms()
        resetLammps = getattr(obj, "resetLammps", False)
        if lmpNone or numAtomsDiffer or resetLammps:
            self._initialiseLammps(obj, numAtoms, currentPos)
        # or just update positions
        else:
            c_double_p = ctypes.POINTER(ctypes.c_double)
            self._lmp.scatter_atoms("x", 1, 3, currentPos.ctypes.data_as(c_double_p))
        
        # set up compute for atom PE's
        storeEnergies = getattr(obj, "storeEnergies", False)
        if not hasattr(self._lmp, "storeEnergies"):
            self._lmp.storeEnergies = False
        if storeEnergies and not self._lmp.storeEnergies:
            self._lmp.command("compute atompe all pe/atom")
        elif not storeEnergies and self._lmp.storeEnergies:
            self._lmp.command("uncompute atompe")
        self._lmp.storeEnergies = storeEnergies
        
        # run lammps
        self._lmp.command("run 0 post no")
        
        # check for atom lost
        status = self._checkForAtomsLost(numAtoms, currentPos, obj)
        if status:
            return status
        
        # total energy
        obj.totalEnergy = self._lmp.extract_compute("thermo_pe", 0, 0)
        log(__name__, "TOTAL ENERGY: %f" % (obj.totalEnergy,), 4, 2)
        
        # atom pe
        if storeEnergies:
            atompe = self._lmp.extract_compute("atompe", 1, 1)
            lmp_ids = self._lmp.extract_atom("id", 0)
            
            ordered = True
            if lmp_ids[0] != 1:
                print "HANDLING LAMMPS DISORDER"
                ordered = False
            
            # TODO: must be better way of doing this!
            if image is None:
                if ordered:
                    for i in xrange(self._lmp.get_natoms()):
                        obj.PE[i] = atompe[i]
                
                else:
                    for i in xrange(self._lmp.get_natoms()):
                        index = lmp_ids[i] - 1
                        obj.PE[index] = atompe[i]
            
            else:
                raise NotImplementedError("Store energies not implemented for Images (LAMMPS)")
        
        # forces
        lmpforce = self._lmp.gather_atoms("f", 1, 3)
        if image is None:
            # should be a better way than this
            obj.force[:] = lmpforce[:]
        
        else:
            offset = 3 * image * numAtoms
            for i in xrange(len(lmpforce)):
                obj.force[offset + i] = lmpforce[i]
        
        # charges
        lmpcharge = self._lmp.gather_atoms("q", 1, 1)
        obj.charge[:] = lmpcharge[:]
        
        # temporary
        obj.maxForce, obj.maxForceAtomNo = Vectors.calcForceInfNorm(obj.force)
        
        return 0
