        if self.params.timeString:
            return self._runWithTiming(initialState, finalState, saddleState=saddleState, NProcs=NProcs, 
                                       initialEnergy=initialEnergy, initialPos=initialPos)
        else:

        """
        Run without timing.

        return self._runMain(*args, **kwargs)

        """
        Run with timing.

    def _runMain(self, initialState, finalState, saddleState=None, NProcs=1, initialEnergy=None, initialPos=None):

        Main run function (not called directly).
        
        """
        self.NForceCalls = 0

        self.initialEnergy = initialEnergy
        self.initialPos = initialPos
        if self.params.stringFixInitialAndFinal and initialEnergy is None:
            log(__name__, "**** WARNING: string end points are moving and you have not provided initial energy...", 0)
        
        try:
            log(__name__, "Running String method", 1)
            startTime = time.time()
            self.failedReason = None
            self.debug_ymax = None
            
            # interpolate images
            if not self.params.stringFixInitialAndFinal and self.params.stringInterpolateStep:
                interpStep = self.params.stringInterpolateStep
            else:
                interpStep = None
            self.interpolateImages(initialState, finalState, self.params.stringNImages, saddleState=saddleState,

                log(__name__, "Barrier is %f eV" % barrier, 0)

                # check saddle point (force mag, etc...)
                
                
                return 0
        
        finally:
            self.initialEnergy = None

        """

        # calculate force

            raise StringFailed("Initial force calculation failed")

        log(__name__, "Step %d: Total energy: %f" % (0, self.totalEnergy), 2, 1)
        
        # normalise the force
        self.normaliseForce()

        stepCount = 0
        elapsedTime = 0
        energyDiff = params.stringEnergTol + 1
        debugCount = 0

                debugCount += 1

            if stepCount >= params.stringMaxSteps:
                self.failedReason = "String: max steps exceeded"
                log(__name__, "Failure: max steps exceeded (%d >= %d)" % (stepCount, params.stringMaxSteps), 0)
                raise StringFailed("Max steps exceeded")

                self.failedReason = "String: max time exceeded"
                log(__name__, "Failure: max time exceeded (%f >= %f)" % (timeElapsed, params.stringMaxRunTimeSecs), 0)
                raise StringFailed("Max time exceeded")
            
            elif math.fabs(energyDiff) <= params.stringEnergTol:
                log(__name__, "Success: energy tolerance satisfied", 1, 1)
                break
            
#            elif forceMagnitude <= 0.1:
#                log(__name__, "Success: force tolerance satisfied", 1, 1)
#                break
            
            # increment step count
            stepCount += 1
            
            # store previous energy
            previousEnergy = self.totalEnergy
            
            # checking if we can do the climbing image
            self.climb = False
            if params.stringClimbCriteria == "iterations":
                if stepCount >= params.stringClimbImageValue:
                    self.climb = True

            elapsedTime = time.time() - startTime

        """
        Move the images:
        
            newPos = oldPos + F * deltaR
        
        """
        if self.climb:
            log(__name__, "Climbing", 3, 2)
            tauVector = self.calculateTangentVector(self.maxEnergyImage)
            c_neb.updateForceClimbingImageNEB(self.maxEnergyImage, 3 * self.imageNAtoms, self.force, tauVector)
        
        self.normaliseForce()
        self.pos += self.params.stringMinStepSize * self.force

        """
        Reinterpolate images (linear for now).
        
        For now I only care about free end points case...
        
        """
        if self.fixedEndPoints:
            raise NotImplementedError("Not implemented String reinterpolation for fixed end points yet")
        
        log(__name__, "Reinterpolating String images (linear)...", 3)
        
        # length of string
        currentLength = self.getLength()
        originalLength = self.initialLength
        deltaLength = originalLength - currentLength
        log(__name__, "Current length %f; Ideal length %f (delta %f)" % (currentLength, originalLength, deltaLength), 3, 1)
        
        # spacing between images
        spacing = self.idealImageSpacing
        
        # interpolate around highest point
        if self.climb or True:
            log(__name__, "Reinterpolating around highest point (climbing...)", 3, 1)
            
            # max energy image
            maxEnergyPos = self.getImagePos(self.maxEnergyImage)
            log(__name__, "Max energy image is: %d" % self.maxEnergyImage, 3, 1)
            
            # reinterpolate images to the right of the max
            vecs = []
            seps = []
            for i in xrange(self.maxEnergyImage + 1, self.NImagesTotal):
                sepVec = self.imageSeparationVector(i - 1, i)
                seps.append(Vectors.magnitude(sepVec))
                vecs.append(Vectors.normalise(sepVec))
            
            count = 1
            for i in xrange(self.maxEnergyImage + 1, self.NImagesTotal):
                # how far to move this image from maxEnergyImage
                currentImageSep = count * spacing
                log(__name__, "Image %d is %f from max energy image" % (i, currentImageSep), 3, 1)
                
                # position of this image
                imagePos = self.getImagePos(i)
                imagePos[:] = copy.deepcopy(maxEnergyPos)
                
                # loop over vectors until we have moved far enough
                sepFromMax = 0.0
                index = 0
                while True:
                    # length of this join
                    joinLength = seps[index]
                    joinVector = vecs[index]
                    
                    # do we stop somewhere along this join
                    if sepFromMax + joinLength >= currentImageSep or index == len(seps) - 1:
                        # how far to move along this one
                        step = currentImageSep - sepFromMax
                        assert step <= joinLength or index == len(seps) - 1
                        
                        # take the step
                        sepFromMax += step
                        log(__name__, "Partial step %f along join %d (sep from ini %f)" % (step, index, sepFromMax), 3, 2)
                        imagePos += step * joinVector
                        
                        # break out of loop (we have moved far enough)
                        break
                    
                    # we move all the way along this join and keep going
                    else:
                        sepFromMax += joinLength
                        log(__name__, "Full step %f along join %d (sep from ini %f)" % (joinLength, index, sepFromMax), 3, 2)
                        imagePos += joinLength * joinVector
                    
                    index += 1
                
                count += 1
            
            # reinterpolate images to the left of the max
            vecs = []
            seps = []
            for i in xrange(self.maxEnergyImage - 1, -1, -1):
                sepVec = self.imageSeparationVector(i, i + 1) * -1.0
                seps.append(Vectors.magnitude(sepVec))
                vecs.append(Vectors.normalise(sepVec))
            
            count = 1
            for i in xrange(self.maxEnergyImage - 1, -1, -1):
                # how far to move this image from maxEnergyImage
                currentImageSep = count * spacing
                log(__name__, "Image %d is %f from max energy image" % (i, currentImageSep), 3, 1)
                
                # position of this image
                imagePos = self.getImagePos(i)
                imagePos[:] = copy.deepcopy(maxEnergyPos)

                    joinVector = vecs[index]
                    
                    # do we stop somewhere along this join
                    if sepFromMax + joinLength >= currentImageSep or index == len(seps) - 1:
                        # how far to move along this one
                        step = currentImageSep - sepFromMax

                        sepFromMax += step

                        imagePos += step * joinVector
                        
                        # break out of loop (we have moved far enough)
                        break
                    
                    # we move all the way along this join and keep going
                    else:
                        sepFromMax += joinLength
                        log(__name__, "Full step %f along join %d (sep from ini %f)" % (joinLength, index, sepFromMax), 3, 2)
                        imagePos += joinLength * joinVector
                    
                    index += 1
                
                count += 1
        
        # interpolate around centre point
        else:
            log(__name__, "Reinterpolating around centre point", 2, 1)
            
            # get the vectors between the current set of images
            stringVectors = []
            stringSeps = []
            for i in xrange(1, self.NImagesTotal):
                sepVec = self.imageSeparationVector(i - 1, i)
                stringSeps.append(Vectors.magnitude(sepVec))
                stringVectors.append(Vectors.normalise(sepVec))
             
            # copy the current initial position
            currentInitialPos = copy.deepcopy(self.getImagePos(0))
             
             
             
            # loop over the images, setting their new positions
            #NOTE: this only works for fixedEndPoints=False at the moment
            for i in xrange(self.NImages):
                # work out how far to move from the current initial
                currentImageSep = deltaLength / 2.0 + i * spacing
                log(__name__, "Image %d is %f from old initial" % (i, currentImageSep), 2, 1)
                 
                # position of this image
                imagePos = self.getImagePos(i)
                imagePos[:] = copy.deepcopy(currentInitialPos)
                 
                # loop over string vectors until we have moved far enough
                sepFromIni = 0.0
                index = 0
                while True:
                    # length of this join
                    joinLength = stringSeps[index]

        """
        Reinterpolate the String images
        
        TODO: this needs sorting out... string should remain a fixed length,
        i.e. the interimage separation should always remain the same! At the
        moment the string grows in length.
        
        """
        length = 0.0
        stringVectors = []
        numberOfJoins = self.NImages - 1
        
        # get the length of the curve and separate vectors
        for i in xrange(numberOfJoins):
            sep = self.imageSeparationVector(i, i + 1)
            length += Vectors.magnitude(sep)
            stringVectors.append(sep)
        
        avgJoinLen = length / numberOfJoins
        
        if not self.fixedEndPoints:
            numberOfJoinsX = numberOfJoins + 1
        else:
            numberOfJoinsX = numberOfJoins
        
        for i in xrange(1, numberOfJoins):
            offset = i * avgJoinLen
            
            # Copying initial lattice atoms positions
            if not self.fixedEndPoints:
                self.pos[(i*self.imageNAtoms*3):((i+1)*self.imageNAtoms*3)] = copy.copy(self.pos[0:self.imageNAtoms*3])
            else:
                self.pos[((i-1)*self.imageNAtoms*3):(i*self.imageNAtoms*3)] = copy.copy(self.initialState.pos)
            
            joinNumber = 0
            sumLen = 0.0
            while sumLen < offset and joinNumber < numberOfJoins:
                joinVectorsMag = Vectors.magnitude((stringVectors[joinNumber]))
                
                if joinVectorsMag + sumLen > offset:
                    joinLenTemp = offset - sumLen
                    
                    normalizedVector = Vectors.normalise(stringVectors[joinNumber])
                    
                    if not self.fixedEndPoints:
                        Vectors.addVectorsInplace(self.pos[(i*self.imageNAtoms*3):((i + 1)*self.imageNAtoms*3)], joinLenTemp * normalizedVector)
                    else:
                        Vectors.addVectorsInplace(self.pos[((i-1)*self.imageNAtoms*3):(i*self.imageNAtoms*3)], joinLenTemp * normalizedVector)
                    
                    sumLen += joinLenTemp
                
                else:
                    if not self.fixedEndPoints:
                        Vectors.addVectorsInplace(self.pos[(i*self.imageNAtoms*3):((i+1)*self.imageNAtoms*3)], stringVectors[joinNumber])
                    else:
                        Vectors.addVectorsInplace(self.pos[((i-1)*self.imageNAtoms*3):(i*self.imageNAtoms*3)], stringVectors[joinNumber])
                    
                    sumLen += joinVectorsMag
                
                joinNumber += 1

[docs]def main():
    # program arguments
    progargs = commandLineArgs()
    
    # input parameters
    params = Input.getLKMCParams(inputFile=progargs.paramFile)
    Input.readGlobals(inputParams=params)
    
    # read initial state and calculate energy
    initialState = Lattice.readLattice(progargs.initialFile)
    status = initialState.calcForce()
    if status:
        raise RuntimeError("Initial state force calculation failed")
    
    # read final state
    finalState = Lattice.readLattice(progargs.finalFile)
    
    # read saddle state
    saddleState = Lattice.readLattice(progargs.saddleFile) if progargs.saddleFile is not None else None
    
    # run String method
    string = String(params, numProcs=progargs.NProcs)
    status = string.run(initialState, finalState, saddleState=saddleState, initialEnergy=initialState.totalEnergy,
                        initialPos=initialState.pos)
    
    if not status:
        # output result
        if params.stringPlotPathway:
            string.plotBarrierSeparation("string_final", label="String", useInterp=1)
        if params.stringVisualisePathway:
            string.visualisePathwayLattice()
